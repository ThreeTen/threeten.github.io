32  package org.threeten.extra;
33  
34  import java.io.Serializable;
35  import java.time.DateTimeException;
36  import java.time.Duration;
37  import java.time.Instant;
38  import java.time.format.DateTimeParseException;
39  import java.util.Objects;
40  
41  /**
42   * An immutable interval of time between two instants.
43   * <p>
44   * An interval represents the time on the time-line between two {@link Instant}s.
45   * The class stores the start and end instants, with the start inclusive and the end exclusive.
46   * The end instant is always greater than or equal to the start instant.
47   * <p>
48   * The {@link Duration} of an interval can be obtained, but is a separate concept.
49   * An interval is connected to the time-line, whereas a duration is not.
50   * <p>
51   * Intervals are not comparable. To compare the length of two intervals, it is
52   * generally recommended to compare their durations.
53   *
54   * <h3>Implementation Requirements:</h3>
55   * This class is immutable and thread-safe.
56   * <p>
57   * This class must be treated as a value type. Do not synchronize, rely on the
58   * identity hash code or use the distinction between equals() and ==.
59   */
60  public final class Interval
61          implements Serializable {
62  
63      /**
64       * Serialization version.
65       */
66      private static final long serialVersionUID = 8375285238652L;
67  
68      /**
69       * The start instant (inclusive).
70       */
71      private final Instant start;
72      /**
73       * The end instant (exclusive).
74       */
75      private final Instant end;
76  
77      //-----------------------------------------------------------------------
78      /**
79       * Obtains an instance of {@code Interval} from the start and end instant.
80       * <p>
81       * The end instant must not be before the start instant.
82       *
83       * @param startInclusive  the start instant, inclusive, not null
84       * @param endExclusive  the end instant, exclusive, not null
85       * @return the interval, not null
86       * @throws DateTimeException if the end is before the start
87       */
88      public static Interval of(Instant startInclusive, Instant endExclusive) {
89          Objects.requireNonNull(startInclusive, "startInclusive");
90          Objects.requireNonNull(endExclusive, "endExclusive");
91          if (endExclusive.isBefore(startInclusive)) {
92              throw new DateTimeException("Start instant must be before end instant");
93          }
94          return new Interval(startInclusive, endExclusive);
95      }
96  
97      /**
98       * Obtains an instance of {@code Interval} from the start and a duration.
99       * <p>
100      * The end instant is calculated as the start plus the duration.
101      * The duration must not be negative.
102      *
103      * @param startInclusive  the start instant, inclusive, not null
104      * @param duration  the duration from the start to the end, not null
105      * @return the interval, not null
106      * @throws DateTimeException if the end is before the start,
107      *  or if the duration addition cannot be made
108      * @throws ArithmeticException if numeric overflow occurs when adding the duration
109      */
110     public static Interval of(Instant startInclusive, Duration duration) {
111         Objects.requireNonNull(startInclusive, "startInclusive");
112         Objects.requireNonNull(duration, "duration");
113         if (duration.isNegative()) {
114             throw new DateTimeException("Duration must not be zero or negative");
115         }
116         return new Interval(startInclusive, startInclusive.plus(duration));
117     }
118 
119     //-----------------------------------------------------------------------
120     /**
121      * Obtains an instance of {@code Interval} from a text string such as
122      * {@code 2007-12-03T10:15:30Z/2007-12-04T10:15:30Z}.
123      * <p>
124      * The string must consist of one of the following three formats:
125      * <ul>
126      * <li>a representations of an {@link Instant}, followed by a forward slash, followed by a representation of a {@link Instant}
127      * <li>a representation of an {@link Instant}, followed by a forward slash, followed by a representation of a {@link Duration}
128      * <li>a representation of a {@link Duration}, followed by a forward slash, followed by a representation of an {@link Instant}
129      * </ul>
130      * 
131      *
132      * @param text  the text to parse, not null
133      * @return the parsed interval, not null
134      * @throws DateTimeParseException if the text cannot be parsed
135      */
136     public static Interval parse(CharSequence text) {
137         Objects.requireNonNull(text, "text");
138         for (int i = 0; i < text.length(); i++) {
139             if (text.charAt(i) == '/') {
140                 char firstChar = text.charAt(0);
141                 if (firstChar == 'P' || firstChar == 'p') {
142                     // duration followed by instant
143                     Duration duration = Duration.parse(text.subSequence(0, i));
144                     Instant end = Instant.parse(text.subSequence(i + 1, text.length()));
145                     return Interval.of(end.minus(duration), end);
146                 } else {
147                     // instant followed by instant or duration
148                     Instant start = Instant.parse(text.subSequence(0, i));
149                     if (i + 1 < text.length()) {
150                         char c = text.charAt(i + 1);
151                         if (c == 'P' || c == 'p') {
152                             Duration duration = Duration.parse(text.subSequence(i + 1, text.length()));
153                             return Interval.of(start, start.plus(duration));
154                         }
155                     }
156                     Instant end = Instant.parse(text.subSequence(i + 1, text.length()));
157                     return Interval.of(start, end);
158                 }
159             }
160         }
161         throw new DateTimeParseException("Interval cannot be parsed, no forward slash found", text, 0);
162     }
163 
164     //-----------------------------------------------------------------------
165     /**
166      * Constructor.
167      *
168      * @param startInclusive  the start instant, inclusive, validated not null
169      * @param endExclusive  the end instant, exclusive, validated not null
170      */
171     private Interval(Instant startInclusive, Instant endExclusive) {
172         this.start = startInclusive;
173         this.end = endExclusive;
174     }
175 
176     //-----------------------------------------------------------------------
177     /**
178      * Gets the start of this time interval, inclusive.
179      *
180      * @return the start of the time interval
181      */
182     public Instant getStart() {
183         return start;
184     }
185 
186     /** 
187      * Gets the end of this time interval, exclusive.
188      *
189      * @return the end of the time interval
190      */
191     public Instant getEnd() {
192         return end;
193     }
194 
195     //-----------------------------------------------------------------------
196     /**
197      * Creates a new interval with the specified start instant.
198      *
199      * @param start  the start instant for the new interval, not null
200      * @return an interval with the end from this interval and the specified start
201      * @throws IllegalArgumentException if the resulting interval has end before start
202      */
203     public Interval withStart(Instant start) {
204         return Interval.of(start, end);
205     }
206 
207     /**
208      * Creates a new interval with the specified end instant.
209      *
210      * @param end  the end instant for the new interval, not null
211      * @return an interval with the start from this interval and the specified end
212      * @throws IllegalArgumentException if the resulting interval has end before start
213      */
214     public Interval withEnd(Instant end) {
215         return Interval.of(start, end);
216     }
217 
218     //-----------------------------------------------------------------------
219     /**
220      * Checks if this interval contains the specified instant.
221      * <p>
222      * The result is true if the instant is equal or after the start and before the end.
223      * An empty interval does not contain anything.
224      *
225      * @param instant  the instant, not null
226      * @return true if this interval contains the instant
227      */
228     public boolean contains(Instant instant) {
229         Objects.requireNonNull(instant, "instant");
230         return (instant.compareTo(start) >= 0 && instant.compareTo(end) < 0);
231     }
232 
233     /**
234      * Checks if this interval encloses the specified interval.
235      * <p>
236      * This checks if the specified interval is fully enclosed by this interval.
237      * The result is true if the start of the specified interval is contained in this interval,
238      * and the end is contained or equal to the end of this interval.
239      * An empty interval contains an equal empty interval, but no other intervals.
240      *
241      * @param interval  the interval, not null
242      * @return true if this interval contains the other interval
243      */
244     public boolean encloses(Interval interval) {
245         Objects.requireNonNull(interval, "interval");
246         return start.compareTo(interval.start) <= 0 &&
247                 interval.end.compareTo(end) <= 0;
248     }
249 
250     /**
251      * Checks if this interval overlaps the specified interval.
252      * <p>
253      * The result is true if the the two intervals share some part of the time-line.
254      * An empty interval overlaps an equal empty interval.
255      *
256      * @param interval  the time interval to compare to, null means a zero length interval now
257      * @return true if the time intervals overlap
258      */
259     public boolean overlaps(Interval interval) {
260         Objects.requireNonNull(interval, "interval");
261         return interval.equals(this) ||
262                 (start.compareTo(interval.end) < 0 && interval.start.compareTo(end) < 0);
263     }
264 
265     /**
266      * Checks if this interval abuts the specified interval.
267      * <p>
268      * The result is true if the the two intervals have one instant in common.
269      * An empty interval does not abut an equal empty interval.
270      *
271      * @param interval  the interval, not null
272      * @return true if this interval abuts the other interval
273      */
274     public boolean abuts(Interval interval) {
275         Objects.requireNonNull(interval, "interval");
276         return end.equals(interval.start) ^ start.equals(interval.end);
277     }
278 
279     //-------------------------------------------------------------------------
280     /**
281      * Checks if this interval is after the specified instant.
282      * <p>
283      * The result is true if the this instant starts after the specified instant.
284      * An empty interval behaves as though it is an instant for comparison purposes.
285      *
286      * @param instant  the other instant to compare to, not null
287      * @return true if the start of this interval is after the specified instant
288      */
289     public boolean isAfter(Instant instant) {
290         return start.compareTo(instant) > 0;
291     }
292 
293     /**
294      * Checks if this interval is before the specified instant.
295      * <p>
296      * The result is true if the this instant ends before the specified instant.
297      * Since intervals do not include their end points, this will return true if the
298      * instant equals the end of the interval.
299      * An empty interval behaves as though it is an instant for comparison purposes.
300      *
301      * @param instant  the other instant to compare to, not null
302      * @return true if the start of this interval is before the specified instant
303      */
304     public boolean isBefore(Instant instant) {
305         return end.compareTo(instant) <= 0 && start.compareTo(instant) < 0;
306     }
307 
308     //-------------------------------------------------------------------------
309     /**
310      * Checks if this interval is after the specified interval.
311      * <p>
312      * The result is true if the this instant starts after the end of the specified interval.
313      * Since intervals do not include their end points, this will return true if the
314      * instant equals the end of the interval.
315      * An empty interval behaves as though it is an instant for comparison purposes.
316      *
317      * @param interval  the other interval to compare to, not null
318      * @return true if this instant is after the specified instant
319      */
320     public boolean isAfter(Interval interval) {
321         return start.compareTo(interval.end) >= 0 && !interval.equals(this);
322     }
323 
324     /**
325      * Checks if this interval is before the specified interval.
326      * <p>
327      * The result is true if the this instant ends before the start of the specified interval.
328      * Since intervals do not include their end points, this will return true if the
329      * two intervals abut.
330      * An empty interval behaves as though it is an instant for comparison purposes.
331      *
332      * @param interval  the other interval to compare to, not null
333      * @return true if this instant is before the specified instant
334      */
335     public boolean isBefore(Interval interval) {
336         return end.compareTo(interval.start) <= 0 && !interval.equals(this);
337     }
338 
339     //-----------------------------------------------------------------------
340     /**
341      * Obtains the duration of this interval.
342      * <p>
343      * An {@code Interval} is associated with two specific instants on the time-line.
344      * A {@code Duration} is simply an amount of time, separate from the time-line.
345      *
346      * @return the duration of the time interval
347      * @throws ArithmeticException if the calculation exceeds the capacity of {@code Duration}
348      */
349     public Duration toDuration() {
350         return Duration.between(start, end);
351     }
352 
353     //-----------------------------------------------------------------------
354     /**
355      * Checks if this interval is equal to another interval.
356      * <p>
357      * Compares this {@code Interval} with another ensuring that the two instants are the same.
358      * Only objects of type {@code Interval} are compared, other types return false.
359      *
360      * @param obj  the object to check, null returns false
361      * @return true if this is equal to the other interval
362      */
363     @Override
364     public boolean equals(Object obj) {
365         if (this == obj) {
366             return true;
367         }
368         if (obj instanceof Interval) {
369             Interval other = (Interval) obj;
370             return start.equals(other.start) && end.equals(other.end);
371         }
372         return false;
373     }
374 
375     /**
376      * A hash code for this interval.
377      *
378      * @return a suitable hash code
379      */
380     @Override
381     public int hashCode() {
382         return start.hashCode() ^ end.hashCode();
383     }
384 
385     //-----------------------------------------------------------------------
386     /**
387      * Outputs this interval as a {@code String}, such as {@code 2007-12-03T10:15:30/2007-12-04T10:15:30}.
388      * <p>
389      * The output will be the ISO-8601 format formed by combining the
390      * {@code toString()} methods of the two instants, separated by a forward slash.
391      *
392      * @return a string representation of this instant, not null
393      */
394     @Override
395     public String toString() {
396         return start.toString() + '/' + end.toString();
397     }
