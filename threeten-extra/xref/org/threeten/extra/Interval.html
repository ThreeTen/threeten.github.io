45  /**
46   * An immutable interval of time between two instants.
47   * <p>
48   * An interval represents the time on the time-line between two {@link Instant}s.
49   * The class stores the start and end instants, with the start inclusive and the end exclusive.
50   * The end instant is always greater than or equal to the start instant.
51   * <p>
52   * The {@link Duration} of an interval can be obtained, but is a separate concept.
53   * An interval is connected to the time-line, whereas a duration is not.
54   * <p>
55   * Intervals are not comparable. To compare the length of two intervals, it is
56   * generally recommended to compare their durations.
57   *
58   * <h3>Implementation Requirements:</h3>
59   * This class is immutable and thread-safe.
60   * <p>
61   * This class must be treated as a value type. Do not synchronize, rely on the
62   * identity hash code or use the distinction between equals() and ==.
63   */
64  public final class Interval
65          implements Serializable {
66  
67      /**
68       * An interval over the whole time-line.
69       */
70      public static final Interval ALL = new Interval(Instant.MIN, Instant.MAX);

72      /**
73       * Serialization version.
74       */
75      private static final long serialVersionUID = 8375285238652L;
76  
77      /**
78       * The start instant (inclusive).
79       */
80      private final Instant start;
81      /**
82       * The end instant (exclusive).
83       */
84      private final Instant end;

86      //-----------------------------------------------------------------------
87      /**
88       * Obtains an instance of {@code Interval} from the start and end instant.
89       * <p>
90       * The end instant must not be before the start instant.
91       *
92       * @param startInclusive  the start instant, inclusive, MIN_DATE treated as unbounded, not null
93       * @param endExclusive  the end instant, exclusive, MAX_DATE treated as unbounded, not null
94       * @return the half-open interval, not null
95       * @throws DateTimeException if the end is before the start
96       */
97      public static Interval of(Instant startInclusive, Instant endExclusive) {
98          Objects.requireNonNull(startInclusive, "startInclusive");
99          Objects.requireNonNull(endExclusive, "endExclusive");
100         if (endExclusive.isBefore(startInclusive)) {
101             throw new DateTimeException("End instant must on or after start instant");
102         }
103         return new Interval(startInclusive, endExclusive);
104     }
105 
106     /**
107      * Obtains an instance of {@code Interval} from the start and a duration.
108      * <p>
109      * The end instant is calculated as the start plus the duration.
110      * The duration must not be negative.
111      *
112      * @param startInclusive  the start instant, inclusive, not null
113      * @param duration  the duration from the start to the end, not null
114      * @return the interval, not null
115      * @throws DateTimeException if the end is before the start,
116      *  or if the duration addition cannot be made
117      * @throws ArithmeticException if numeric overflow occurs when adding the duration
118      */
119     public static Interval of(Instant startInclusive, Duration duration) {
120         Objects.requireNonNull(startInclusive, "startInclusive");
121         Objects.requireNonNull(duration, "duration");
122         if (duration.isNegative()) {
123             throw new DateTimeException("Duration must not be zero or negative");
124         }
125         return new Interval(startInclusive, startInclusive.plus(duration));
126     }

128     //-----------------------------------------------------------------------
129     /**
130      * Obtains an instance of {@code Interval} from a text string such as
131      * {@code 2007-12-03T10:15:30Z/2007-12-04T10:15:30Z}, where the end instant is exclusive.
132      * <p>
133      * The string must consist of one of the following three formats:
134      * <ul>
135      * <li>a representations of an {@link OffsetDateTime}, followed by a forward slash,
136      *  followed by a representation of a {@link OffsetDateTime}
137      * <li>a representations of an {@link OffsetDateTime}, followed by a forward slash,
138      *  followed by a representation of a {@link LocalDateTime}, where the end offset is implied.
139      * <li>a representation of an {@link OffsetDateTime}, followed by a forward slash,
140      *  followed by a representation of a {@link PeriodDuration}
141      * <li>a representation of a {@link PeriodDuration}, followed by a forward slash,
142      *  followed by a representation of an {@link OffsetDateTime}
143      * </ul>
144      * <p>
145      * ISO-8601 supports a very wide range of possible inputs, many of which are not supported here.
146      * For example, basic format, week-based dates, ordinal dates and date-style period formats are not supported.
147      *
148      * @param text  the text to parse, not null
149      * @return the parsed interval, not null
150      * @throws DateTimeParseException if the text cannot be parsed
151      */
152     public static Interval parse(CharSequence text) {
153         Objects.requireNonNull(text, "text");
154         for (int i = 0; i < text.length(); i++) {
155             if (text.charAt(i) == '/') {
156                 char firstChar = text.charAt(0);
157                 if (firstChar == 'P' || firstChar == 'p') {
158                     // duration followed by instant
159                     PeriodDuration amount = PeriodDuration.parse(text.subSequence(0, i));
160                     OffsetDateTime end = OffsetDateTime.parse(text.subSequence(i + 1, text.length()));
161                     return Interval.of(end.minus(amount).toInstant(), end.toInstant());
162                 } else {
163                     // instant followed by instant or duration
164                     OffsetDateTime start = OffsetDateTime.parse(text.subSequence(0, i));
165                     if (i + 1 < text.length()) {
166                         char c = text.charAt(i + 1);
167                         if (c == 'P' || c == 'p') {
168                             PeriodDuration amount = PeriodDuration.parse(text.subSequence(i + 1, text.length()));
169                             return Interval.of(start.toInstant(), start.plus(amount).toInstant());
170                         }
171                     }
172                     // infer offset from start if not specified by end
173                     String remStr = text.subSequence(i + 1, text.length()).toString();
174                     TemporalAccessor temporal = DateTimeFormatter.ISO_DATE_TIME.parseBest(remStr, OffsetDateTime::from, LocalDateTime::from);
175                     if (temporal instanceof OffsetDateTime) {
176                         OffsetDateTime odt = (OffsetDateTime) temporal;
177                         return Interval.of(start.toInstant(), odt.toInstant());
178                     } else {
179                         LocalDateTime ldt = (LocalDateTime) temporal;
180                         return Interval.of(start.toInstant(), ldt.toInstant(start.getOffset()));
181                     }
182                 }
183             }
184         }
185         throw new DateTimeParseException("Interval cannot be parsed, no forward slash found", text, 0);
186     }

189     /**
190      * Constructor.
191      *
192      * @param startInclusive  the start instant, inclusive, validated not null
193      * @param endExclusive  the end instant, exclusive, validated not null
194      */
195     private Interval(Instant startInclusive, Instant endExclusive) {
196         this.start = startInclusive;
197         this.end = endExclusive;
198     }

201     /**
202      * Gets the start of this time interval, inclusive.
203      * <p>
204      * This will return {@link Instant#MIN} if the range is unbounded at the start.
205      * In this case, the range includes all dates into the far-past.
206      *
207      * @return the start of the time interval
208      */
209     public Instant getStart() {
210         return start;
211     }
212 
213     /** 
214      * Gets the end of this time interval, exclusive.
215      * <p>
216      * This will return {@link Instant#MAX} if the range is unbounded at the end.
217      * In this case, the range includes all dates into the far-future.
218      *
219      * @return the end of the time interval, exclusive
220      */
221     public Instant getEnd() {
222         return end;
223     }

226     /**
227      * Checks if the range is empty.
228      * <p>
229      * An empty range occurs when the start date equals the inclusive end date.
230      * 
231      * @return true if the range is empty
232      */
233     public boolean isEmpty() {
234         return start.equals(end);
235     }

237     /**
238      * Checks if the start of the interval is unbounded.
239      * 
240      * @return true if start is unbounded
241      */
242     public boolean isUnboundedStart() {
243         return start.equals(Instant.MIN);
244     }
245 
246     /**
247      * Checks if the end of the interval is unbounded.
248      * 
249      * @return true if end is unbounded
250      */
251     public boolean isUnboundedEnd() {
252         return end.equals(Instant.MAX);
253     }

256     /**
257      * Returns a copy of this range with the specified start instant.
258      *
259      * @param start  the start instant for the new interval, not null
260      * @return an interval with the end from this interval and the specified start
261      * @throws DateTimeException if the resulting interval has end before start

263     public Interval withStart(Instant start) {
264         return Interval.of(start, end);
265     }

267     /**
268      * Returns a copy of this range with the specified end instant.
269      *
270      * @param end  the end instant for the new interval, not null
271      * @return an interval with the start from this interval and the specified end
272      * @throws DateTimeException if the resulting interval has end before start
273      */
274     public Interval withEnd(Instant end) {
275         return Interval.of(start, end);
276     }

278     //-----------------------------------------------------------------------
279     /**
280      * Checks if this interval contains the specified instant.
281      * <p>
282      * This checks if the specified instant is within the bounds of this interval.
283      * If this range has an unbounded start then {@code contains(Instant#MIN)} returns true.
284      * If this range has an unbounded end then {@code contains(Instant#MAX)} returns true.
285      * If this range is empty then this method always returns false.
286      *
287      * @param instant  the instant, not null
288      * @return true if this interval contains the instant
289      */
290     public boolean contains(Instant instant) {
291         Objects.requireNonNull(instant, "instant");
292         return start.compareTo(instant) <= 0 && (instant.compareTo(end) < 0 || isUnboundedEnd());
293     }
294 
295     /**
296      * Checks if this interval encloses the specified interval.
297      * <p>
298      * This checks if the bounds of the specified interval are within the bounds of this interval.
299      * An empty interval encloses itself.
300      *
301      * @param other  the other interval, not null
302      * @return true if this interval contains the other interval
303      */
304     public boolean encloses(Interval other) {
305         Objects.requireNonNull(other, "other");
306         return start.compareTo(other.start) <= 0 && other.end.compareTo(end) <= 0;
307     }
308 
309     /**
310      * Checks if this interval abuts the specified interval.
311      * <p>
312      * The result is true if the end of this interval is the start of the other, or vice versa.
313      * An empty interval does not abut itself.
314      *
315      * @param other  the other interval, not null
316      * @return true if this interval abuts the other interval
317      */
318     public boolean abuts(Interval other) {
319         Objects.requireNonNull(other, "other");
320         return end.equals(other.start) ^ start.equals(other.end);
321     }
322 
323     /**
324      * Checks if this interval is connected to the specified interval.
325      * <p>
326      * The result is true if the two intervals have an enclosed interval in common, even if that interval is empty.
327      * An empty interval is connected to itself.
328      * <p>
329      * This is equivalent to {@code (overlaps(other) || abuts(other))}.
330      *
331      * @param other  the other interval, not null
332      * @return true if this interval is connected to the other interval
333      */
334     public boolean isConnected(Interval other) {
335         Objects.requireNonNull(other, "other");
336         return this.equals(other) || (start.compareTo(other.end) <= 0 && other.start.compareTo(end) <= 0);
337     }
338 
339     /**
340      * Checks if this interval overlaps the specified interval.
341      * <p>
342      * The result is true if the the two intervals share some part of the time-line.
343      * An empty interval overlaps itself.
344      * <p>
345      * This is equivalent to {@code (isConnected(other) && !abuts(other))}.
346      *
347      * @param other  the time interval to compare to, null means a zero length interval now
348      * @return true if the time intervals overlap
349      */
350     public boolean overlaps(Interval other) {
351         Objects.requireNonNull(other, "other");
352         return other.equals(this) || (start.compareTo(other.end) < 0 && other.start.compareTo(end) < 0);
353     }
354 
355     //-----------------------------------------------------------------------
356     /**
357      * Calculates the interval that is the intersection of this interval and the specified interval.
358      * <p>
359      * This finds the intersection of two intervals.
360      * This throws an exception if the two intervals are not {@linkplain #isConnected(Interval) connected}.
361      * 
362      * @param other  the other interval to check for, not null
363      * @return the interval that is the intersection of the two intervals
364      * @throws DateTimeException if the intervals do not connect
365      */
366     public Interval intersection(Interval other) {
367         Objects.requireNonNull(other, "other");
368         if (isConnected(other) == false) {
369             throw new DateTimeException("Intervals do not connect: " + this + " and " + other);
370         }
371         int cmpStart = start.compareTo(other.start);
372         int cmpEnd = end.compareTo(other.end);
373         if (cmpStart >= 0 && cmpEnd <= 0) {
374             return this;
375         } else if (cmpStart <= 0 && cmpEnd >= 0) {
376             return other;
377         } else {
378             Instant newStart = (cmpStart >= 0 ? start : other.start);
379             Instant newEnd = (cmpEnd <= 0 ? end : other.end);
380             return Interval.of(newStart, newEnd);
381         }
382     }
383 
384     /**
385      * Calculates the interval that is the union of this interval and the specified interval.
386      * <p>
387      * This finds the union of two intervals.
388      * This throws an exception if the two intervals are not {@linkplain #isConnected(Interval) connected}.
389      * 
390      * @param other  the other interval to check for, not null
391      * @return the interval that is the union of the two intervals
392      * @throws DateTimeException if the intervals do not connect
393      */
394     public Interval union(Interval other) {
395         Objects.requireNonNull(other, "other");
396         if (isConnected(other) == false) {
397             throw new DateTimeException("Intervals do not connect: " + this + " and " + other);
398         }
399         int cmpStart = start.compareTo(other.start);
400         int cmpEnd = end.compareTo(other.end);
401         if (cmpStart >= 0 && cmpEnd <= 0) {
402             return other;
403         } else if (cmpStart <= 0 && cmpEnd >= 0) {
404             return this;
405         } else {
406             Instant newStart = (cmpStart >= 0 ? other.start : start);
407             Instant newEnd = (cmpEnd <= 0 ? other.end : end);
408             return Interval.of(newStart, newEnd);
409         }
410     }

412     /**
413      * Calculates the smallest interval that encloses this interval and the specified interval.
414      * <p>
415      * The result of this method will {@linkplain #encloses(Interval) enclose}
416      * this interval and the specified interval.
417      * 
418      * @param other  the other interval to check for, not null
419      * @return the interval that spans the two intervals
420      */
421     public Interval span(Interval other) {
422         Objects.requireNonNull(other, "other");
423         int cmpStart = start.compareTo(other.start);
424         int cmpEnd = end.compareTo(other.end);
425         Instant newStart = (cmpStart >= 0 ? other.start : start);
426         Instant newEnd = (cmpEnd <= 0 ? other.end : end);
427         return Interval.of(newStart, newEnd);
428     }

432      * Checks if this interval is after the specified instant.
433      * <p>
434      * The result is true if the this instant starts after the specified instant.
435      * An empty interval behaves as though it is an instant for comparison purposes.
436      *
437      * @param instant  the other instant to compare to, not null
438      * @return true if the start of this interval is after the specified instant
439      */
440     public boolean isAfter(Instant instant) {
441         return start.compareTo(instant) > 0;
442     }

444     /**
445      * Checks if this interval is before the specified instant.
446      * <p>
447      * The result is true if the this instant ends before the specified instant.
448      * Since intervals do not include their end points, this will return true if the
449      * instant equals the end of the interval.
450      * An empty interval behaves as though it is an instant for comparison purposes.
451      *
452      * @param instant  the other instant to compare to, not null
453      * @return true if the start of this interval is before the specified instant
454      */
455     public boolean isBefore(Instant instant) {
456         return end.compareTo(instant) <= 0 && start.compareTo(instant) < 0;
457     }

459     //-------------------------------------------------------------------------
460     /**
461      * Checks if this interval is after the specified interval.
462      * <p>
463      * The result is true if the this instant starts after the end of the specified interval.
464      * Since intervals do not include their end points, this will return true if the
465      * instant equals the end of the interval.
466      * An empty interval behaves as though it is an instant for comparison purposes.
467      *
468      * @param interval  the other interval to compare to, not null
469      * @return true if this instant is after the specified instant
470      */
471     public boolean isAfter(Interval interval) {
472         return start.compareTo(interval.end) >= 0 && !interval.equals(this);
473     }
474 
475     /**
476      * Checks if this interval is before the specified interval.
477      * <p>
478      * The result is true if the this instant ends before the start of the specified interval.
479      * Since intervals do not include their end points, this will return true if the
480      * two intervals abut.
481      * An empty interval behaves as though it is an instant for comparison purposes.
482      *
483      * @param interval  the other interval to compare to, not null
484      * @return true if this instant is before the specified instant
485      */
486     public boolean isBefore(Interval interval) {
487         return end.compareTo(interval.start) <= 0 && !interval.equals(this);
488     }

490     //-----------------------------------------------------------------------
491     /**
492      * Obtains the duration of this interval.
493      * <p>
494      * An {@code Interval} is associated with two specific instants on the time-line.
495      * A {@code Duration} is simply an amount of time, separate from the time-line.
496      *
497      * @return the duration of the time interval
498      * @throws ArithmeticException if the calculation exceeds the capacity of {@code Duration}
499      */
500     public Duration toDuration() {
501         return Duration.between(start, end);
502     }
503 
504     //-----------------------------------------------------------------------
505     /**
506      * Checks if this interval is equal to another interval.
507      * <p>
508      * Compares this {@code Interval} with another ensuring that the two instants are the same.
509      * Only objects of type {@code Interval} are compared, other types return false.
510      *
511      * @param obj  the object to check, null returns false
512      * @return true if this is equal to the other interval
513      */
514     @Override
515     public boolean equals(Object obj) {
516         if (this == obj) {
517             return true;
518         }
519         if (obj instanceof Interval) {
520             Interval other = (Interval) obj;
521             return start.equals(other.start) && end.equals(other.end);
522         }
523         return false;
524     }
525 
526     /**
527      * A hash code for this interval.
528      *
529      * @return a suitable hash code
530      */
531     @Override
532     public int hashCode() {
533         return start.hashCode() ^ end.hashCode();
534     }

536     //-----------------------------------------------------------------------
537     /**
538      * Outputs this interval as a {@code String}, such as {@code 2007-12-03T10:15:30/2007-12-04T10:15:30}.
539      * <p>
540      * The output will be the ISO-8601 format formed by combining the
541      * {@code toString()} methods of the two instants, separated by a forward slash.
542      *
543      * @return a string representation of this instant, not null
544      */
545     @Override
546     public String toString() {
547         return start.toString() + '/' + end.toString();
548     }
