32  package org.threeten.extra.chrono;
33  
34  import java.time.chrono.AbstractChronology;
35  import java.time.temporal.ChronoField;
36  import java.time.temporal.ValueRange;
37  
38  /**
39   * A chronology describing one of the Nile river calendar systems.
40   *
41   * <h3>Implementation Requirements</h3>
42   * Implementations must be immutable and thread-safe.
43   */
44  abstract class AbstractNileChronology
45          extends AbstractChronology {
46  
47      /**
48       * Range of proleptic-year.
49       */
50      static final ValueRange YEAR_RANGE = ValueRange.of(-999_998, 999_999);
51      /**
52       * Range of year.
53       */
54      static final ValueRange YOE_RANGE = ValueRange.of(1, 999_999);
55      /**
56       * Range of proleptic month.
57       */
58      static final ValueRange PROLEPTIC_MONTH_RANGE = ValueRange.of(-999_998 * 13L, 999_999 * 13L + 12);
59      /**
60       * Range of months.
61       */
62      static final ValueRange MOY_RANGE = ValueRange.of(1, 13);
63      /**
64       * Range of weeks.
65       */
66      static final ValueRange ALIGNED_WOM_RANGE = ValueRange.of(1, 1, 5);
67      /**
68       * Range of days.
69       */
70      static final ValueRange DOM_RANGE = ValueRange.of(1, 5, 30);
71      /**
72       * Range of days.
73       */
74      static final ValueRange DOM_RANGE_NONLEAP = ValueRange.of(1, 5);
75      /**
76       * Range of days.
77       */
78      static final ValueRange DOM_RANGE_LEAP = ValueRange.of(1, 6);
79  
80      /**
81       * Private constructor.
82       */
83      AbstractNileChronology() {
84      }
85  
86      //-----------------------------------------------------------------------
87      /**
88       * Checks if the specified year is a leap year.
89       * <p>
90       * The proleptic-year is leap if the remainder after division by four equals three.
91       * This method does not validate the year passed in, and only has a
92       * well-defined result for years in the supported range.
93       *
94       * @param prolepticYear  the proleptic-year to check, not validated for range
95       * @return true if the year is a leap year
96       */
97      @Override
98      public boolean isLeapYear(long prolepticYear) {
99          return Math.floorMod(prolepticYear, 4) == 3;
100     }
101 
102     //-----------------------------------------------------------------------
103     @Override
104     public ValueRange range(ChronoField field) {
105         switch (field) {
106             case DAY_OF_MONTH:
107                 return DOM_RANGE;
108             case ALIGNED_WEEK_OF_MONTH:
109                 return ALIGNED_WOM_RANGE;
110             case MONTH_OF_YEAR:
111                 return MOY_RANGE;
112             case PROLEPTIC_MONTH:
113                 return PROLEPTIC_MONTH_RANGE;
114             case YEAR_OF_ERA:
115                 return YOE_RANGE;
116             case YEAR:
117                 return YEAR_RANGE;
118             default:
119                 break;
120         }
121         return field.range();
122     }
123 
124 }
