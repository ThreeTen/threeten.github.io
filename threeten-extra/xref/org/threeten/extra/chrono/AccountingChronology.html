56  /**
57   * An Accounting calendar system.
58   * <p>
59   * This chronology defines the rules of a proleptic 52/53-week Accounting calendar system.
60   * This calendar system follows the rules as laid down in <a href="http://www.irs.gov/publications/p538/ar02.html">IRS Publication 538</a>
61   * and the <a href="http://www.ifrs.org">International Financial Reporting Standards</a>.
62   * The start of the Accounting calendar will vary against the ISO calendar.
63   * Depending on options chosen, it can start as early as {@code 0000-01-26 (ISO)} or as late as {@code 0001-01-04 (ISO)}.
64   * <p>
65   * This class is proleptic. It implements Accounting chronology rules for the entire time-line.
66   * <p>
67   * The fields are defined as follows:
68   * <ul>
69   * <li>era - There are two eras, the current 'Current Era' (CE) and the previous era 'Before Current Era' (BCE).
70   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year one.
71   *  For the previous era the year increases from one as time goes backwards.
72   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the
73   *  current era. For the previous era, years have zero, then negative values.
74   * <li>month-of-year - There are 12 or 13 months (periods) in an Accounting year, numbered from 1 to 12 or 13.
75   * <li>day-of-month - There are 28 or 35 days in each Accounting month, numbered from 1 to 35.
76   *  Month length depends on how the year has been divided.
77   *  When the Accounting leap year occurs, a week (7 days) is added to a specific month;
78   *  this increases to maximum day-of-month numbering to 35 or 42.
79   * <li>day-of-year - There are 364 days in a standard Accounting year and 371 in a leap year.
80   *  The days are numbered from 1 to 364 or 1 to 371.
81   * <li>leap-year - Leap years usually occur every 5 or 6 years.  Timing depends on settings chosen.
82   * </ul>
83   *
84   * <h3>Implementation Requirements</h3>
85   * This class is immutable and thread-safe.
86   */

150     /**
151      * Creates an {@code AccountingChronology} validating the input.
152      * Package private as only meant to be called from the builder.
153      * 
154      * @param endsOn  The day-of-week a given year ends on.
155      * @param end The  month-end the year is based on.
156      * @param inLastWeek  Whether the year ends in the last week of the month, or nearest the end-of-month.
157      * @param division  How the year is divided.
158      * @param leapWeekInMonth  The month in which the leap-week resides.
159      * @return The created Chronology, not null.
160      * @throws DateTimeException if the chronology cannot be built.
161      */
162     static AccountingChronology create(DayOfWeek endsOn, Month end, boolean inLastWeek, AccountingYearDivision division, int leapWeekInMonth) {
163         if (endsOn == null || end == null || division == null || leapWeekInMonth == 0) {
164             throw new IllegalStateException("AccountingCronology cannot be built: "
165                     + (endsOn == null ? "| ending day-of-week |" : "")
166                     + (end == null ? "| month ending in/nearest to |" : "")
167                     + (division == null ? "| how year divided |" : "")
168                     + (leapWeekInMonth == 0 ? "| leap-week month |" : "")
169                     + " not set.");
170         }
171         if (!division.getMonthsInYearRange().isValidValue(leapWeekInMonth)) {
172             throw new IllegalStateException("Leap week cannot not be placed in non-existant month " + leapWeekInMonth
173                     + ", range is [" + division.getMonthsInYearRange() + "].");
174         }
175 
176         // Derive cached information.
177         LocalDate endingLimit = inLastWeek ? LocalDate.of(0, end, 1).with(TemporalAdjusters.lastDayOfMonth()) :
178                 LocalDate.of(0, end, 1).with(TemporalAdjusters.lastDayOfMonth()).plusDays(3);
179         LocalDate yearZeroEnd = endingLimit.with(TemporalAdjusters.previousOrSame(endsOn));
180         int yearZeroDifference = (int) yearZeroEnd.until(endingLimit, ChronoUnit.DAYS);
181         // Longest/shortest month lengths and related
182         int longestMonthLength = 0;
183         int shortestMonthLength = Integer.MAX_VALUE;
184         for (int month = 1; month <= division.getMonthsInYearRange().getMaximum(); month++) {
185             int monthLength = division.getWeeksInMonth(month);
186             shortestMonthLength = Math.min(shortestMonthLength, monthLength);
187             longestMonthLength = Math.max(longestMonthLength, monthLength + (month == leapWeekInMonth ? 1 : 0));
188         }
189         ValueRange alignedWeekOfMonthRange = ValueRange.of(1, shortestMonthLength, longestMonthLength);
190         ValueRange dayOfMonthRange = ValueRange.of(1, shortestMonthLength * 7, longestMonthLength * 7);
191         int daysToEpoch = Math.toIntExact(0 - yearZeroEnd.plusDays(1).toEpochDay());
192 
193         return new AccountingChronology(endsOn, end, inLastWeek, division, leapWeekInMonth, yearZeroDifference, alignedWeekOfMonthRange, dayOfMonthRange, daysToEpoch);
194     }

445     /**
446      * Checks if the specified year is a leap year.
447      * <p>
448      * An Accounting proleptic-year is leap if the time between the end of the previous year 
449      * and the end of the current year is 371 days.
450      * This method does not validate the year passed in, and only has a
451      * well-defined result for years in the supported range.
452      *
453      * @param prolepticYear  the proleptic-year to check, not validated for range
454      * @return true if the year is a leap year
455      */
456     @Override
457     public boolean isLeapYear(long prolepticYear) {
458         return Math.floorMod(prolepticYear + getISOLeapYearCount(prolepticYear) + yearZeroDifference, 7) == 0
459                 || Math.floorMod(prolepticYear + getISOLeapYearCount(prolepticYear + 1) + yearZeroDifference, 7) == 0;
460     }

462     /**
463      * Return the number of ISO Leap Years since Accounting Year 1.
464      * <p>
465      * This method calculates how many ISO leap years have passed since year 1.
466      * The count returned will be negative for years before 1.
467      * This method does not validate the year passed in, and only has a
468      * well-defined result for years in the supported range.
469      * 
470      * @param prolepticYear  the proleptic-year to check, not validated for range
471      * @return the count of leap years since year 1.
472      */
473     private long getISOLeapYearCount(long prolepticYear) {
474         long offsetYear = prolepticYear - (end == Month.JANUARY ? 1 : 0) - 1;
475         return Math.floorDiv(offsetYear, 4) - Math.floorDiv(offsetYear, 100) + Math.floorDiv(offsetYear, 400) + (end == Month.JANUARY ? 1 : 0);
476     }

482      * The count returned will be negative for years before 1.

512     @Override
513     public ValueRange range(ChronoField field) {
514         switch (field) {
515             case ALIGNED_WEEK_OF_MONTH:
516                 return alignedWeekOfMonthRange;
517             case ALIGNED_WEEK_OF_YEAR:
518                 return ALIGNED_WEEK_OF_YEAR_RANGE;
519             case DAY_OF_MONTH:
520                 return dayOfMonthRange;
521             case DAY_OF_YEAR:
522                 return DAY_OF_YEAR_RANGE;
523             case MONTH_OF_YEAR:
524                 return division.getMonthsInYearRange();
525             case PROLEPTIC_MONTH:
526                 return division == THIRTEEN_EVEN_MONTHS_OF_4_WEEKS ? PROLEPTIC_MONTH_RANGE_13 : PROLEPTIC_MONTH_RANGE_12;
527             default:
528                 break;
529         }
530         return field.range();
531     }
