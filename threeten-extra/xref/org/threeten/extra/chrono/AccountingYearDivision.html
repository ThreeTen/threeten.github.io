32  package org.threeten.extra.chrono;
33  
34  import java.time.DateTimeException;
35  import java.time.temporal.ChronoField;
36  import java.time.temporal.ValueRange;
37  import java.util.Arrays;
38  
39  /**
40   * How an Accounting year is divided.
41   * <p>
42   * An Accounting calendar system generally divides a year into smaller periods, similar in length to regular calendar months.
43   * The most common divisions either use 12 such 'months' (requiring one every quarter to be 5 weeks instead of 4),
44   * or use 13 of 4 weeks each (making one quarter have an extra month, or each quarter have partial months).
45   *
46   * <h3>Implementation Requirements:</h3>
47   * This is an immutable and thread-safe enum.
48   */
49  public enum AccountingYearDivision {
50  
51      /**
52       * The singleton instance for a year divided into 4 quarters, 
53       * each having 3 months with lengths of 4, 4, and 5 weeks, respectively.
54       */
55      QUARTERS_OF_PATTERN_4_4_5_WEEKS(new int[] {4, 4, 5, 4, 4, 5, 4, 4, 5, 4, 4, 5}),
56      /**
57       * The singleton instance for a year divided into 4 quarters, 
58       * each having 3 months with lengths of 4, 5, and 4 weeks, respectively.
59       */
60      QUARTERS_OF_PATTERN_4_5_4_WEEKS(new int[] {4, 5, 4, 4, 5, 4, 4, 5, 4, 4, 5, 4}),
61      /**
62       * The singleton instance for a year divided into 4 quarters, 
63       * each having 3 months with lengths of 5, 4, and 4 weeks, respectively.
64       */
65      QUARTERS_OF_PATTERN_5_4_4_WEEKS(new int[] {5, 4, 4, 5, 4, 4, 5, 4, 4, 5, 4, 4}),
66      /**
67       * The singleton instance for a year divided into 13 even months,
68       * each having 4 weeks.
69       */
70      THIRTEEN_EVEN_MONTHS_OF_4_WEEKS(new int[] {4, 4, 4, 4, 4, 4, 4, 4, 4, 4, 4, 4, 4});
71  
72      /**
73       * The number of weeks in each month.
74       */
75      private final int[] weeksInMonths;
76      /**
77       * The range of months in each year.
78       */
79      private final ValueRange monthsInYearRange;
80      /**
81       * The elapsed number of weeks at the start of each month.
82       */
83      private final int[] elapsedWeeks;
84  
85      //-----------------------------------------------------------------------
86      /**
87       * Private constructor for enum, for creating cached info.
88       * 
89       * @param weeksInMonths The number of weeks in each month (period).
90       */
91      private AccountingYearDivision(int[] weeksInMonths) {
92          this.weeksInMonths = weeksInMonths;
93  
94          this.monthsInYearRange = ValueRange.of(1, weeksInMonths.length);
95  
96          this.elapsedWeeks = new int[weeksInMonths.length];
97          for (int i = 1; i < weeksInMonths.length; i++) {
98              elapsedWeeks[i] = elapsedWeeks[i - 1] + weeksInMonths[i - 1];
99          }
100     }
101 
102     //-----------------------------------------------------------------------
103     /**
104      * Gets the range of months in a year.
105      * <p>
106      * <ul>
107      * <li>The AccountingYearDivision {@code QUARTERS_OF_PATTERN_4_4_5_WEEKS} range is [1, 12].
108      * <li>The AccountingYearDivision {@code QUARTERS_OF_PATTERN_4_5_4_WEEKS} range is [1, 12].
109      * <li>The AccountingYearDivision {@code QUARTERS_OF_PATTERN_5_4_4_WEEKS} range is [1, 12].
110      * <li>The AccountingYearDivision {@code THIRTEEN_EVEN_MONTHS_OF_4_WEEKS} range is [1, 13].
111      * </ul>
112      *
113      * @return the range of months (periods) in a year.
114      */
115     ValueRange getMonthsInYearRange() {
116         return monthsInYearRange;
117     }
118 
119     /**
120      * Gets the length of the year in months.
121      * 
122      * @return The length of the year in months.
123      */
124     int lengthOfYearInMonths() {
125         return weeksInMonths.length;
126     }
127 
128     //-----------------------------------------------------------------------
129     /**
130      * Get the number of weeks in the given month (period). 
131      * 
132      * @param month  The month for which to get the count of weeks.
133      * @return The count of weeks in the given month.
134      * @throws DateTimeException if the month isn't in the valid range of months.
135      */
136     int getWeeksInMonth(int month) {
137         return getWeeksInMonth(month, 0);
138     }
139 
140     /**
141      * Get the number of weeks in the given month (period), with the leap year in the indicated month.
142      * 
143      * @param month  The month for which to get the count of weeks.
144      * @param leapWeekInMonth  The month in which the leap-week resides
145      * @return The count of weeks in the given month, including any leap week.
146      * @throws DateTimeException if the month isn't in the valid range of months.
147      */
148     int getWeeksInMonth(int month, int leapWeekInMonth) {
149         month = monthsInYearRange.checkValidIntValue(month, ChronoField.MONTH_OF_YEAR);
150         leapWeekInMonth = (leapWeekInMonth == 0 ? 0 : monthsInYearRange.checkValidIntValue(leapWeekInMonth, ChronoField.MONTH_OF_YEAR));
151         return weeksInMonths[month - 1] + (month == leapWeekInMonth ? 1 : 0);
152     }
153 
154     //-----------------------------------------------------------------------
155     /**
156      * Get the number of weeks elapsed before the start of the month.
157      * 
158      * @param month The month
159      * @return The number of weeks elapsed before the start of the month.
160      * @throws DateTimeException if the month isn't in the valid range of months.
161      */
162     int getWeeksAtStartOfMonth(int month) {
163         return getWeeksAtStartOfMonth(month, 0);
164     }
165 
166     /**
167      * Get the number of weeks elapsed before the start of the month.
168      * 
169      * @param month The month
170      * @param leapWeekInMonth  The month in which the leap-week resides
171      * @return The number of weeks elapsed before the start of the month, including any leap week.
172      * @throws DateTimeException if the month isn't in the valid range of months.
173      */
174     int getWeeksAtStartOfMonth(int month, int leapWeekInMonth) {
175         month = monthsInYearRange.checkValidIntValue(month, ChronoField.MONTH_OF_YEAR);
176         leapWeekInMonth = (leapWeekInMonth == 0 ? 0 : monthsInYearRange.checkValidIntValue(leapWeekInMonth, ChronoField.MONTH_OF_YEAR));
177         return elapsedWeeks[month - 1] + (leapWeekInMonth != 0 && month > leapWeekInMonth ? 1 : 0);
178     }
179 
180     /**
181      * Get the month from a count of elapsed weeks.
182      * 
183      * @param weeksElapsed The weeks elapsed since the start of the year.
184      * @return the month
185      * @throws DateTimeException if the month isn't in the valid range of months, 
186      *   or the week isn't in the valid range.
187      */
188     int getMonthFromElapsedWeeks(int weeksElapsed) {
189         return getMonthFromElapsedWeeks(weeksElapsed, 0);
190     }
191 
192     /**
193      * Get the month from a count of elapsed weeks.
194      * 
195      * @param weeksElapsed The weeks elapsed since the start of the year.
196      * @param leapWeekInMonth  The month in which the leap-week resides
197      * @return the month
198      * @throws DateTimeException if the month isn't in the valid range of months, 
199      *   or the week isn't in the valid range.
200      */
201     int getMonthFromElapsedWeeks(int weeksElapsed, int leapWeekInMonth) {
202         if (weeksElapsed < 0 || weeksElapsed >= (leapWeekInMonth == 0 ? 52 : 53)) {
203             throw new DateTimeException("Count of '" + elapsedWeeks + "' elapsed weeks not valid,"
204                     + " should be in the range [0, " + (leapWeekInMonth == 0 ? 52 : 53) + ")");
205         }
206         leapWeekInMonth = (leapWeekInMonth == 0 ? 0 : monthsInYearRange.checkValidIntValue(leapWeekInMonth, ChronoField.MONTH_OF_YEAR));
207 
208         int month = Arrays.binarySearch(elapsedWeeks, weeksElapsed);
209         // Binary search returns 0-indexed if found, negative - 1 for insert position if not.
210         month = (month >= 0 ? month + 1 : 0 - month - 1);
211         // Need to move to previous month if there was a leap week and in the first week.
212         return leapWeekInMonth == 0 || month <= leapWeekInMonth || weeksElapsed > elapsedWeeks[month - 1] ? month : month - 1;
213     }
214 }
