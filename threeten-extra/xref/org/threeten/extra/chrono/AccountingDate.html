60  /**
61   * A date in an Accounting calendar system.
62   * <p>
63   * This date operates using a given {@linkplain AccountingChronology Accounting calendar}.
64   * An Accounting calendar differs greatly from the ISO calendar.
65   * The start of the Accounting calendar will vary against the ISO calendar.
66   * Depending on options chosen, it can start as early as {@code 0000-01-26 (ISO)} or as late as {@code 0001-01-04 (ISO)}.
67   *
68   * <h3>Implementation Requirements</h3>
69   * This class is immutable and thread-safe.
70   * <p>
71   * This class must be treated as a value type. Do not synchronize, rely on the
72   * identity hash code or use the distinction between equals() and ==.
73   */
74  public final class AccountingDate extends AbstractDate implements ChronoLocalDate, Serializable {

197      * <p>
198      * This method almost matches the signature of the functional interface {@link TemporalQuery}
199      * and must be used as a query via something that supplies the missing parameter,
200      * such as a curried method reference, {@code temporal -> AccountingDate.from(chronology, temporal)}
201      * (where {@code chronology} resolves to a set up {@code AccountingChronology}).
202      *

207      *  NullPointerException if an AccountingChronology was not provided
208      */
209     public static AccountingDate from(AccountingChronology chronology, TemporalAccessor temporal) {
210         if (temporal instanceof AccountingDate && ((AccountingDate) temporal).getChronology().equals(chronology)) {
211             return (AccountingDate) temporal;
212         }
213         return ofEpochDay(chronology, temporal.getLong(EPOCH_DAY));

217     /**
218      * Obtains an {@code AccountingDate} representing a date in the given Accounting calendar
219      * system from the proleptic-year and day-of-year fields.
220      * <p>
221      * This returns an {@code AccountingDate} with the specified fields.
222      * The day must be valid for the year, otherwise an exception will be thrown.
223      *
224      * @param chronology  the Accounting chronology to base the date on, not null
225      * @param prolepticYear  the Accounting proleptic-year
226      * @param dayOfYear  the Accounting day-of-year, from 1 to 371
227      * @return the date in Accounting calendar system, not null
228      * @throws DateTimeException if the value of any field is out of range,
229      *  or if the day-of-year is invalid for the year,
230      *  NullPointerException if an AccountingChronology was not provided
231      */
232     static AccountingDate ofYearDay(AccountingChronology chronology, int prolepticYear, int dayOfYear) {
233         Objects.requireNonNull(chronology, "A previously setup chronology is required.");
234         YEAR.checkValidValue(prolepticYear);
235         DAY_OF_YEAR_RANGE.checkValidValue(dayOfYear, DAY_OF_YEAR);
236         boolean leap = chronology.isLeapYear(prolepticYear);
237         if (dayOfYear > WEEKS_IN_YEAR * DAYS_IN_WEEK && !leap) {
238             throw new DateTimeException("Invalid date 'DayOfYear " + dayOfYear + "' as '" + prolepticYear + "' is not a leap year");
239         }
240 
241         int month = (leap ? chronology.division.getMonthFromElapsedWeeks((dayOfYear - 1) / DAYS_IN_WEEK, chronology.leapWeekInMonth)
242                 : chronology.division.getMonthFromElapsedWeeks((dayOfYear - 1) / DAYS_IN_WEEK));
243         int dayOfMonth = dayOfYear - (leap ? chronology.division.getWeeksAtStartOfMonth(month, chronology.leapWeekInMonth)
244                 : chronology.division.getWeeksAtStartOfMonth(month)) * DAYS_IN_WEEK;
245 
246         return new AccountingDate(chronology, prolepticYear, month, dayOfMonth);
247     }

249     /**
250      * Obtains an {@code AccountingDate} representing a date in the given Accounting calendar
251      * system from the epoch-day.
252      *
253      * @param chronology  the Accounting chronology to base the date on, not null
254      * @param epochDay  the epoch day to convert based on 1970-01-01 (ISO)
255      * @return the date in given Accounting calendar system, not null
256      * @throws DateTimeException if the epoch-day is out of range,
257      *  NullPointerException if an AccountingChronology was not provided
258      */
259     static AccountingDate ofEpochDay(AccountingChronology chronology, long epochDay) {
260         EPOCH_DAY.range().checkValidValue(epochDay, EPOCH_DAY);  // validate outer bounds
261         // Use Accounting 1 to help with 0-counts.  Leap years can occur at any time.
262         long accountingEpochDay = epochDay + chronology.ACCOUNTING_0001_TO_ISO_1970;
263 
264         int longCycle = (int) Math.floorDiv(accountingEpochDay, DAYS_PER_LONG_CYCLE);
265         int daysInLongCycle = (int) Math.floorMod(accountingEpochDay, DAYS_PER_LONG_CYCLE);
266 
267         // Value is an estimate, as the floating leap-years make this difficult.
268         int year = (daysInLongCycle - (daysInLongCycle / 365 + daysInLongCycle / (4 * 365 + 1) - daysInLongCycle / (100 * 365 + 24)) / 7) / (DAYS_IN_WEEK * WEEKS_IN_YEAR);
269         int yearStart = (int) (WEEKS_IN_YEAR * (year - 1) + chronology.previousLeapYears(year)) * DAYS_IN_WEEK;
270 
271         // Despite the year being an estimate, the effect should still be within a few days.
272         if (yearStart > daysInLongCycle) {
273             year--;
274             yearStart -= (WEEKS_IN_YEAR + (chronology.isLeapYear(year) ? 1 : 0)) * DAYS_IN_WEEK;
275         } else if (daysInLongCycle - yearStart >= (WEEKS_IN_YEAR + (chronology.isLeapYear(year) ? 1 : 0)) * DAYS_IN_WEEK) {
276             yearStart += (WEEKS_IN_YEAR + (chronology.isLeapYear(year) ? 1 : 0)) * DAYS_IN_WEEK;
277             year++;
278         }
279 
280         return ofYearDay(chronology, year + 400 * longCycle, daysInLongCycle - yearStart + 1);
281     }

283     private static AccountingDate resolvePreviousValid(AccountingChronology chronology, int prolepticYear, int month, int day) {
284         day = Math.min(day, lengthOfMonth(chronology, prolepticYear, month));
285         return new AccountingDate(chronology, prolepticYear, month, day);

288     private static int lengthOfMonth(AccountingChronology chronology, int prolepticYear, int month) {
289         return (chronology.isLeapYear(prolepticYear) ? chronology.division.getWeeksInMonth(month, chronology.leapWeekInMonth)
290                 : chronology.division.getWeeksInMonth(month)) * DAYS_IN_WEEK;
291     }

293     /**
294      * Creates an {@code AccountingDate} validating the input.
295      *
296      * @param chronology  the Accounting chronology to base the date on, not null
297      * @param prolepticYear  the Accounting proleptic-year
298      * @param dayOfYear  the Accounting day-of-year, from 1 to 371
299      * @return the date in Accounting calendar system, not null
300      * @throws DateTimeException if the value of any field is out of range,
301      *  or if the day-of-year is invalid for the month-year,
302      *  NullPointerException if an AccountingChronology was not provided
303      */
304     static AccountingDate create(AccountingChronology chronology, int prolepticYear, int month, int dayOfMonth) {
305         Objects.requireNonNull(chronology, "A previously setup chronology is required.");
306         YEAR.checkValidValue(prolepticYear);
307         chronology.range(MONTH_OF_YEAR).checkValidValue(month, MONTH_OF_YEAR);
308 
309         if (dayOfMonth < 1 || dayOfMonth > lengthOfMonth(chronology, prolepticYear, month)) {
310             if (month == chronology.leapWeekInMonth && dayOfMonth < (chronology.division.getWeeksInMonth(month) + 1) * DAYS_IN_WEEK
311                     && !chronology.isLeapYear(prolepticYear)) {
312                 throw new DateTimeException("Invalid date '" + month + "/" + dayOfMonth + "' as '" + prolepticYear + "' is not a leap year");
313             } else {
314                 throw new DateTimeException("Invalid date '" + month + "/" + dayOfMonth + "'");
315             }
316         }
317 
318         return new AccountingDate(chronology, prolepticYear, month, dayOfMonth);
319     }

466         long accountingEpochDay = ((year - 1) * WEEKS_IN_YEAR + chronology.previousLeapYears(year)) * DAYS_IN_WEEK + (getDayOfYear() - 1);
