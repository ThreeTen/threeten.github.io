52  /**
53   * The Pax calendar system.
54   * <p>
55   * This chronology defines the rules of the proleptic Pax calendar system.
56   * This calendar system is a proposed reform calendar system, and is not in common use.
57   * The Pax differs from the Gregorian in terms of month count and length, and the leap year rule.
58   * Dates are aligned such that {@code 0001-01-01 (Pax)} is {@code 0000-12-31 (ISO)}.
59   * <p>
60   * This class is proleptic. It implements Pax rules to the entire time-line.
61   * <p>
62   * The fields are defined as follows:
63   * <ul>
64   * <li>era - There are two eras, the current 'Current Era' (CE) and the previous era 'Before Current Era' (BCE).
65   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year one.
66   *  For the previous era the year increases from one as time goes backwards.
67   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the
68   *  current era. For the previous era, years have zero, then negative values.
69   * <li>month-of-year - There are 13 or 14 months in a Pax year, numbered from 1 to 13 (or 14).
70   * <li>day-of-month - There are 28 days in each Pax month, numbered from 1 to 28.
71   * In a leap year a one-week month ('Pax') is inserted between months 12 and 13, shifting 13 to 14.
72   * <li>day-of-year - There are 364 days in a standard Pax year and 371 in a leap year.
73   * The days are numbered from 1 to 364 or 1 to 371.
74   * <li>leap-year - Leap years occur in every year whose last two digits are divisible by {@code 6}, are {@code 99}, or are {@code 00} and the year is <b>not</b> divisible by 400.
75   * </ul>
76   * <p>
77   * For more information, please read the <a href="http://en.wikipedia.org/wiki/Pax_Calendar">Pax Calendar</a> Wikipedia article.
78   *
79   * <h3>Implementation Requirements</h3>
80   * This class is immutable and thread-safe.
81   */
82  public final class PaxChronology extends AbstractChronology implements Serializable {

94      /**
95       * The leap-month of Pax is only one week long.
96       */
97      static final int WEEKS_IN_LEAP_MONTH = 1;
98      /**
99       * Standard 7-day week.
100      */
101     static final int DAYS_IN_WEEK = 7;
102     /**
103      * In all months (except Pax), there are 4 complete weeks.
104      */
105     static final int WEEKS_IN_MONTH = 4;
106     /**
107      * There are 13 months in a (non-leap) year.
108      */
109     static final int MONTHS_IN_YEAR = 13;
110     /**
111      * There are 4 weeks of 7 days, or 28 total days in a month.
112      */
113     static final int DAYS_IN_MONTH = WEEKS_IN_MONTH * DAYS_IN_WEEK;
114     /**
115      * There are 13 months of 28 days, or 364 days in a (non-leap) year.
116      */
117     static final int DAYS_IN_YEAR = MONTHS_IN_YEAR * DAYS_IN_MONTH;
118     /**
119      * There are 52 weeks in a (non-leap) year.
120      */
121     static final int WEEKS_IN_YEAR = DAYS_IN_YEAR / DAYS_IN_WEEK;
122 
123     /**
124      * Range of aligned week of month.
125      */
126     static final ValueRange ALIGNED_WEEK_OF_MONTH_RANGE = ValueRange.of(1, WEEKS_IN_LEAP_MONTH, WEEKS_IN_MONTH);
127     /**
128      * Range of aligned week of year.
129      */
130     static final ValueRange ALIGNED_WEEK_OF_YEAR_RANGE = ValueRange.of(1, WEEKS_IN_YEAR, WEEKS_IN_YEAR + 1);
131     /**
132      * Range of day of month.
133      */
134     static final ValueRange DAY_OF_MONTH_RANGE = ValueRange.of(1, DAYS_IN_WEEK, DAYS_IN_MONTH);
135     /**
136      * Range of day of year.
137      */
138     static final ValueRange DAY_OF_YEAR_RANGE = ValueRange.of(1, DAYS_IN_YEAR, DAYS_IN_YEAR + DAYS_IN_WEEK);
139     /**
140      * Range of month of year.
141      */
142     static final ValueRange MONTH_OF_YEAR_RANGE = ValueRange.of(1, MONTHS_IN_YEAR, MONTHS_IN_YEAR + 1);

372     //-----------------------------------------------------------------------
373     /**
374      * Checks if the specified year is a leap year.
375      * <p>
376      * In general, a year is a leap year if the last two digits are divisible by 6 without remainder, or are 99. 
377      * Years with the last two digits of 00 are also leap years, with the exception of years divisible by 400, which are not.
378      * <p>
379      * For example, 2012 is a leap year because the last two digits (12) are divisible by 6. 
380      * 1999 is a leap year as the last two digits are both 9's (99). 
381      * 1900 is a leap year as the last two digits are both 0's (00), however 2000 was not a leap year as it is divisible by 400. 
382      * The year 0 is not a leap year.
383      *
384      * @param prolepticYear  the proleptic-year to check, not validated for range
385      * @return true if the year is a leap year
386      */
387     @Override
388     public boolean isLeapYear(long prolepticYear) {
389         long lastTwoDigits = prolepticYear % 100;
390         return Math.abs(lastTwoDigits) == 99 || (prolepticYear % 400 != 0 && (lastTwoDigits == 0 || lastTwoDigits % 6 == 0));
391     }

393     @Override
394     public int prolepticYear(Era era, int yearOfEra) {
395         if (!(era instanceof PaxEra)) {
396             throw new ClassCastException("Era must be PaxEra");
397         }
398         return (era == PaxEra.CE ? yearOfEra : 1 - yearOfEra);
399     }

412     @Override
413     public ValueRange range(ChronoField field) {
414         switch (field) {
415             case ALIGNED_WEEK_OF_MONTH:
416                 return ALIGNED_WEEK_OF_MONTH_RANGE;
417             case ALIGNED_WEEK_OF_YEAR:
418                 return ALIGNED_WEEK_OF_YEAR_RANGE;
419             case DAY_OF_MONTH:
420                 return DAY_OF_MONTH_RANGE;
421             case DAY_OF_YEAR:
422                 return DAY_OF_YEAR_RANGE;
423             case MONTH_OF_YEAR:
424                 return MONTH_OF_YEAR_RANGE;
425             default:
426                 return field.range();
427         }
428     }
