50  /**
51   * The Ethiopic calendar system.
52   * <p>
53   * This chronology defines the rules of the Ethiopic calendar system.
54   * This calendar system is primarily used in Ethiopia.
55   * Dates are aligned such that {@code 0001-01-01 (Ethiopic)} is {@code 0284-08-29 (ISO)}.
56   * <p>
57   * The fields are defined as follows:
58   * <ul>
59   * <li>era - There are two eras, the current 'Incarnation Era' (INCARNATION) and the previous era (BEFORE_INCARNATION).
60   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year one.
61   *  For the previous era the year increases from one as time goes backwards.
62   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the
63   *  current era. For the previous era, years have zero, then negative values.
64   * <li>month-of-year - There are 13 months in a Ethiopic year, numbered from 1 to 13.
65   * <li>day-of-month - There are 30 days in each of the first 12 Ethiopic months, numbered 1 to 30.
66   *  The 13th month has 5 days, or 6 in a leap year, numbered 1 to 5 or 1 to 6.
67   * <li>day-of-year - There are 365 days in a standard Ethiopic year and 366 in a leap year.
68   *  The days are numbered from 1 to 365 or 1 to 366.
69   * <li>leap-year - Leap years occur every 4 years.
70   * </ul>
71   *
72   * <h3>Implementation Requirements</h3>
73   * This class is immutable and thread-safe.
74   */
75  public final class EthiopicChronology
76          extends AbstractNileChronology
77          implements Serializable {

138     //-----------------------------------------------------------------------
139     /**
140      * Obtains a local date in Ethiopic calendar system from the
141      * era, year-of-era, month-of-year and day-of-month fields.
142      *
143      * @param era  the Ethiopic era, not null
144      * @param yearOfEra  the year-of-era
145      * @param month  the month-of-year
146      * @param dayOfMonth  the day-of-month
147      * @return the Ethiopic local date, not null
148      * @throws DateTimeException if unable to create the date
149      * @throws ClassCastException if the {@code era} is not a {@code EthiopicEra}
150      */
151     @Override
152     public EthiopicDate date(Era era, int yearOfEra, int month, int dayOfMonth) {
153         return date(prolepticYear(era, yearOfEra), month, dayOfMonth);
154     }

318     //-----------------------------------------------------------------------
319     @Override
320     public int prolepticYear(Era era, int yearOfEra) {
321         if (era instanceof EthiopicEra == false) {
322             throw new ClassCastException("Era must be EthiopicEra");
323         }
324         return (era == EthiopicEra.INCARNATION ? yearOfEra : 1 - yearOfEra);
325     }

330     }

335     }
