49  /**
50   * The International Fixed calendar system.
51   * <p>
52   * This chronology defines the rules of the International Fixed calendar system.
53   * It shares the leap year rule with the Gregorian calendar.
54   * Dates are aligned such that {@code 0001-01-01 (International Fixed)} is {@code 0001-01-01 (ISO)}.
55   * <p>
56   * This class is proleptic. It implements only years greater or equal to 1.
57   * <p>
58   * This class implements a calendar where January 1st is the start of the year.
59   * <p>
60   * The fields are defined as follows:
61   * <ul>
62   * <li>era - There is only one era, the current 'Common Era' (CE).
63   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year 1.
64   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the current era.
65   * <li>month-of-year - There are 13 months in an International Fixed year, numbered from 1 to 13.
66   * <li>day-of-month - There are 28 days in a standard International Fixed month, numbered from 1 to 28,
67   *  with day 29 used in months 6 and 12 (month 6 only has 29 days in a leap year).
68   * <li>day-of-year - There are 365 days in a standard International Fixed year and 366 days in a leap year.
69   *  The days are numbered accordingly.
70   * <li>leap-year - Leap years occur every 4 years, but skips 3 out of four centuries, i.e. when the century
71   *  is not divisible by 400. This is the same rule in use for the Gregorian calendar.
72   * <li>Week day - every month starts on a Sunday.  Leap-day and year-day are neither part of a week, nor of any month.
73   * </ul>
74   *
75   * <h3>Implementation Requirements</h3>
76   * This class is immutable and thread-safe.
77   */
78  public final class InternationalFixedChronology extends AbstractChronology implements Serializable {

87      private static final long serialVersionUID = -8252657100538813526L;
88      /**
89       * Standard 7-day week.
90       */
91      static final int DAYS_IN_WEEK = 7;
92      /**
93       * In all months, there are 4 complete weeks.
94       */
95      static final int WEEKS_IN_MONTH = 4;
96      /**
97       * There are 13 months in a year.
98       */
99      static final int MONTHS_IN_YEAR = 13;
100     /**
101      * There are 4 weeks of 7 days, or 28 total days in a month.
102      */
103     static final int DAYS_IN_MONTH = WEEKS_IN_MONTH * DAYS_IN_WEEK;
104     /**
105      * There are 29 days in a long month.
106      */
107     static final int DAYS_IN_LONG_MONTH = DAYS_IN_MONTH + 1;
108     /**
109      * There are 13 months of 28 days, or 365 days in a (non-leap) year.
110      */
111     static final int DAYS_IN_YEAR = MONTHS_IN_YEAR * DAYS_IN_MONTH + 1;
112     /**
113      * There are 52 weeks in a year.
114      */
115     static final int WEEKS_IN_YEAR = DAYS_IN_YEAR / DAYS_IN_WEEK;
116     /**
117      * The number of days in a 400 year cycle.
118      */
119     static final int DAYS_PER_CYCLE = 146097;
120     /**
121      * The number of days from year zero to year 1970, still the era only allows year 1 and higher.
122      * There are five 400 year cycles from year zero to 2000.
123      * There are 7 leap years from 1970 to 2000.
124      */
125     static final long DAYS_0000_TO_1970 = (DAYS_PER_CYCLE * 5L) - (30L * 365L + 7L);
126     /**
127      * Range of year.
128      */
129     static final ValueRange YEAR_RANGE = ValueRange.of(1, 1_000_000L);
130     /**
131      * Epoch day range.
132      */
133     static final ValueRange EPOCH_DAY_RANGE = ValueRange.of(-DAYS_0000_TO_1970, 1_000_000L * DAYS_IN_YEAR + getLeapYearsBefore(1_000_000L) - DAYS_0000_TO_1970);
134     /**
135      * Range of proleptic month.
136      */
137     private static final ValueRange PROLEPTIC_MONTH_RANGE = ValueRange.of(13, 1_000_000 * 13L - 1);
138     /**
139      * Range of day of month.
140      */
141     static final ValueRange DAY_OF_MONTH_RANGE = ValueRange.of(1, DAYS_IN_MONTH + 1);
142     /**
143      * Range of day of year.
144      */
145     static final ValueRange DAY_OF_YEAR_NORMAL_RANGE = ValueRange.of(1, DAYS_IN_YEAR);
146     /**
147      * Range of day of leap year.
148      */
149     static final ValueRange DAY_OF_YEAR_LEAP_RANGE = ValueRange.of(1, DAYS_IN_YEAR + 1);
150     /**
151      * Range of month of year.
152      */
153     static final ValueRange MONTH_OF_YEAR_RANGE = ValueRange.of(1, MONTHS_IN_YEAR);
154     /**
155      * Range of eras.
156      */
157     static final ValueRange ERA_RANGE = ValueRange.of(1, 1);
158     /**
159      * Empty range: [0, 0].
160      */
161     static final ValueRange EMPTY_RANGE = ValueRange.of(0, 0);

391     /**
392      * Checks if the specified year is a leap year.
393      * <p>
394      * A leap-year is a year of a longer length than normal.
395      * Leap years in the calendar system match those of the ISO calendar system.
396      *
397      * @param year  the proleptic-year to check, not validated for range
398      * @return true if the year is a leap year
399      */
400     @Override
401     public boolean isLeapYear(long year) {
402         return ((year & 3) == 0) && ((year % 100) != 0 || (year % 400) == 0);
403     }

432     //-----------------------------------------------------------------------
433     @Override
434     public ValueRange range(ChronoField field) {
435         switch (field) {
436             case ALIGNED_DAY_OF_WEEK_IN_YEAR:
437             case ALIGNED_DAY_OF_WEEK_IN_MONTH:
438             case DAY_OF_WEEK:
439                 return ValueRange.of(0, 1, 0, DAYS_IN_WEEK);
440             case ALIGNED_WEEK_OF_MONTH:
441                 return ValueRange.of(0, 1, 0, WEEKS_IN_MONTH);
442             case ALIGNED_WEEK_OF_YEAR:
443                 return ValueRange.of(0, 1, 0, WEEKS_IN_YEAR);
444             case DAY_OF_MONTH:
445                 return DAY_OF_MONTH_RANGE;
446             case DAY_OF_YEAR:
447                 return ChronoField.DAY_OF_YEAR.range();
448             case EPOCH_DAY:
449                 return EPOCH_DAY_RANGE;
450             case ERA:
451                 return ERA_RANGE;
452             case MONTH_OF_YEAR:
453                 return MONTH_OF_YEAR_RANGE;
454             case PROLEPTIC_MONTH:
455                 return PROLEPTIC_MONTH_RANGE;
456             case YEAR_OF_ERA:
457             case YEAR:
458                 return YEAR_RANGE;
459             default:
460                 return field.range();
461         }
462     }

464     @Override
465     public int prolepticYear(Era era, int yearOfEra) {
466         if (!(era instanceof InternationalFixedEra)) {
467             throw new ClassCastException("Invalid era: " + era);
468         }
469         YEAR_RANGE.checkValidIntValue(yearOfEra, ChronoField.YEAR_OF_ERA);
470         return yearOfEra;
471     }

473     /**
474      * Get the count of leap years since International fixed year 1.
475      *
476      * @param prolepticYear The year.
477      * @return The number of leap years since International fixed year 1.
478      */
479     static long getLeapYearsBefore(long prolepticYear) {
480         long yearBefore = prolepticYear - 1;
481         return (yearBefore / 4) - (yearBefore / 100) + (yearBefore / 400);
482     }
