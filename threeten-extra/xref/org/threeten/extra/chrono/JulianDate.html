60  /**
61   * A date in the Julian calendar system.
62   * <p>
63   * This date operates using the {@linkplain JulianChronology Julian calendar}.
64   * This calendar system is the forerunner to the modern Gregorian and ISO calendars.
65   * The Julian differs from the Gregorian only in terms of the leap year rule.
66   * Dates are aligned such that {@code 0001-01-01 (Julian)} is {@code 0000-12-30 (ISO)}.
67   *
68   * <h3>Implementation Requirements</h3>
69   * This class is immutable and thread-safe.
70   * <p>
71   * This class must be treated as a value type. Do not synchronize, rely on the
72   * identity hash code or use the distinction between equals() and ==.
73   */
74  public final class JulianDate
75          extends AbstractDate
76          implements ChronoLocalDate, Serializable {

78      /**
79       * Serialization version.
80       */
81      private static final long serialVersionUID = -7920528871688876868L;
82      /**
83       * The difference between the ISO and Julian epoch day count (Julian 0001-01-01 to ISO 1970-01-01).
84       */
85      private static final int JULIAN_0001_TO_ISO_1970 = 678577 + 40587;  // MJD values
86      /**
87       * The days per 4 year cycle.
88       */
89      private static final int DAYS_PER_CYCLE = (365 * 4) + 1;
90

92       * The proleptic year.

94      private final int prolepticYear;
95      /**
96       * The month.
97       */

99      /**
100      * The day.
101      */
102     private final short day;

136     /**
137      * Obtains the current {@code JulianDate} from the specified clock.
138      * <p>
139      * This will query the specified clock to obtain the current date - today.
140      * Using this method allows the use of an alternate clock for testing.
141      * The alternate clock may be introduced using {@linkplain Clock dependency injection}.
142      *
143      * @param clock  the clock to use, not null
144      * @return the current date, not null
145      * @throws DateTimeException if the current date cannot be obtained
146      */
147     public static JulianDate now(Clock clock) {
148         LocalDate now = LocalDate.now(clock);
149         return JulianDate.ofEpochDay(now.toEpochDay());
150     }

152     /**
153      * Obtains a {@code JulianDate} representing a date in the Julian calendar
154      * system from the proleptic-year, month-of-year and day-of-month fields.
155      * <p>
156      * This returns a {@code JulianDate} with the specified fields.
157      * The day must be valid for the year and month, otherwise an exception will be thrown.
158      *
159      * @param prolepticYear  the Julian proleptic-year
160      * @param month  the Julian month-of-year, from 1 to 12
161      * @param dayOfMonth  the Julian day-of-month, from 1 to 31
162      * @return the date in Julian calendar system, not null
163      * @throws DateTimeException if the value of any field is out of range,
164      *  or if the day-of-month is invalid for the month-year
165      */
166     public static JulianDate of(int prolepticYear, int month, int dayOfMonth) {
167         return JulianDate.create(prolepticYear, month, dayOfMonth);
168     }

170     /**
171      * Obtains a {@code JulianDate} from a temporal object.
172      * <p>
173      * This obtains a date in the Julian calendar system based on the specified temporal.
174      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
175      * which this factory converts to an instance of {@code JulianDate}.
176      * <p>
177      * The conversion typically uses the {@link ChronoField#EPOCH_DAY EPOCH_DAY}
178      * field, which is standardized across calendar systems.
179      * <p>
180      * This method matches the signature of the functional interface {@link TemporalQuery}
181      * allowing it to be used as a query via method reference, {@code JulianDate::from}.
182      *
183      * @param temporal  the temporal object to convert, not null
184      * @return the date in Julian calendar system, not null
185      * @throws DateTimeException if unable to convert to a {@code JulianDate}
186      */
187     public static JulianDate from(TemporalAccessor temporal) {
188         if (temporal instanceof JulianDate) {
189             return (JulianDate) temporal;
190         }
191         return JulianDate.ofEpochDay(temporal.getLong(EPOCH_DAY));
192     }

194     //-----------------------------------------------------------------------
195     /**
196      * Obtains a {@code JulianDate} representing a date in the Julian calendar
197      * system from the proleptic-year and day-of-year fields.
198      * <p>
199      * This returns a {@code JulianDate} with the specified fields.
200      * The day must be valid for the year, otherwise an exception will be thrown.
201      *
202      * @param prolepticYear  the Julian proleptic-year
203      * @param dayOfYear  the Julian day-of-year, from 1 to 366
204      * @return the date in Julian calendar system, not null
205      * @throws DateTimeException if the value of any field is out of range,
206      *  or if the day-of-year is invalid for the year
207      */
208     static JulianDate ofYearDay(int prolepticYear, int dayOfYear) {
209         JulianChronology.YEAR_RANGE.checkValidValue(prolepticYear, YEAR);
210         DAY_OF_YEAR.checkValidValue(dayOfYear);
211         boolean leap = JulianChronology.INSTANCE.isLeapYear(prolepticYear);
212         if (dayOfYear == 366 && leap == false) {
213             throw new DateTimeException("Invalid date 'DayOfYear 366' as '" + prolepticYear + "' is not a leap year");
214         }
215         Month moy = Month.of((dayOfYear - 1) / 31 + 1);
216         int monthEnd = moy.firstDayOfYear(leap) + moy.length(leap) - 1;
217         if (dayOfYear > monthEnd) {
218             moy = moy.plus(1);
219         }
220         int dom = dayOfYear - moy.firstDayOfYear(leap) + 1;
221         return new JulianDate(prolepticYear, moy.getValue(), dom);
222     }
223 
224     /**
225      * Obtains a {@code JulianDate} representing a date in the Julian calendar
226      * system from the epoch-day.
227      *
228      * @param epochDay  the epoch day to convert based on 1970-01-01 (ISO)
229      * @return the date in Julian calendar system, not null
230      * @throws DateTimeException if the epoch-day is out of range
231      */
232     static JulianDate ofEpochDay(final long epochDay) {
233         EPOCH_DAY.range().checkValidValue(epochDay, EPOCH_DAY);  // validate outer bounds
234         // use of Julian 0001 makes leap year at end of cycle
235         long julianEpochDay = epochDay + JULIAN_0001_TO_ISO_1970;
236         long cycle = Math.floorDiv(julianEpochDay, DAYS_PER_CYCLE);
237         long daysInCycle = Math.floorMod(julianEpochDay, DAYS_PER_CYCLE);
238         if (daysInCycle == DAYS_PER_CYCLE - 1) {
239             int year = (int) ((cycle * 4 + 3) + 1);
240             return ofYearDay(year, 366);
241         }
242         int year = (int) ((cycle * 4 + daysInCycle / 365) + 1);
243         int doy = (int) ((daysInCycle % 365) + 1);
244         return ofYearDay(year, doy);
245     }

247     private static JulianDate resolvePreviousValid(int prolepticYear, int month, int day) {
248         switch (month) {
249             case 2:
250                 day = Math.min(day, JulianChronology.INSTANCE.isLeapYear(prolepticYear) ? 29 : 28);
251                 break;
252             case 4:
253             case 6:
254             case 9:
255             case 11:
256                 day = Math.min(day, 30);
257                 break;
258         }
259         return new JulianDate(prolepticYear, month, day);
260     }

262     /**
263      * Creates a {@code JulianDate} validating the input.
264      *
265      * @param prolepticYear  the Julian proleptic-year
266      * @param month  the Julian month-of-year, from 1 to 12
267      * @param dayOfMonth  the Julian day-of-month, from 1 to 31
268      * @return the date in Julian calendar system, not null
269      * @throws DateTimeException if the value of any field is out of range,
270      *  or if the day-of-year is invalid for the month-year
271      */
272     static JulianDate create(int prolepticYear, int month, int dayOfMonth) {
273         JulianChronology.YEAR_RANGE.checkValidValue(prolepticYear, YEAR);
274         MONTH_OF_YEAR.checkValidValue(month);
275         DAY_OF_MONTH.checkValidValue(dayOfMonth);
276         if (dayOfMonth > 28) {
277             int dom = 31;
278             switch (month) {
279                 case 2:
280                     dom = (JulianChronology.INSTANCE.isLeapYear(prolepticYear) ? 29 : 28);
281                     break;
282                 case 4:
283                 case 6:
284                 case 9:
285                 case 11:
286                     dom = 30;
287                     break;
288             }
289             if (dayOfMonth > dom) {
290                 if (dayOfMonth == 29) {
291                     throw new DateTimeException("Invalid date 'February 29' as '" + prolepticYear + "' is not a leap year");
292                 } else {
293                     throw new DateTimeException("Invalid date '" + Month.of(month).name() + " " + dayOfMonth + "'");
294                 }
295             }
296         }
297         return new JulianDate(prolepticYear, month, dayOfMonth);
298     }

356      * Gets the chronology of this date, which is the Julian calendar system.
357      * <p>
358      * The {@code Chronology} represents the calendar system in use.
359      * The era and other fields in {@link ChronoField} are defined by the chronology.
360      *
361      * @return the Julian chronology, not null
362      */
363     @Override
364     public JulianChronology getChronology() {
365         return JulianChronology.INSTANCE;
366     }
367 
368     /**
369      * Gets the era applicable at this date.
370      * <p>
371      * The Julian calendar system has two eras, 'AD' and 'BC',
372      * defined by {@link JulianEra}.
373      *
374      * @return the era applicable at this date, not null
375      */
376     @Override
377     public JulianEra getEra() {
378         return (prolepticYear >= 1 ? JulianEra.AD : JulianEra.BC);
379     }

381     /**
382      * Returns the length of the month represented by this date.
383      * <p>
384      * This returns the length of the month in days.
385      * Month lengths match those of the ISO calendar system.
386      *
387      * @return the length of the month in days, from 28 to 31
388      */
389     @Override
390     public int lengthOfMonth() {
391         switch (month) {
392             case 2:
393                 return (isLeapYear() ? 29 : 28);
394             case 4:
395             case 6:
396             case 9:
397             case 11:
398                 return 30;
399             default:
400                 return 31;
401         }
402     }

453     //-----------------------------------------------------------------------
454     @Override
455     public long toEpochDay() {
456         long year = (long) prolepticYear;
457         long julianEpochDay = ((year - 1) * 365) + Math.floorDiv((year - 1), 4) + (getDayOfYear() - 1);
458         return julianEpochDay - JULIAN_0001_TO_ISO_1970;
459     }
