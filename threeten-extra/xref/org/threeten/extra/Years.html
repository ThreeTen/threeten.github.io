51  /**
52   * A year-based amount of time, such as '12 years'.
53   * <p>
54   * This class models a quantity or amount of time in terms of years.
55   * It is a type-safe way of representing a number of years in an application.
56   * <p>
57   * The model is of a directed amount, meaning that the amount may be negative.
58   *
59   * <h3>Implementation Requirements:</h3>
60   * This class is immutable and thread-safe.
61   * <p>
62   * This class must be treated as a value type. Do not synchronize, rely on the
63   * identity hash code or use the distinction between equals() and ==.
64   */
65  public final class Years
66          implements TemporalAmount, Comparable<Years>, Serializable {
67  
68      /**
69       * A constant for zero years.
70       */
71      public static final Years ZERO = new Years(0);
72      /**
73       * A constant for one year.
74       */
75      public static final Years ONE = new Years(1);
76  
77      /**
78       * A serialization identifier for this class.
79       */
80      private static final long serialVersionUID = -8903767091325669093L;
81      /**
82       * The pattern for parsing.
83       */
84      private static final Pattern PATTERN =
85              Pattern.compile("([-+]?)P([-+]?[0-9]+)Y", Pattern.CASE_INSENSITIVE);
86  
87      /**
88       * The number of years.
89       */
90      private final int years;
91  
92      /**
93       * Obtains a {@code Years} representing a number of years.
94       * <p>
95       * The resulting amount will have the specified years.
96       *
97       * @param years  the number of years, positive or negative
98       * @return the number of years, not null
99       */
100     public static Years of(int years) {
101         if (years == 0) {
102             return ZERO;
103         } else if (years == 1) {
104             return ONE;
105         }
106         return new Years(years);
107     }
108 
109     //-----------------------------------------------------------------------
110     /**
111      * Obtains an instance of {@code Years} from a temporal amount.
112      * <p>
113      * This obtains an instance based on the specified amount.
114      * A {@code TemporalAmount} represents an amount of time, which may be
115      * date-based or time-based, which this factory extracts to a {@code Years}.
116      * <p>
117      * The result is calculated by looping around each unit in the specified amount.
118      * Each amount is converted to years using {@link Temporals#convertAmount}.
119      * If the conversion yields a remainder, an exception is thrown.
120      * If the amount is zero, the unit is ignored.
121      * For example, "12 months" can be converted to years but "11 months" cannot.
122      *
123      * @param amount  the temporal amount to convert, not null
124      * @return the equivalent amount, not null
125      * @throws DateTimeException if unable to convert to a {@code Years}
126      * @throws ArithmeticException if numeric overflow occurs
127      */
128     public static Years from(TemporalAmount amount) {
129         if (amount instanceof Years) {
130             return (Years) amount;
131         }
132         Objects.requireNonNull(amount, "amount");
133         int years = 0;
134         for (TemporalUnit unit : amount.getUnits()) {
135             long value = amount.get(unit);
136             if (value != 0) {
137                 long[] converted = Temporals.convertAmount(value, unit, YEARS);
138                 if (converted[1] != 0) {
139                     throw new DateTimeException(
140                             "Amount could not be converted to a whole number of years: " + value + " " + unit);
141                 }
142                 years = Math.addExact(years, Math.toIntExact(converted[0]));
143             }
144         }
145         return of(years);
146     }

148     //-----------------------------------------------------------------------
149     /**
150      * Obtains a {@code Years} from a text string such as {@code PnY}.
151      * <p>
152      * This will parse the string produced by {@code toString()} which is
153      * based on the ISO-8601 period formats {@code PnY}.
154      * <p>
155      * The string starts with an optional sign, denoted by the ASCII negative
156      * or positive symbol. If negative, the whole amount is negated.
157      * The ASCII letter "P" is next in upper or lower case.
158      * The ASCII integer amount is next, which may be negative.
159      * The ASCII letter "Y" is next in upper or lower case.
160      * <p>
161      * The leading plus/minus sign, and negative values for years are
162      * not part of the ISO-8601 standard.
163      * <p>
164      * For example, the following are valid inputs:
165      * <pre>
166      *   "P2Y"             -- Years.of(2)
167      *   "P-2Y"            -- Years.of(-2)
168      *   "-P2Y"            -- Years.of(-2)
169      *   "-P-2Y"           -- Years.of(2)
170      * </pre>
171      *
172      * @param text  the text to parse, not null
173      * @return the parsed period, not null
174      * @throws DateTimeParseException if the text cannot be parsed to a period
175      */
176     public static Years parse(CharSequence text) {
177         Objects.requireNonNull(text, "text");
178         Matcher matcher = PATTERN.matcher(text);
179         if (matcher.matches()) {
180             int negate = ("-".equals(matcher.group(1)) ? -1 : 1);
181             String str = matcher.group(2);
182             try {
183                 int val = Integer.parseInt(str);
184                 return of(Math.multiplyExact(val, negate));
185             } catch (NumberFormatException ex) {
186                 throw new DateTimeParseException("Text cannot be parsed to a Years", text, 0, ex);
187             }
188         }
189         throw new DateTimeParseException("Text cannot be parsed to a Years", text, 0);
190     }

192     //-----------------------------------------------------------------------
193     /**
194      * Obtains a {@code Years} consisting of the number of years between two dates.
195      * <p>
196      * The start date is included, but the end date is not.
197      * The result of this method can be negative if the end is before the start.
198      *
199      * @param startDateInclusive  the start date, inclusive, not null
200      * @param endDateExclusive  the end date, exclusive, not null
201      * @return the number of years between this date and the end date, not null
202      */
203     public static Years between(Temporal startDateInclusive, Temporal endDateExclusive) {
204         return of(Math.toIntExact(YEARS.between(startDateInclusive, endDateExclusive)));
205     }

228     /**
229      * Gets the value of the requested unit.
230      * <p>
231      * This returns a value for the supported unit - {@link ChronoUnit#YEARS YEARS}.
232      * All other units throw an exception.
233      *
234      * @param unit  the {@code TemporalUnit} for which to return the value
235      * @return the long value of the unit
236      * @throws UnsupportedTemporalTypeException if the unit is not supported
237      */
238     @Override
239     public long get(TemporalUnit unit) {
240         if (unit == ChronoUnit.YEARS) {
241             return years;
242         }
243         throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
244     }

246     /**
247      * Gets the set of units supported by this amount.
248      * <p>
249      * The single supported unit is {@link ChronoUnit#YEARS YEARS}.
250      * <p>
251      * This set can be used in conjunction with {@link #get(TemporalUnit)}
252      * to access the entire state of the amount.
253      *
254      * @return a list containing the years unit, not null
255      */
256     @Override
257     public List<TemporalUnit> getUnits() {
258         return Collections.singletonList(YEARS);
259     }

263      * Gets the number of years in this amount.

267     public int getAmount() {
268         return years;

271     //-----------------------------------------------------------------------
272     /**
273      * Returns a copy of this amount with the specified amount added.
274      * <p>
275      * The parameter is converted using {@link Years#from(TemporalAmount)}.
276      * <p>
277      * This instance is immutable and unaffected by this method call.
278      *
279      * @param amountToAdd  the amount to add, not null
280      * @return a {@code Years} based on this instance with the requested amount added, not null
281      * @throws DateTimeException if the specified amount contains an invalid unit
282      * @throws ArithmeticException if numeric overflow occurs
283      */
284     public Years plus(TemporalAmount amountToAdd) {
285         return plus(Years.from(amountToAdd).getAmount());
286     }

288     /**
289      * Returns a copy of this amount with the specified number of years added.
290      * <p>
291      * This instance is immutable and unaffected by this method call.
292      *
293      * @param years  the amount of years to add, may be negative
294      * @return a {@code Years} based on this instance with the requested amount added, not null
295      * @throws ArithmeticException if the result overflows an int
296      */
297     public Years plus(int years) {
298         if (years == 0) {
299             return this;
300         }
301         return of(Math.addExact(this.years, years));
302     }

305     /**
306      * Returns a copy of this amount with the specified amount subtracted.
307      * <p>
308      * The parameter is converted using {@link Years#from(TemporalAmount)}.
309      * <p>
310      * This instance is immutable and unaffected by this method call.
311      *
312      * @param amountToAdd  the amount to add, not null
313      * @return a {@code Years} based on this instance with the requested amount subtracted, not null
314      * @throws DateTimeException if the specified amount contains an invalid unit
315      * @throws ArithmeticException if numeric overflow occurs
316      */
317     public Years minus(TemporalAmount amountToAdd) {
318         return minus(Years.from(amountToAdd).getAmount());
319     }

321     /**
322      * Returns a copy of this amount with the specified number of years subtracted.
323      * <p>
324      * This instance is immutable and unaffected by this method call.
325      *
326      * @param years  the amount of years to add, may be negative
327      * @return a {@code Years} based on this instance with the requested amount subtracted, not null
328      * @throws ArithmeticException if the result overflows an int
329      */
330     public Years minus(int years) {
331         if (years == 0) {
332             return this;
333         }
334         return of(Math.subtractExact(this.years, years));
335     }

337     //-----------------------------------------------------------------------
338     /**
339      * Returns an instance with the amount multiplied by the specified scalar.
340      * <p>
341      * This instance is immutable and unaffected by this method call.
342      *
343      * @param scalar  the scalar to multiply by, not null
344      * @return the amount multiplied by the specified scalar, not null
345      * @throws ArithmeticException if numeric overflow occurs
346      */
347     public Years multipliedBy(int scalar) {
348         if (scalar == 1) {
349             return this;
350         }
351         return of(Math.multiplyExact(years, scalar));
352     }
353 
354     /**
355      * Returns an instance with the amount divided by the specified divisor.
356      * <p>
357      * The calculation uses integer division, thus 3 divided by 2 is 1.
358      * <p>
359      * This instance is immutable and unaffected by this method call.
360      *
361      * @param divisor  the amount to divide by, may be negative
362      * @return the amount divided by the specified divisor, not null
363      * @throws ArithmeticException if the divisor is zero
364      */
365     public Years dividedBy(int divisor) {
366         if (divisor == 1) {
367             return this;
368         }
369         return of(years / divisor);
370     }
371 
372     /**
373      * Returns an instance with the amount negated.
374      * <p>
375      * This instance is immutable and unaffected by this method call.
376      *
377      * @return the negated amount, not null
378      * @throws ArithmeticException if numeric overflow occurs, which only happens if
379      *  the amount is {@code Long.MIN_VALUE}
380      */
381     public Years negated() {
382         return multipliedBy(-1);
383     }
384 
385     /**
386      * Returns a copy of this duration with a positive length.
387      * <p>
388      * This method returns a positive duration by effectively removing the sign from any negative total length.
389      * <p>
390      * This instance is immutable and unaffected by this method call.
391      *
392      * @return the absolute amount, not null
393      * @throws ArithmeticException if numeric overflow occurs, which only happens if
394      *  the amount is {@code Long.MIN_VALUE}
395      */
396     public Years abs() {
397         return years < 0 ? negated() : this;
398     }

412     //-----------------------------------------------------------------------
413     /**
414      * Adds this amount to the specified temporal object.
415      * <p>
416      * This returns a temporal object of the same observable type as the input
417      * with this amount added.
418      * <p>
419      * In most cases, it is clearer to reverse the calling pattern by using
420      * {@link Temporal#plus(TemporalAmount)}.
421      * <pre>
422      *   // these two lines are equivalent, but the second approach is recommended
423      *   dateTime = thisAmount.addTo(dateTime);
424      *   dateTime = dateTime.plus(thisAmount);
425      * </pre>
426      * <p>
427      * Only non-zero amounts will be added.
428      * <p>
429      * This instance is immutable and unaffected by this method call.
430      *
431      * @param temporal  the temporal object to adjust, not null
432      * @return an object of the same type with the adjustment made, not null
433      * @throws DateTimeException if unable to add
434      * @throws UnsupportedTemporalTypeException if the YEARS unit is not supported
435      * @throws ArithmeticException if numeric overflow occurs
436      */
437     @Override
438     public Temporal addTo(Temporal temporal) {
439         if (years != 0) {
440             temporal = temporal.plus(years, YEARS);
441         }
442         return temporal;
443     }
444 
445     /**
446      * Subtracts this amount from the specified temporal object.
447      * <p>
448      * This returns a temporal object of the same observable type as the input
449      * with this amount subtracted.
450      * <p>
451      * In most cases, it is clearer to reverse the calling pattern by using
452      * {@link Temporal#minus(TemporalAmount)}.
453      * <pre>
454      *   // these two lines are equivalent, but the second approach is recommended
455      *   dateTime = thisAmount.subtractFrom(dateTime);
456      *   dateTime = dateTime.minus(thisAmount);
457      * </pre>
458      * <p>
459      * Only non-zero amounts will be subtracted.
460      * <p>
461      * This instance is immutable and unaffected by this method call.
462      *
463      * @param temporal  the temporal object to adjust, not null
464      * @return an object of the same type with the adjustment made, not null
465      * @throws DateTimeException if unable to subtract
466      * @throws UnsupportedTemporalTypeException if the YEARS unit is not supported
467      * @throws ArithmeticException if numeric overflow occurs
468      */
469     @Override
470     public Temporal subtractFrom(Temporal temporal) {
471         if (years != 0) {
472             temporal = temporal.minus(years, YEARS);
473         }
474         return temporal;
475     }

477     //-----------------------------------------------------------------------
478     /**
479      * Compares this amount to the specified {@code Years}.
480      * <p>
481      * The comparison is based on the total length of the amounts.
482      * It is "consistent with equals", as defined by {@link Comparable}.
483      *
484      * @param otherAmount  the other amount, not null
485      * @return the comparator value, negative if less, positive if greater
486      */
487     @Override
488     public int compareTo(Years otherAmount) {
489         int thisValue = this.years;
490         int otherValue = otherAmount.years;
491         return Integer.compare(thisValue, otherValue);
492     }

494     //-----------------------------------------------------------------------
495     /**
496      * Checks if this amount is equal to the specified {@code Years}.
497      * <p>
498      * The comparison is based on the total length of the durations.
499      *
500      * @param otherAmount  the other amount, null returns false
501      * @return true if the other amount is equal to this one
502      */
503     @Override
504     public boolean equals(Object otherAmount) {
505         if (this == otherAmount) {
506             return true;
507         }
508         if (otherAmount instanceof Years) {
509             Years other = (Years) otherAmount;
510             return this.years == other.years;
511         }
512         return false;
513     }

516      * A hash code for this amount.

518      * @return a suitable hash code

522         return years;
523     }

534         return "P" + years + "Y";
