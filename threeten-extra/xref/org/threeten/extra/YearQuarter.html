75  /**
76   * A year-quarter in the ISO-8601 calendar system, such as {@code 2007-Q2}.
77   * <p>
78   * {@code YearQuarter} is an immutable date-time object that represents the combination
79   * of a year and quarter. Any field that can be derived from a year and quarter can be obtained.
80   * A quarter is defined by {@link Quarter} and {@link Month#firstMonthOfQuarter()} - Q1, Q2, Q3 and Q4.
81   * Q1 is January to March, Q2 is April to June, Q3 is July to September and Q4 is October to December.
82   * <p>
83   * This class does not store or represent a day, time or time-zone.
84   * For example, the value "2nd quarter 2007" can be stored in a {@code YearQuarter}.
85   * <p>
86   * The ISO-8601 calendar system is the modern civil calendar system used today
87   * in most of the world. It is equivalent to the proleptic Gregorian calendar
88   * system, in which today's rules for leap years are applied for all time.
89   * For most applications written today, the ISO-8601 rules are entirely suitable.
90   * However, any application that makes use of historical dates, and requires them
91   * to be accurate will find the ISO-8601 approach unsuitable.
92   * Note that the ISO-8601 standard does not define or refer to quarters.
93   *
94   * <h3>Implementation Requirements:</h3>
95   * This class is immutable and thread-safe.
96   * <p>
97   * This class must be treated as a value type. Do not synchronize, rely on the
98   * identity hash code or use the distinction between equals() and ==.
99   */
100 public final class YearQuarter
101         implements Temporal, TemporalAdjuster, Comparable<YearQuarter>, Serializable {

203     //-----------------------------------------------------------------------
204     /**
205      * Obtains an instance of {@code YearQuarter} from a temporal object.
206      * <p>
207      * This obtains a year-quarter based on the specified temporal.
208      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
209      * which this factory converts to an instance of {@code YearQuarter}.
210      * <p>
211      * The conversion extracts the {@link ChronoField#YEAR YEAR} and
212      * {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} fields.
213      * The extraction is only permitted if the temporal object has an ISO
214      * chronology, or can be converted to a {@code LocalDate}.
215      * <p>
216      * This method matches the signature of the functional interface {@link TemporalQuery}
217      * allowing it to be used in queries via method reference, {@code YearQuarter::from}.
218      *
219      * @param temporal  the temporal object to convert, not null
220      * @return the year-quarter, not null
221      * @throws DateTimeException if unable to convert to a {@code YearQuarter}
222      */
223     public static YearQuarter from(TemporalAccessor temporal) {
224         if (temporal instanceof YearQuarter) {
225             return (YearQuarter) temporal;
226         }
227         Objects.requireNonNull(temporal, "temporal");
228         try {
229             if (IsoChronology.INSTANCE.equals(Chronology.from(temporal)) == false) {
230                 temporal = LocalDate.from(temporal);
231             }
232             return of(temporal.get(YEAR), temporal.get(QUARTER_OF_YEAR));
233         } catch (DateTimeException ex) {
234             throw new DateTimeException("Unable to obtain YearQuarter from TemporalAccessor: " +
235                     temporal + " of type " + temporal.getClass().getName(), ex);
236         }
237     }

307     /**
308      * Checks if the specified field is supported.
309      * <p>
310      * This checks if this year-quarter can be queried for the specified field.
311      * If false, then calling the {@link #range(TemporalField) range},
312      * {@link #get(TemporalField) get} and {@link #with(TemporalField, long)}
313      * methods will throw an exception.
314      * <p>
315      * If the field is a {@link ChronoField} then the query is implemented here.
316      * The supported fields are:
317      * <ul>
318      * <li>{@code QUARTER_OF_YEAR}
319      * <li>{@code YEAR_OF_ERA}
320      * <li>{@code YEAR}
321      * <li>{@code ERA}
322      * </ul>
323      * All other {@code ChronoField} instances will return false.
324      * <p>
325      * If the field is not a {@code ChronoField}, then the result of this method
326      * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
327      * passing {@code this} as the argument.
328      * Whether the field is supported is determined by the field.
329      *
330      * @param field  the field to check, null returns false
331      * @return true if the field is supported on this year-quarter, false if not
332      */
333     @Override
334     public boolean isSupported(TemporalField field) {
335         if (field == QUARTER_OF_YEAR) {
336             return true;
337         } else if (field instanceof ChronoField) {
338             return field == YEAR || field == YEAR_OF_ERA || field == ERA;
339         }
340         return field != null && field.isSupportedBy(this);
341     }

343     /**
344      * Checks if the specified unit is supported.
345      * <p>
346      * This checks if the specified unit can be added to, or subtracted from, this date-time.
347      * If false, then calling the {@link #plus(long, TemporalUnit)} and
348      * {@link #minus(long, TemporalUnit) minus} methods will throw an exception.
349      * <p>
350      * If the unit is a {@link ChronoUnit} then the query is implemented here.
351      * The supported units are:
352      * <ul>
353      * <li>{@code QUARTER_YEARS}
354      * <li>{@code YEARS}
355      * <li>{@code DECADES}
356      * <li>{@code CENTURIES}
357      * <li>{@code MILLENNIA}
358      * <li>{@code ERAS}
359      * </ul>
360      * All other {@code ChronoUnit} instances will return false.
361      * <p>
362      * If the unit is not a {@code ChronoUnit}, then the result of this method
363      * is obtained by invoking {@code TemporalUnit.isSupportedBy(Temporal)}
364      * passing {@code this} as the argument.
365      * Whether the unit is supported is determined by the unit.
366      *
367      * @param unit  the unit to check, null returns false
368      * @return true if the unit can be added/subtracted, false if not
369      */
370     @Override
371     public boolean isSupported(TemporalUnit unit) {
372         if (unit == QUARTER_YEARS) {
373             return true;
374         } else if (unit instanceof ChronoUnit) {
375             return unit == YEARS || unit == DECADES || unit == CENTURIES || unit == MILLENNIA || unit == ERAS;
376         }
377         return unit != null && unit.isSupportedBy(this);
378     }

385      * This year-quarter is used to enhance the accuracy of the returned range.
386      * If it is not possible to return the range, because the field is not supported

389      * If the field is a {@link ChronoField} then the query is implemented here.

392      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
393      * <p>
394      * If the field is not a {@code ChronoField}, then the result of this method
395      * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
396      * passing {@code this} as the argument.
397      * Whether the range can be obtained is determined by the field.
398      *
399      * @param field  the field to query the range for, not null
400      * @return the range of valid values for the field, not null
401      * @throws DateTimeException if the range for the field cannot be obtained
402      * @throws UnsupportedTemporalTypeException if the field is not supported

406         if (field == YEAR_OF_ERA) {
407             return (getYear() <= 0 ? ValueRange.of(1, Year.MAX_VALUE + 1) : ValueRange.of(1, Year.MAX_VALUE));
408         }
409         return Temporal.super.range(field);

443     /**
444      * Gets the value of the specified field from this year-quarter as a {@code long}.
445      * <p>
446      * This queries this year-quarter for the value for the specified field.
447      * If it is not possible to return the value, because the field is not supported
448      * or for some other reason, an exception is thrown.
449      * <p>
450      * If the field is a {@link ChronoField} then the query is implemented here.
451      * The {@link #isSupported(TemporalField) supported fields} will return valid
452      * values based on this year-quarter.
453      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
454      * <p>
455      * If the field is not a {@code ChronoField}, then the result of this method
456      * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
457      * passing {@code this} as the argument. Whether the value can be obtained,
458      * and what the value represents, is determined by the field.
459      *
460      * @param field  the field to get, not null
461      * @return the value for the field
462      * @throws DateTimeException if a value for the field cannot be obtained
463      * @throws UnsupportedTemporalTypeException if the field is not supported
464      * @throws ArithmeticException if numeric overflow occurs
465      */
466     @Override
467     public long getLong(TemporalField field) {
468         if (field == QUARTER_OF_YEAR) {
469             return quarter.getValue();
470         } else if (field instanceof ChronoField) {
471             switch ((ChronoField) field) {
472                 case YEAR_OF_ERA:
473                     return (year < 1 ? 1 - year : year);
474                 case YEAR:
475                     return year;
476                 case ERA:
477                     return (year < 1 ? 0 : 1);
478                 default:
479                     throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
480             }
481         }
482         return field.getFrom(this);
483     }

533     /**
534      * Checks if the year is a leap year, according to the ISO proleptic
535      * calendar system rules.
536      * <p>
537      * This method applies the current rules for leap years across the whole time-line.
538      * In general, a year is a leap year if it is divisible by four without
539      * remainder. However, years divisible by 100, are not leap years, with
540      * the exception of years divisible by 400 which are.
541      * <p>
542      * For example, 1904 is a leap year it is divisible by 4.
543      * 1900 was not a leap year as it is divisible by 100, however 2000 was a
544      * leap year as it is divisible by 400.
545      * <p>
546      * The calculation is proleptic - applying the same rules into the far future and far past.
547      * This is historically inaccurate, but is correct for the ISO-8601 standard.
548      *
549      * @return true if the year is leap, false otherwise
550      */
551     public boolean isLeapYear() {
552         return IsoChronology.INSTANCE.isLeapYear(year);
553     }

555     /**
556      * Checks if the day-of-quarter is valid for this year-quarter.
557      * <p>
558      * This method checks whether this year and quarter and the input day form
559      * a valid date.
560      *
561      * @param dayOfQuarter  the day-of-quarter to validate, from 1 to 92, invalid value returns false
562      * @return true if the day is valid for this year-quarter
563      */
564     public boolean isValidDay(int dayOfQuarter) {
565         return dayOfQuarter >= 1 && dayOfQuarter <= lengthOfQuarter();
566     }

591     /**
592      * Returns an adjusted copy of this year-quarter.
593      * <p>
594      * This returns a {@code YearQuarter}, based on this one, with the year-quarter adjusted.
595      * The adjustment takes place using the specified adjuster strategy object.
596      * Read the documentation of the adjuster to understand what adjustment will be made.
597      * <p>
598      * A simple adjuster might simply set the one of the fields, such as the year field.
599      * A more complex adjuster might set the year-quarter to the next quarter that
600      * Halley's comet will pass the Earth.
601      * <p>
602      * The result of this method is obtained by invoking the
603      * {@link TemporalAdjuster#adjustInto(Temporal)} method on the
604      * specified adjuster passing {@code this} as the argument.
605      * <p>
606      * This instance is immutable and unaffected by this method call.
607      *
608      * @param adjuster the adjuster to use, not null
609      * @return a {@code YearQuarter} based on {@code this} with the adjustment made, not null
610      * @throws DateTimeException if the adjustment cannot be made

618     /**
619      * Returns a copy of this year-quarter with the specified field set to a new value.
620      * <p>
621      * This returns a {@code YearQuarter}, based on this one, with the value
622      * for the specified field changed.
623      * This can be used to change any supported field, such as the year or quarter.
624      * If it is not possible to set the value, because the field is not supported or for
625      * some other reason, an exception is thrown.
626      * <p>
627      * If the field is a {@link ChronoField} then the adjustment is implemented here.
628      * The supported fields behave as follows:
629      * <ul>
630      * <li>{@code QUARTER_OF_YEAR} -
631      *  Returns a {@code YearQuarter} with the specified quarter-of-year.
632      *  The year will be unchanged.
633      * <li>{@code YEAR_OF_ERA} -
634      *  Returns a {@code YearQuarter} with the specified year-of-era
635      *  The quarter and era will be unchanged.
636      * <li>{@code YEAR} -
637      *  Returns a {@code YearQuarter} with the specified year.
638      *  The quarter will be unchanged.
639      * <li>{@code ERA} -
640      *  Returns a {@code YearQuarter} with the specified era.
641      *  The quarter and year-of-era will be unchanged.
642      * </ul>
643      * <p>
644      * In all cases, if the new value is outside the valid range of values for the field
645      * then a {@code DateTimeException} will be thrown.
646      * <p>
647      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
648      * <p>
649      * If the field is not a {@code ChronoField}, then the result of this method
650      * is obtained by invoking {@code TemporalField.adjustInto(Temporal, long)}
651      * passing {@code this} as the argument. In this case, the field determines
652      * whether and how to adjust the instant.
653      * <p>
654      * This instance is immutable and unaffected by this method call.
655      *
656      * @param field  the field to set in the result, not null
657      * @param newValue  the new value of the field in the result
658      * @return a {@code YearQuarter} based on {@code this} with the specified field set, not null
659      * @throws DateTimeException if the field cannot be set
660      * @throws UnsupportedTemporalTypeException if the field is not supported
661      * @throws ArithmeticException if numeric overflow occurs
662      */
663     @Override
664     public YearQuarter with(TemporalField field, long newValue) {
665         if (field == QUARTER_OF_YEAR) {
666             return withQuarter(QUARTER_OF_YEAR.range().checkValidIntValue(newValue, QUARTER_OF_YEAR));
667         } else if (field instanceof ChronoField) {
668             ChronoField f = (ChronoField) field;
669             f.checkValidValue(newValue);
670             switch (f) {
671                 case YEAR_OF_ERA:
672                     return withYear((int) (year < 1 ? 1 - newValue : newValue));
673                 case YEAR:
674                     return withYear((int) newValue);
675                 case ERA:
676                     return (getLong(ERA) == newValue ? this : withYear(1 - year));
677                 default:
678                     throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
679             }
680         }
681         return field.adjustInto(this, newValue);
682     }

739     /**
740      * Returns a copy of this year-quarter with the specified amount added.
741      * <p>
742      * This returns a {@code YearQuarter}, based on this one, with the amount
743      * in terms of the unit added. If it is not possible to add the amount, because the
744      * unit is not supported or for some other reason, an exception is thrown.
745      * <p>
746      * If the field is a {@link ChronoUnit} then the addition is implemented here.
747      * The supported fields behave as follows:
748      * <ul>
749      * <li>{@code QUARTER_YEARS} -
750      *  Returns a {@code YearQuarter} with the specified number of quarters added.
751      *  This is equivalent to {@link #plusQuarters(long)}.
752      * <li>{@code YEARS} -
753      *  Returns a {@code YearQuarter} with the specified number of years added.
754      *  This is equivalent to {@link #plusYears(long)}.
755      * <li>{@code DECADES} -
756      *  Returns a {@code YearQuarter} with the specified number of decades added.
757      *  This is equivalent to calling {@link #plusYears(long)} with the amount
758      *  multiplied by 10.
759      * <li>{@code CENTURIES} -
760      *  Returns a {@code YearQuarter} with the specified number of centuries added.
761      *  This is equivalent to calling {@link #plusYears(long)} with the amount
762      *  multiplied by 100.
763      * <li>{@code MILLENNIA} -
764      *  Returns a {@code YearQuarter} with the specified number of millennia added.
765      *  This is equivalent to calling {@link #plusYears(long)} with the amount
766      *  multiplied by 1,000.
767      * <li>{@code ERAS} -
768      *  Returns a {@code YearQuarter} with the specified number of eras added.
769      *  Only two eras are supported so the amount must be one, zero or minus one.
770      *  If the amount is non-zero then the year is changed such that the year-of-era
771      *  is unchanged.
772      * </ul>
773      * <p>
774      * All other {@code ChronoUnit} instances will throw an {@code UnsupportedTemporalTypeException}.
775      * <p>
776      * If the field is not a {@code ChronoUnit}, then the result of this method
777      * is obtained by invoking {@code TemporalUnit.addTo(Temporal, long)}
778      * passing {@code this} as the argument. In this case, the unit determines
779      * whether and how to perform the addition.
780      * <p>
781      * This instance is immutable and unaffected by this method call.
782      *
783      * @param amountToAdd  the amount of the unit to add to the result, may be negative
784      * @param unit  the unit of the amount to add, not null
785      * @return a {@code YearQuarter} based on this year-quarter with the specified amount added, not null
786      * @throws DateTimeException if the addition cannot be made
787      * @throws UnsupportedTemporalTypeException if the unit is not supported
788      * @throws ArithmeticException if numeric overflow occurs
789      */
790     @Override
791     public YearQuarter plus(long amountToAdd, TemporalUnit unit) {
792         if (unit == QUARTER_YEARS) {
793             return plusQuarters(amountToAdd);
794         } else if (unit instanceof ChronoUnit) {
795             switch ((ChronoUnit) unit) {
796                 case YEARS:
797                     return plusYears(amountToAdd);
798                 case DECADES:
799                     return plusYears(Math.multiplyExact(amountToAdd, 10));
800                 case CENTURIES:
801                     return plusYears(Math.multiplyExact(amountToAdd, 100));
802                 case MILLENNIA:
803                     return plusYears(Math.multiplyExact(amountToAdd, 1000));
804                 case ERAS:
805                     return with(ERA, Math.addExact(getLong(ERA), amountToAdd));
806                 default:
807                     throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
808             }
809         }
810         return unit.addTo(this, amountToAdd);
811     }

956     /**
957      * Adjusts the specified temporal object to have this year-quarter.
958      * <p>
959      * This returns a temporal object of the same observable type as the input
960      * with the year and quarter changed to be the same as this.
961      * <p>
962      * The adjustment is equivalent to using {@link Temporal#plus(long, TemporalUnit)}
963      * passing the number of quarters to adjust by.
964      * If the specified temporal object does not use the ISO calendar system then
965      * a {@code DateTimeException} is thrown.
966      * <p>
967      * In most cases, it is clearer to reverse the calling pattern by using
968      * {@link Temporal#with(TemporalAdjuster)}:
969      * <pre>
970      *   // these two lines are equivalent, but the second approach is recommended
971      *   temporal = thisYearQuarter.adjustInto(temporal);
972      *   temporal = temporal.with(thisYearQuarter);
973      * </pre>
974      * <p>
975      * This instance is immutable and unaffected by this method call.
976      *
977      * @param temporal  the target object to be adjusted, not null
978      * @return the adjusted object, not null
979      * @throws DateTimeException if unable to make the adjustment
980      * @throws ArithmeticException if numeric overflow occurs
981      */
982     @Override
983     public Temporal adjustInto(Temporal temporal) {
984         if (Chronology.from(temporal).equals(IsoChronology.INSTANCE) == false) {
985             throw new DateTimeException("Adjustment only supported on ISO date-time");
986         }
987         long newProlepticQuarter = getProlepticQuarter();
988         long oldProlepticQuarter = temporal.get(YEAR) * 4L + (temporal.get(QUARTER_OF_YEAR) - 1);
989         return temporal.plus(newProlepticQuarter - oldProlepticQuarter, QUARTER_YEARS);
990     }

992     /**
993      * Calculates the amount of time until another year-quarter in terms of the specified unit.
994      * <p>
995      * This calculates the amount of time between two {@code YearQuarter}
996      * objects in terms of a single {@code TemporalUnit}.
997      * The start and end points are {@code this} and the specified year-quarter.
998      * The result will be negative if the end is before the start.
999      * The {@code Temporal} passed to this method is converted to a
1000      * {@code YearQuarter} using {@link #from(TemporalAccessor)}.
1001      * For example, the period in years between two year-quarters can be calculated
1002      * using {@code startYearQuarter.until(endYearQuarter, YEARS)}.
1003      * <p>
1004      * The calculation returns a whole number, representing the number of
1005      * complete units between the two year-quarters.
1006      * For example, the period in decades between 2012-Q3 and 2032-Q2
1007      * will only be one decade as it is one quarter short of two decades.
1008      * <p>
1009      * There are two equivalent ways of using this method.
1010      * The first is to invoke this method.
1011      * The second is to use {@link TemporalUnit#between(Temporal, Temporal)}:
1012      * <pre>
1013      *   // these two lines are equivalent
1014      *   amount = start.until(end, QUARTER_YEARS);
1015      *   amount = QUARTER_YEARS.between(start, end);
1016      * </pre>
1017      * The choice should be made based on which makes the code more readable.
1018      * <p>
1019      * The calculation is implemented in this method for {@link ChronoUnit}.
1020      * The units {@code QUARTER_YEARS}, {@code YEARS}, {@code DECADES},
1021      * {@code CENTURIES}, {@code MILLENNIA} and {@code ERAS} are supported.
1022      * Other {@code ChronoUnit} values will throw an exception.
1023      * <p>
1024      * If the unit is not a {@code ChronoUnit}, then the result of this method
1025      * is obtained by invoking {@code TemporalUnit.between(Temporal, Temporal)}
1026      * passing {@code this} as the first argument and the converted input temporal
1027      * as the second argument.
1028      * <p>
1029      * This instance is immutable and unaffected by this method call.
1030      *
1031      * @param endExclusive  the end date, exclusive, which is converted to a {@code YearQuarter}, not null
1032      * @param unit  the unit to measure the amount in, not null
1033      * @return the amount of time between this year-quarter and the end year-quarter
1034      * @throws DateTimeException if the amount cannot be calculated, or the end
1035      *  temporal cannot be converted to a {@code YearQuarter}
1036      * @throws UnsupportedTemporalTypeException if the unit is not supported
1037      * @throws ArithmeticException if numeric overflow occurs
1038      */
1039     @Override
1040     public long until(Temporal endExclusive, TemporalUnit unit) {
1041         YearQuarter end = YearQuarter.from(endExclusive);
1042         long quartersUntil = end.getProlepticQuarter() - getProlepticQuarter();  // no overflow
1043         if (unit == QUARTER_YEARS) {
1044             return quartersUntil;
1045         } else if (unit instanceof ChronoUnit) {
1046             switch ((ChronoUnit) unit) {
1047                 case YEARS:
1048                     return quartersUntil / 4;
1049                 case DECADES:
1050                     return quartersUntil / 40;
1051                 case CENTURIES:
1052                     return quartersUntil / 400;
1053                 case MILLENNIA:
1054                     return quartersUntil / 4000;
1055                 case ERAS:
1056                     return end.getLong(ERA) - getLong(ERA);
1057                 default:
1058                     throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
1059             }
1060         }
1061         return unit.between(this, end);
1062     }

1083      * <p>
