32  package org.threeten.bp.temporal;
33  
34  import org.threeten.bp.DateTimeException;
35  import org.threeten.bp.Duration;
36  import org.threeten.bp.Period;
37  
38  /**
39   * A unit of date-time, such as Days or Hours.
40   * <p>
41   * Measurement of time is built on units, such as years, months, days, hours, minutes and seconds.
42   * Implementations of this interface represent those units.
43   * <p>
44   * An instance of this interface represents the unit itself, rather than an amount of the unit.
45   * See {@link Period} for a class that represents an amount in terms of the common units.
46   * <p>
47   * The most commonly used units are defined in {@link ChronoUnit}.
48   * Further units are supplied in {@link IsoFields}.
49   * Units can also be written by application code by implementing this interface.
50   * <p>
51   * The unit works using double dispatch. Client code calls methods on a date-time like
52   * {@code LocalDateTime} which check if the unit is a {@code ChronoUnit}.
53   * If it is, then the date-time must handle it.
54   * Otherwise, the method call is re-dispatched to the matching method in this interface.
55   *
56   * <h3>Specification for implementors</h3>
57   * This interface must be implemented with care to ensure other classes operate correctly.
58   * All implementations that can be instantiated must be final, immutable and thread-safe.
59   * It is recommended to use an enum where possible.
60   */
61  public interface TemporalUnit {
62  
63      /**
64       * Gets the duration of this unit, which may be an estimate.
65       * <p>
66       * All units return a duration measured in standard nanoseconds from this method.
67       * The duration will be positive and non-zero.
68       * For example, an hour has a duration of {@code 60 * 60 * 1,000,000,000ns}.
69       * <p>
70       * Some units may return an accurate duration while others return an estimate.
71       * For example, days have an estimated duration due to the possibility of
72       * daylight saving time changes.
73       * To determine if the duration is an estimate, use {@link #isDurationEstimated()}.
74       *
75       * @return the duration of this unit, which may be an estimate, not null
76       */
77      Duration getDuration();
78  
79      /**
80       * Checks if the duration of the unit is an estimate.
81       * <p>
82       * All units have a duration, however the duration is not always accurate.
83       * For example, days have an estimated duration due to the possibility of
84       * daylight saving time changes.
85       * This method returns true if the duration is an estimate and false if it is
86       * accurate. Note that accurate/estimated ignores leap seconds.
87       *
88       * @return true if the duration is estimated, false if accurate
89       */
90      boolean isDurationEstimated();
91  
92      //-----------------------------------------------------------------------
93      /**
94       * Checks if this unit is date-based.
95       * 
96       * @return true if date-based
97       */
98      boolean isDateBased();
99  
100     /**
101      * Checks if this unit is time-based.
102      * 
103      * @return true if time-based
104      */
105     boolean isTimeBased();
106 
107     //-----------------------------------------------------------------------
108     /**
109      * Checks if this unit is supported by the specified temporal object.
110      * <p>
111      * This checks that the implementing date-time can add/subtract this unit.
112      * This can be used to avoid throwing an exception.
113      *
114      * @param temporal  the temporal object to check, not null
115      * @return true if the unit is supported
116      */
117     boolean isSupportedBy(Temporal temporal);
118 
119     /**
120      * Returns a copy of the specified temporal object with the specified period added.
121      * <p>
122      * The period added is a multiple of this unit. For example, this method
123      * could be used to add "3 days" to a date by calling this method on the
124      * instance representing "days", passing the date and the period "3".
125      * The period to be added may be negative, which is equivalent to subtraction.
126      * <p>
127      * There are two equivalent ways of using this method.
128      * The first is to invoke this method directly.
129      * The second is to use {@link Temporal#plus(long, TemporalUnit)}:
130      * <pre>
131      *   // these two lines are equivalent, but the second approach is recommended
132      *   temporal = thisUnit.doPlus(temporal);
133      *   temporal = temporal.plus(thisUnit);
134      * </pre>
135      * It is recommended to use the second approach, {@code plus(TemporalUnit)},
136      * as it is a lot clearer to read in code.
137      * <p>
138      * Implementations should perform any queries or calculations using the units
139      * available in {@link ChronoUnit} or the fields available in {@link ChronoField}.
140      * If the field is not supported a {@code DateTimeException} must be thrown.
141      * <p>
142      * Implementations must not alter the specified temporal object.
143      * Instead, an adjusted copy of the original must be returned.
144      * This provides equivalent, safe behavior for immutable and mutable implementations.
145      *
146      * @param <R>  the type of the Temporal object
147      * @param dateTime  the temporal object to adjust, not null
148      * @param periodToAdd  the period of this unit to add, positive or negative
149      * @return the adjusted temporal object, not null
150      * @throws DateTimeException if the period cannot be added
151      */
152     <R extends Temporal> R addTo(R dateTime, long periodToAdd);
153 
154     //-----------------------------------------------------------------------
155     /**
156      * Calculates the period in terms of this unit between two temporal objects of the same type.
157      * <p>
158      * This calculates the period between two temporals in terms of this unit.
159      * The start and end points are supplied as temporal objects and must be of the same type.
160      * The result will be negative if the end is before the start.
161      * For example, the period in hours between two temporal objects can be calculated
162      * using {@code HOURS.between(startTime, endTime)}.
163      * <p>
164      * The calculation returns a whole number, representing the number of complete units between the two temporals.
165      * For example, the period in hours between the times 11:30 and 13:29 will only b
166      * one hour as it is one minute short of two hours.
167      * <p>
168      * There are two equivalent ways of using this method.
169      * The first is to invoke this method directly.
170      * The second is to use {@link Temporal#until(Temporal, TemporalUnit)}:
171      * <pre>
172      *   // these two lines are equivalent
173      *   between = thisUnit.between(start, end);
174      *   between = start.until(end, thisUnit);
175      * </pre>
176      * The choice should be made based on which makes the code more readable. 
177      * <p>
178      * For example, this method allows the number of days between two dates to be calculated:
179      * <pre>
180      *   long daysBetween = DAYS.between(start, end);
181      *   // or alternatively
182      *   long daysBetween = start.until(end, DAYS);
183      * </pre>
184      * Implementations should perform any queries or calculations using the units available in
185      * {@link ChronoUnit} or the fields available in {@link ChronoField}.
186      * If the unit is not supported a DateTimeException must be thrown.
187      * Implementations must not alter the specified temporal objects.
188      *
189      * @param temporal1  the base temporal object, not null
190      * @param temporal2  the other temporal object, not null
191      * @return the period between temporal1 and temporal2 in terms of this unit;
192      *  positive if temporal2 is later than temporal1, negative if earlier
193      * @throws DateTimeException if the period cannot be calculated
194      * @throws ArithmeticException if numeric overflow occurs
195      */
196     long between(Temporal temporal1, Temporal temporal2);
197 
198     //-----------------------------------------------------------------------
199     /**
200      * Outputs this unit as a {@code String} using the name.
201      *
202      * @return the name of this unit, not null
203      */
204     @Override
205     String toString();
206 
207 }
