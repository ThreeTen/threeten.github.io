46  /**
47   * A set of date fields that provide access to Julian Days.
48   * <p>
49   * The Julian Day is a standard way of expressing date and time commonly used in the scientific community.
50   * It is expressed as a decimal number of whole days where days start at midday.
51   * This class represents variations on Julian Days that count whole days from midnight.
52   *
53   * <h3>Specification for implementors</h3>
54   * This is an immutable and thread-safe class.
55   */
56  public final class JulianFields {
57  
58      /**
59       * Julian Day field.
60       * <p>
61       * This is an integer-based version of the Julian Day Number.
62       * Julian Day is a well-known system that represents the count of whole days since day 0,
63       * which is defined to be January 1, 4713 BCE in the Julian calendar, and -4713-11-24 Gregorian.
64       * The field  has "JulianDay" as 'name', and 'DAYS' as 'baseUnit'.
65       * The field always refers to the local date-time, ignoring the offset or zone.
66       * <p>
67       * For date-times, 'JULIAN_DAY.getFrom()' assumes the same value from
68       * midnight until just before the next midnight.
69       * When 'JULIAN_DAY.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
70       * 'JULIAN_DAY.adjustInto()' and 'JULIAN_DAY.getFrom()' only apply to {@code Temporal} objects that
71       * can be converted into {@link ChronoField#EPOCH_DAY}.
72       * A {@link DateTimeException} is thrown for any other type of object.
73       * <p>
74       * <h3>Astronomical and Scientific Notes</h3>
75       * The standard astronomical definition uses a fraction to indicate the time-of-day,
76       * thus 3.25 would represent the time 18:00, since days start at midday.
77       * This implementation uses an integer and days starting at midnight.
78       * The integer value for the Julian Day Number is the astronomical Julian Day value at midday
79       * of the date in question.
80       * This amounts to the astronomical Julian Day, rounded to an integer {@code JDN = floor(JD + 0.5)}.
81       * <p>
82       * <pre>
83       *  | ISO date          |  Julian Day Number | Astronomical Julian Day |
84       *  | 1970-01-01T00:00  |         2,440,588  |         2,440,587.5     |
85       *  | 1970-01-01T06:00  |         2,440,588  |         2,440,587.75    |
86       *  | 1970-01-01T12:00  |         2,440,588  |         2,440,588.0     |
87       *  | 1970-01-01T18:00  |         2,440,588  |         2,440,588.25    |
88       *  | 1970-01-02T00:00  |         2,440,589  |         2,440,588.5     |
89       *  | 1970-01-02T06:00  |         2,440,589  |         2,440,588.75    |
90       *  | 1970-01-02T12:00  |         2,440,589  |         2,440,589.0     |
91       * </pre>
92       * <p>
93       * Julian Days are sometimes taken to imply Universal Time or UTC, but this
94       * implementation always uses the Julian Day number for the local date,
95       * regardless of the offset or time-zone.
96       */
97      public static final TemporalField JULIAN_DAY = Field.JULIAN_DAY;
98      /**
99       * Modified Julian Day field.
100      * <p>
101      * This is an integer-based version of the Modified Julian Day Number.
102      * Modified Julian Day (MJD) is a well-known system that counts days continuously.
103      * It is defined relative to astronomical Julian Day as  {@code MJD = JD - 2400000.5}.
104      * Each Modified Julian Day runs from midnight to midnight.
105      * The field always refers to the local date-time, ignoring the offset or zone.
106      * <p>
107      * For date-times, 'MODIFIED_JULIAN_DAY.getFrom()' assumes the same value from
108      * midnight until just before the next midnight.
109      * When 'MODIFIED_JULIAN_DAY.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
110      * 'MODIFIED_JULIAN_DAY.adjustInto()' and 'MODIFIED_JULIAN_DAY.getFrom()' only apply to {@code Temporal} objects
111      * that can be converted into {@link ChronoField#EPOCH_DAY}.
112      * A {@link DateTimeException} is thrown for any other type of object.
113      * <p>
114      * This implementation is an integer version of MJD with the decimal part rounded to floor.
115      * <p>
116      * <h3>Astronomical and Scientific Notes</h3>
117      * <pre>
118      *  | ISO date          | Modified Julian Day |      Decimal MJD |
119      *  | 1970-01-01T00:00  |             40,587  |       40,587.0   |
120      *  | 1970-01-01T06:00  |             40,587  |       40,587.25  |
121      *  | 1970-01-01T12:00  |             40,587  |       40,587.5   |
122      *  | 1970-01-01T18:00  |             40,587  |       40,587.75  |
123      *  | 1970-01-02T00:00  |             40,588  |       40,588.0   |
124      *  | 1970-01-02T06:00  |             40,588  |       40,588.25  |
125      *  | 1970-01-02T12:00  |             40,588  |       40,588.5   |
126      * </pre>
127      * <p>
128      * Modified Julian Days are sometimes taken to imply Universal Time or UTC, but this
129      * implementation always uses the Modified Julian Day for the local date,
130      * regardless of the offset or time-zone.
131      */
132     public static final TemporalField MODIFIED_JULIAN_DAY = Field.MODIFIED_JULIAN_DAY;
133     /**
134      * Rata Die field.
135      * <p>
136      * Rata Die counts whole days continuously starting day 1 at midnight at the beginning of 0001-01-01 (ISO).
137      * The field always refers to the local date-time, ignoring the offset or zone.
138      * <p>
139      * For date-times, 'RATA_DIE.getFrom()' assumes the same value from
140      * midnight until just before the next midnight.
141      * When 'RATA_DIE.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
142      * 'MODIFIED_JULIAN_DAY.adjustInto()' and 'RATA_DIE.getFrom()' only apply to {@code Temporal} objects
143      * that can be converted into {@link ChronoField#EPOCH_DAY}.
144      * A {@link DateTimeException} is thrown for any other type of object.
145      */
146     public static final TemporalField RATA_DIE = Field.RATA_DIE;
147 
148     /**
149      * Hidden implementation.
150      */
151     private static enum Field implements TemporalField {
152         /**
153          * Julian Day field.
154          */
155         // 719163L + 1721425L = 2440588L
156         JULIAN_DAY("JulianDay", DAYS, FOREVER, 2440588L),
157         /**
158          * Modified Julian Day field.
159          */
160         // 719163L - 678576L = 40587L
161         MODIFIED_JULIAN_DAY("ModifiedJulianDay", DAYS, FOREVER, 40587L),
162         /**
163          * Rata Die field.
164          */
165         RATA_DIE("RataDie", DAYS, FOREVER, 719163L),
166         // lots of others Truncated,Lilian, ANSI COBOL (also dotnet related), Excel?
167         ;
168 
169         private final String name;
170         private final TemporalUnit baseUnit;
171         private final TemporalUnit rangeUnit;
172         private final ValueRange range;
173         private final long offset;
174 
175         private Field(String name, TemporalUnit baseUnit, TemporalUnit rangeUnit, long offset) {
176             this.name = name;
177             this.baseUnit = baseUnit;
178             this.rangeUnit = rangeUnit;
179             this.range = ValueRange.of(-365243219162L + offset, 365241780471L + offset);
180             this.offset = offset;
181         }
182 
183         //-----------------------------------------------------------------------
184         @Override
185         public TemporalUnit getBaseUnit() {
186             return baseUnit;
187         }
188 
189         @Override
190         public TemporalUnit getRangeUnit() {
191             return rangeUnit;
192         }
193 
194         @Override
195         public ValueRange range() {
196             return range;
197         }
198 
199         @Override
200         public boolean isDateBased() {
201             return true;
202         }
203 
204         @Override
205         public boolean isTimeBased() {
206             return false;
207         }
208 
209         //-----------------------------------------------------------------------
210         @Override
211         public boolean isSupportedBy(TemporalAccessor temporal) {
212             return temporal.isSupported(EPOCH_DAY);
213         }
214 
215         @Override
216         public ValueRange rangeRefinedBy(TemporalAccessor temporal) {
217             if (isSupportedBy(temporal) == false) {
218                 throw new UnsupportedTemporalTypeException("Unsupported field: " + this);
219             }
220             return range();
221         }
222 
223         @Override
224         public long getFrom(TemporalAccessor temporal) {
225             return temporal.getLong(EPOCH_DAY) + offset;
226         }
227 
228         @SuppressWarnings("unchecked")
229         @Override
230         public <R extends Temporal> R adjustInto(R dateTime, long newValue) {
231             if (range().isValidValue(newValue) == false) {
232                 throw new DateTimeException("Invalid value: " + name + " " + newValue);
233             }
234             return (R) dateTime.with(EPOCH_DAY, Jdk8Methods.safeSubtract(newValue, offset));
235         }
236 
237         @Override
238         public String getDisplayName(Locale locale) {
239             Jdk8Methods.requireNonNull(locale, "locale");
240             return toString();
241         }
242 
243         //-----------------------------------------------------------------------
244         @Override
245         public TemporalAccessor resolve(Map<TemporalField, Long> fieldValues,
246                         TemporalAccessor partialTemporal, ResolverStyle resolverStyle) {
247             long value = fieldValues.remove(this);
248             Chronology chrono = Chronology.from(partialTemporal);
249             return chrono.dateEpochDay(Jdk8Methods.safeSubtract(value, offset));
250         }
251 
252         //-----------------------------------------------------------------------
253         @Override
254         public String toString() {
255             return name;
256         }
257 
258     }
