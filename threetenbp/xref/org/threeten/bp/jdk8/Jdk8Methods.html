53      /**
54       * Safely adds two int values.
55       *
56       * @param a  the first value
57       * @param b  the second value
58       * @return the result
59       * @throws ArithmeticException if the result overflows an int
60       */
61      public static int safeAdd(int a, int b) {
62          int sum = a + b;
63          // check for a change of sign in the result when the inputs have the same sign
64          if ((a ^ sum) < 0 && (a ^ b) >= 0) {
65              throw new ArithmeticException("Addition overflows an int: " + a + " + " + b);
66          }
67          return sum;
68      }

124      * Safely multiply one int by another.

127      * @param b  the second value
128      * @return the result
129      * @throws ArithmeticException if the result overflows an int

131     public static int safeMultiply(int a, int b) {
132         long total = (long) a * (long) b;
133         if (total < Integer.MIN_VALUE || total > Integer.MAX_VALUE) {
134             throw new ArithmeticException("Multiplication overflows an int: " + a + " * " + b);
135         }
136         return (int) total;
137     }

139     /**
140      * Safely multiply a long by an int.
141      *
142      * @param a  the first value
143      * @param b  the second value
144      * @return the new total
145      * @throws ArithmeticException if the result overflows a long
146      */
147     public static long safeMultiply(long a, int b) {
148         switch (b) {
149             case -1:
150                 if (a == Long.MIN_VALUE) {
151                     throw new ArithmeticException("Multiplication overflows a long: " + a + " * " + b);
152                 }
153                 return -a;
154             case 0:
155                 return 0L;
156             case 1:
157                 return a;
158         }
159         long total = a * b;
160         if (total / b != a) {
161             throw new ArithmeticException("Multiplication overflows a long: " + a + " * " + b);
162         }
163         return total;
164     }

166     /**
167      * Multiply two values throwing an exception if overflow occurs.
168      *
169      * @param a  the first value
170      * @param b  the second value
171      * @return the new total
172      * @throws ArithmeticException if the result overflows a long
173      */
174     public static long safeMultiply(long a, long b) {
175         if (b == 1) {
176             return a;
177         }
178         if (a == 1) {
179             return b;
180         }
181         if (a == 0 || b == 0) {
182             return 0;
183         }
184         long total = a * b;
185         if (total / b != a || (a == Long.MIN_VALUE && b == -1) || (b == Long.MIN_VALUE && a == -1)) {
186             throw new ArithmeticException("Multiplication overflows a long: " + a + " * " + b);
187         }
188         return total;
189     }
