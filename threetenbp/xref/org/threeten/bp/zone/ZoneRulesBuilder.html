56  /**
57   * A mutable builder used to create all the rules for a historic time-zone.
58   * <p>
59   * The rules of a time-zone describe how the offset changes over time.
60   * The rules are created by building windows on the time-line within which
61   * the different rules apply. The rules may be one of two kinds:
62   * <p><ul>
63   * <li>Fixed savings - A single fixed amount of savings from the standard offset will apply.</li>
64   * <li>Rules - A set of one or more rules describe how daylight savings changes during the window.</li>
65   * </ul><p>
66   *
67   * <h3>Specification for implementors</h3>
68   * This class is a mutable builder used to create zone instances.
69   * It must only be used from a single thread.
70   * The created instances are immutable and thread-safe.
71   */
72  class ZoneRulesBuilder {

75       * The list of windows.

77      private List<TZWindow> windowList = new ArrayList<ZoneRulesBuilder.TZWindow>();
78      /**
79       * A map for deduplicating the output.
80       */
81      private Map<Object, Object> deduplicateMap;

84      /**
85       * Constructs an instance of the builder that can be used to create zone rules.
86       * <p>
87       * The builder is used by adding one or more windows representing portions
88       * of the time-line. The standard offset from UTC/Greenwich will be constant
89       * within a window, although two adjacent windows can have the same standard offset.
90       * <p>
91       * Within each window, there can either be a
92       * {@link #setFixedSavingsToWindow fixed savings amount} or a
93       * {@link #addRuleToWindow list of rules}.
94       */
95      public ZoneRulesBuilder() {
96      }

98      //-----------------------------------------------------------------------
99      /**
100      * Adds a window to the builder that can be used to filter a set of rules.
101      * <p>
102      * This method defines and adds a window to the zone where the standard offset is specified.
103      * The window limits the effect of subsequent additions of transition rules
104      * or fixed savings. If neither rules or fixed savings are added to the window
105      * then the window will default to no savings.
106      * <p>
107      * Each window must be added sequentially, as the start instant of the window
108      * is derived from the until instant of the previous window.
109      *
110      * @param standardOffset  the standard offset, not null
111      * @param until  the date-time that the offset applies until, not null
112      * @param untilDefinition  the time type for the until date-time, not null
113      * @return this, for chaining
114      * @throws IllegalStateException if the window order is invalid
115      */
116     public ZoneRulesBuilder addWindow(
117             ZoneOffset standardOffset,
118             LocalDateTime until,
119             TimeDefinition untilDefinition) {
120         Jdk8Methods.requireNonNull(standardOffset, "standardOffset");
121         Jdk8Methods.requireNonNull(until, "until");
122         Jdk8Methods.requireNonNull(untilDefinition, "untilDefinition");
123         TZWindow window = new TZWindow(standardOffset, until, untilDefinition);
124         if (windowList.size() > 0) {
125             TZWindow previous = windowList.get(windowList.size() - 1);
126             window.validateWindowOrder(previous);
127         }
128         windowList.add(window);
129         return this;
130     }

132     /**
133      * Adds a window that applies until the end of time to the builder that can be
134      * used to filter a set of rules.
135      * <p>
136      * This method defines and adds a window to the zone where the standard offset is specified.
137      * The window limits the effect of subsequent additions of transition rules
138      * or fixed savings. If neither rules or fixed savings are added to the window
139      * then the window will default to no savings.
140      * <p>
141      * This must be added after all other windows.
142      * No more windows can be added after this one.
143      *
144      * @param standardOffset  the standard offset, not null
145      * @return this, for chaining
146      * @throws IllegalStateException if a forever window has already been added
147      */
148     public ZoneRulesBuilder addWindowForever(ZoneOffset standardOffset) {
149         return addWindow(standardOffset, LocalDateTime.MAX, TimeDefinition.WALL);
150     }

153     /**
154      * Sets the previously added window to have fixed savings.
155      * <p>
156      * Setting a window to have fixed savings simply means that a single daylight
157      * savings amount applies throughout the window. The window could be small,
158      * such as a single summer, or large, such as a multi-year daylight savings.
159      * <p>
160      * A window can either have fixed savings or rules but not both.
161      *
162      * @param fixedSavingAmountSecs  the amount of saving to use for the whole window, not null
163      * @return this, for chaining
164      * @throws IllegalStateException if no window has yet been added
165      * @throws IllegalStateException if the window already has rules
166      */
167     public ZoneRulesBuilder setFixedSavingsToWindow(int fixedSavingAmountSecs) {
168         if (windowList.isEmpty()) {
169             throw new IllegalStateException("Must add a window before setting the fixed savings");
170         }
171         TZWindow window = windowList.get(windowList.size() - 1);
172         window.setFixedSavings(fixedSavingAmountSecs);
173         return this;
174     }

203      * Adds a single transition rule to the current window.
204      * <p>
205      * This adds a rule such that the offset, expressed as a daylight savings amount,
206      * changes at the specified date-time.
207      *
208      * @param year  the year of the transition, from MIN_VALUE to MAX_VALUE
209      * @param month  the month of the transition, not null
210      * @param dayOfMonthIndicator  the day-of-month of the transition, adjusted by dayOfWeek,
211      *   from 1 to 31 adjusted later, or -1 to -28 adjusted earlier from the last day of the month
212      * @param time  the time that the transition occurs as defined by timeDefintion, not null
213      * @param timeEndOfDay  whether midnight is at the end of day
214      * @param timeDefinition  the definition of how to convert local to actual time, not null
215      * @param savingAmountSecs  the amount of saving from the standard offset after the transition in seconds
216      * @return this, for chaining
217      * @throws DateTimeException if a date-time field is out of range
218      * @throws IllegalStateException if no window has yet been added
219      * @throws IllegalStateException if the window already has fixed savings
220      * @throws IllegalStateException if the window has reached the maximum capacity of 2000 rules
221      */

223             int year,
224             Month month,
225             int dayOfMonthIndicator,
226             LocalTime time,
227             boolean timeEndOfDay,
228             TimeDefinition timeDefinition,
229             int savingAmountSecs) {
230         return addRuleToWindow(year, year, month, dayOfMonthIndicator, null, time, timeEndOfDay, timeDefinition, savingAmountSecs);
231     }

233     /**
234      * Adds a multi-year transition rule to the current window.
235      * <p>
236      * This adds a rule such that the offset, expressed as a daylight savings amount,
237      * changes at the specified date-time for each year in the range.
238      *
239      * @param startYear  the start year of the rule, from MIN_VALUE to MAX_VALUE
240      * @param endYear  the end year of the rule, from MIN_VALUE to MAX_VALUE
241      * @param month  the month of the transition, not null
242      * @param dayOfMonthIndicator  the day-of-month of the transition, adjusted by dayOfWeek,
243      *   from 1 to 31 adjusted later, or -1 to -28 adjusted earlier from the last day of the month
244      * @param dayOfWeek  the day-of-week to adjust to, null if day-of-month should not be adjusted
245      * @param time  the time that the transition occurs as defined by timeDefintion, not null
246      * @param timeEndOfDay  whether midnight is at the end of day
247      * @param timeDefinition  the definition of how to convert local to actual time, not null
248      * @param savingAmountSecs  the amount of saving from the standard offset after the transition in seconds
249      * @return this, for chaining
250      * @throws DateTimeException if a date-time field is out of range
251      * @throws IllegalArgumentException if the day of month indicator is invalid
252      * @throws IllegalArgumentException if the end of day midnight flag does not match the time
253      * @throws IllegalStateException if no window has yet been added
254      * @throws IllegalStateException if the window already has fixed savings
255      * @throws IllegalStateException if the window has reached the maximum capacity of 2000 rules
256      */
257     public ZoneRulesBuilder addRuleToWindow(
258             int startYear,
259             int endYear,
260             Month month,
261             int dayOfMonthIndicator,
262             DayOfWeek dayOfWeek,
263             LocalTime time,
264             boolean timeEndOfDay,
265             TimeDefinition timeDefinition,
266             int savingAmountSecs) {
267         Jdk8Methods.requireNonNull(month, "month");
268         Jdk8Methods.requireNonNull(time, "time");
269         Jdk8Methods.requireNonNull(timeDefinition, "timeDefinition");
270         YEAR.checkValidValue(startYear);
271         YEAR.checkValidValue(endYear);
272         if (dayOfMonthIndicator < -28 || dayOfMonthIndicator > 31 || dayOfMonthIndicator == 0) {
273             throw new IllegalArgumentException("Day of month indicator must be between -28 and 31 inclusive excluding zero");
274         }
275         if (timeEndOfDay && time.equals(LocalTime.MIDNIGHT) == false) {
276             throw new IllegalArgumentException("Time must be midnight when end of day flag is true");
277         }
278         if (windowList.isEmpty()) {
279             throw new IllegalStateException("Must add a window before adding a rule");
280         }
281         TZWindow window = windowList.get(windowList.size() - 1);
282         window.addRule(startYear, endYear, month, dayOfMonthIndicator, dayOfWeek, time, timeEndOfDay, timeDefinition, savingAmountSecs);
283         return this;
284     }

302     /**
303      * Completes the build converting the builder to a set of time-zone rules.
304      * <p>
305      * Calling this method alters the state of the builder.
306      * Further rules should not be added to this builder once this method is called.
307      *
308      * @param zoneId  the time-zone ID, not null
309      * @param deduplicateMap  a map for deduplicating the values, not null
310      * @return the zone rules, not null
311      * @throws IllegalStateException if no windows have been added
312      * @throws IllegalStateException if there is only one rule defined as being forever for any given window
313      */
314     ZoneRules toRules(String zoneId, Map<Object, Object> deduplicateMap) {
315         Jdk8Methods.requireNonNull(zoneId, "zoneId");
316         this.deduplicateMap = deduplicateMap;
317         if (windowList.isEmpty()) {
318             throw new IllegalStateException("No windows have been added to the builder");
319         }
320 
321         final List<ZoneOffsetTransition> standardTransitionList = new ArrayList<ZoneOffsetTransition>(4);
322         final List<ZoneOffsetTransition> transitionList = new ArrayList<ZoneOffsetTransition>(256);
323         final List<ZoneOffsetTransitionRule> lastTransitionRuleList = new ArrayList<ZoneOffsetTransitionRule>(2);
324 
325         // initialize the standard offset calculation
326         final TZWindow firstWindow = windowList.get(0);
327         ZoneOffset loopStandardOffset = firstWindow.standardOffset;
328         int loopSavings = 0;
329         if (firstWindow.fixedSavingAmountSecs != null) {
330             loopSavings = firstWindow.fixedSavingAmountSecs;
331         }
332         final ZoneOffset firstWallOffset = deduplicate(ZoneOffset.ofTotalSeconds(loopStandardOffset.getTotalSeconds() + loopSavings));
333         LocalDateTime loopWindowStart = deduplicate(LocalDateTime.of(Year.MIN_VALUE, 1, 1, 0, 0));
334         ZoneOffset loopWindowOffset = firstWallOffset;
335 
336         // build the windows and rules to interesting data
337         for (TZWindow window : windowList) {
338             // tidy the state
339             window.tidy(loopWindowStart.getYear());
340 
341             // calculate effective savings at the start of the window
342             Integer effectiveSavings = window.fixedSavingAmountSecs;
343             if (effectiveSavings == null) {
344                 // apply rules from this window together with the standard offset and
345                 // savings from the last window to find the savings amount applicable
346                 // at start of this window
347                 effectiveSavings = 0;
348                 for (TZRule rule : window.ruleList) {
349                     ZoneOffsetTransition trans = rule.toTransition(loopStandardOffset, loopSavings);
350                     if (trans.toEpochSecond() > loopWindowStart.toEpochSecond(loopWindowOffset)) {
351                         // previous savings amount found, which could be the savings amount at
352                         // the instant that the window starts (hence isAfter)
353                         break;
354                     }
355                     effectiveSavings = rule.savingAmountSecs;
356                 }
357             }
358 
359             // check if standard offset changed, and update it
360             if (loopStandardOffset.equals(window.standardOffset) == false) {
361                 standardTransitionList.add(deduplicate(
362                     new ZoneOffsetTransition(
363                         LocalDateTime.ofEpochSecond(loopWindowStart.toEpochSecond(loopWindowOffset), 0, loopStandardOffset),
364                         loopStandardOffset, window.standardOffset)));
365                 loopStandardOffset = deduplicate(window.standardOffset);
366             }
367 
368             // check if the start of the window represents a transition
369             ZoneOffset effectiveWallOffset = deduplicate(ZoneOffset.ofTotalSeconds(loopStandardOffset.getTotalSeconds() + effectiveSavings));
370             if (loopWindowOffset.equals(effectiveWallOffset) == false) {
371                 ZoneOffsetTransition trans = deduplicate(
372                     new ZoneOffsetTransition(loopWindowStart, loopWindowOffset, effectiveWallOffset));
373                 transitionList.add(trans);
374             }
375             loopSavings = effectiveSavings;
376 
377             // apply rules within the window
378             for (TZRule rule : window.ruleList) {
379                 ZoneOffsetTransition trans = deduplicate(rule.toTransition(loopStandardOffset, loopSavings));
380                 if (trans.toEpochSecond() < loopWindowStart.toEpochSecond(loopWindowOffset) == false &&
381                         trans.toEpochSecond() < window.createDateTimeEpochSecond(loopSavings) &&
382                         trans.getOffsetBefore().equals(trans.getOffsetAfter()) == false) {
383                     transitionList.add(trans);
384                     loopSavings = rule.savingAmountSecs;
385                 }
386             }
387 
388             // calculate last rules
389             for (TZRule lastRule : window.lastRuleList) {
390                 ZoneOffsetTransitionRule transitionRule = deduplicate(lastRule.toTransitionRule(loopStandardOffset, loopSavings));
391                 lastTransitionRuleList.add(transitionRule);
392                 loopSavings = lastRule.savingAmountSecs;
393             }
394 
395             // finally we can calculate the true end of the window, passing it to the next window
396             loopWindowOffset = deduplicate(window.createWallOffset(loopSavings));
397             loopWindowStart = deduplicate(LocalDateTime.ofEpochSecond(
398                     window.createDateTimeEpochSecond(loopSavings), 0, loopWindowOffset));
399         }
400         return new StandardZoneRules(
401                 firstWindow.standardOffset, firstWallOffset, standardTransitionList,
402                 transitionList, lastTransitionRuleList);
403     }

405     /**
406      * Deduplicates an object instance.
407      *
408      * @param <T> the generic type
409      * @param object  the object to deduplicate
410      * @return the deduplicated object
411      */
412     @SuppressWarnings("unchecked")
413     <T> T deduplicate(T object) {
414         if (deduplicateMap.containsKey(object) == false) {
415             deduplicateMap.put(object, object);
416         }
417         return (T) deduplicateMap.get(object);
418     }

421     /**
422      * A definition of a window in the time-line.
423      * The window will have one standard offset and will either have a
424      * fixed DST savings or a set of rules.
425      */
426     class TZWindow {
427         /** The standard offset during the window, not null. */
428         private final ZoneOffset standardOffset;
429         /** The end local time, not null. */
430         private final LocalDateTime windowEnd;
431         /** The type of the end time, not null. */
432         private final TimeDefinition timeDefinition;
433 
434         /** The fixed amount of the saving to be applied during this window. */
435         private Integer fixedSavingAmountSecs;
436         /** The rules for the current window. */
437         private List<TZRule> ruleList = new ArrayList<TZRule>();
438         /** The latest year that the last year starts at. */
439         private int maxLastRuleStartYear = Year.MIN_VALUE;
440         /** The last rules. */
441         private List<TZRule> lastRuleList = new ArrayList<TZRule>();
442 
443         /**
444          * Constructor.
445          *
446          * @param standardOffset  the standard offset applicable during the window, not null
447          * @param windowEnd  the end of the window, relative to the time definition, null if forever
448          * @param timeDefinition  the time definition for calculating the true end, not null
449          */
450         TZWindow(
451                 ZoneOffset standardOffset,
452                 LocalDateTime windowEnd,
453                 TimeDefinition timeDefinition) {
454             super();
455             this.windowEnd = windowEnd;
456             this.timeDefinition = timeDefinition;
457             this.standardOffset = standardOffset;
458         }
459 
460         /**
461          * Sets the fixed savings amount for the window.
462          *
463          * @param fixedSavingAmount  the amount of daylight saving to apply throughout the window, may be null
464          * @throws IllegalStateException if the window already has rules
465          */
466         void setFixedSavings(int fixedSavingAmount) {
467             if (ruleList.size() > 0 || lastRuleList.size() > 0) {
468                 throw new IllegalStateException("Window has DST rules, so cannot have fixed savings");
469             }
470             this.fixedSavingAmountSecs = fixedSavingAmount;
471         }
472 
473         /**
474          * Adds a rule to the current window.
475          *
476          * @param startYear  the start year of the rule, from MIN_VALUE to MAX_VALUE
477          * @param endYear  the end year of the rule, from MIN_VALUE to MAX_VALUE
478          * @param month  the month of the transition, not null
479          * @param dayOfMonthIndicator  the day-of-month of the transition, adjusted by dayOfWeek,
480          *   from 1 to 31 adjusted later, or -1 to -28 adjusted earlier from the last day of the month
481          * @param dayOfWeek  the day-of-week to adjust to, null if day-of-month should not be adjusted
482          * @param time  the time that the transition occurs as defined by timeDefintion, not null
483          * @param timeEndOfDay  whether midnight is at the end of day
484          * @param timeDefinition  the definition of how to convert local to actual time, not null
485          * @param savingAmountSecs  the amount of saving from the standard offset in seconds
486          * @throws IllegalStateException if the window already has fixed savings
487          * @throws IllegalStateException if the window has reached the maximum capacity of 2000 rules
488          */
489         void addRule(
490                 int startYear,
491                 int endYear,
492                 Month month,
493                 int dayOfMonthIndicator,
494                 DayOfWeek dayOfWeek,
495                 LocalTime time,
496                 boolean timeEndOfDay,
497                 TimeDefinition timeDefinition,
498                 int savingAmountSecs) {
499 
500             if (fixedSavingAmountSecs != null) {
501                 throw new IllegalStateException("Window has a fixed DST saving, so cannot have DST rules");
502             }
503             if (ruleList.size() >= 2000) {
504                 throw new IllegalStateException("Window has reached the maximum number of allowed rules");
505             }
506             boolean lastRule = false;
507             if (endYear == Year.MAX_VALUE) {
508                 lastRule = true;
509                 endYear = startYear;
510             }
511             int year = startYear;
512             while (year <= endYear) {
513                 TZRule rule = new TZRule(year, month, dayOfMonthIndicator, dayOfWeek, time, timeEndOfDay, timeDefinition, savingAmountSecs);
514                 if (lastRule) {
515                     lastRuleList.add(rule);
516                     maxLastRuleStartYear = Math.max(startYear, maxLastRuleStartYear);
517                 } else {
518                     ruleList.add(rule);
519                 }
520                 year++;
521             }
522         }
523 
524         /**
525          * Validates that this window is after the previous one.
526          *
527          * @param previous  the previous window, not null
528          * @throws IllegalStateException if the window order is invalid
529          */
530         void validateWindowOrder(TZWindow previous) {
531             if (windowEnd.isBefore(previous.windowEnd)) {
532                 throw new IllegalStateException("Windows must be added in date-time order: " +
533                         windowEnd + " < " + previous.windowEnd);
534             }
535         }
536 
537         /**
538          * Adds rules to make the last rules all start from the same year.
539          * Also add one more year to avoid weird case where penultimate year has odd offset.
540          *
541          * @param windowStartYear  the window start year
542          * @throws IllegalStateException if there is only one rule defined as being forever
543          */
544         void tidy(int windowStartYear) {
545             if (lastRuleList.size() == 1) {
546                 throw new IllegalStateException("Cannot have only one rule defined as being forever");
547             }
548 
549             // handle last rules
550             if (windowEnd.equals(LocalDateTime.MAX)) {
551                 // setup at least one real rule, which closes off other windows nicely
552                 maxLastRuleStartYear = Math.max(maxLastRuleStartYear, windowStartYear) + 1;
553                 for (TZRule lastRule : lastRuleList) {
554                     addRule(lastRule.year, maxLastRuleStartYear, lastRule.month, lastRule.dayOfMonthIndicator,
555                         lastRule.dayOfWeek, lastRule.time, lastRule.timeEndOfDay, lastRule.timeDefinition, lastRule.savingAmountSecs);
556                     lastRule.year = maxLastRuleStartYear + 1;
557                 }
558                 if (maxLastRuleStartYear == Year.MAX_VALUE) {
559                     lastRuleList.clear();
560                 } else {
561                     maxLastRuleStartYear++;
562                 }
563             } else {
564                 // convert all within the endYear limit
565                 int endYear = windowEnd.getYear();
566                 for (TZRule lastRule : lastRuleList) {
567                     addRule(lastRule.year, endYear + 1, lastRule.month, lastRule.dayOfMonthIndicator,
568                         lastRule.dayOfWeek, lastRule.time, lastRule.timeEndOfDay, lastRule.timeDefinition, lastRule.savingAmountSecs);
569                 }
570                 lastRuleList.clear();
571                 maxLastRuleStartYear = Year.MAX_VALUE;
572             }
573 
574             // ensure lists are sorted
575             Collections.sort(ruleList);
576             Collections.sort(lastRuleList);
577 
578             // default fixed savings to zero
579             if (ruleList.size() == 0 && fixedSavingAmountSecs == null) {
580                 fixedSavingAmountSecs = 0;
581             }
582         }
583 
584         /**
585          * Checks if the window is empty.
586          *
587          * @return true if the window is only a standard offset
588          */
589         boolean isSingleWindowStandardOffset() {
590             return windowEnd.equals(LocalDateTime.MAX) && timeDefinition == TimeDefinition.WALL &&
591                     fixedSavingAmountSecs == null && lastRuleList.isEmpty() && ruleList.isEmpty();
592         }

594         /**
595          * Creates the wall offset for the local date-time at the end of the window.
596          *
597          * @param savingsSecs  the amount of savings in use in seconds
598          * @return the created date-time epoch second in the wall offset, not null
599          */
600         ZoneOffset createWallOffset(int savingsSecs) {
601             return ZoneOffset.ofTotalSeconds(standardOffset.getTotalSeconds() + savingsSecs);
602         }
603 
604         /**
605          * Creates the offset date-time for the local date-time at the end of the window.
606          *
607          * @param savingsSecs  the amount of savings in use in seconds
608          * @return the created date-time epoch second in the wall offset, not null
609          */
610         long createDateTimeEpochSecond(int savingsSecs) {
611             ZoneOffset wallOffset = createWallOffset(savingsSecs);
612             LocalDateTime ldt = timeDefinition.createDateTime(windowEnd, standardOffset, wallOffset);
613             return ldt.toEpochSecond(wallOffset);
614         }
615     }
616 
617     //-----------------------------------------------------------------------
618     /**
619      * A definition of the way a local time can be converted to an offset time.
620      */
621     class TZRule implements Comparable<TZRule> {
622         /** The year. */
623         private int year;
624         /** The month. */
625         private Month month;
626         /** The day-of-month. */
627         private int dayOfMonthIndicator;
628         /** The day-of-month. */
629         private DayOfWeek dayOfWeek;
630         /** The local time. */
631         private LocalTime time;
632         /** Whether the local time is end of day. */
633         private boolean timeEndOfDay;
634         /** The type of the time. */
635         private TimeDefinition timeDefinition;
636         /** The amount of the saving to be applied after this point. */
637         private int savingAmountSecs;
638 
639         /**
640          * Constructor.
641          *
642          * @param year  the year
643          * @param month  the month, not null
644          * @param dayOfMonthIndicator  the day-of-month of the transition, adjusted by dayOfWeek,
645          *   from 1 to 31 adjusted later, or -1 to -28 adjusted earlier from the last day of the month
646          * @param dayOfWeek  the day-of-week, null if day-of-month is exact
647          * @param time  the time, not null
648          * @param timeEndOfDay  whether midnight is at the end of day
649          * @param timeDefinition  the time definition, not null
650          * @param savingAfterSecs  the savings amount in seconds
651          */
652         TZRule(int year, Month month, int dayOfMonthIndicator,
653                 DayOfWeek dayOfWeek, LocalTime time, boolean timeEndOfDay,
654                 TimeDefinition timeDefinition, int savingAfterSecs) {
655             super();
656             this.year = year;
657             this.month = month;
658             this.dayOfMonthIndicator = dayOfMonthIndicator;
659             this.dayOfWeek = dayOfWeek;
660             this.time = time;
661             this.timeEndOfDay = timeEndOfDay;
662             this.timeDefinition = timeDefinition;
663             this.savingAmountSecs = savingAfterSecs;
664         }
665 
666         /**
667          * Converts this to a transition.
668          *
669          * @param standardOffset  the active standard offset, not null
670          * @param savingsBeforeSecs  the active savings in seconds
671          * @return the transition, not null
672          */
673         ZoneOffsetTransition toTransition(ZoneOffset standardOffset, int savingsBeforeSecs) {
674             // copy of code in ZoneOffsetTransitionRule to avoid infinite loop
675             LocalDate date = toLocalDate();
676             date = deduplicate(date);
677             LocalDateTime ldt = deduplicate(LocalDateTime.of(date, time));
678             ZoneOffset wallOffset = deduplicate(ZoneOffset.ofTotalSeconds(standardOffset.getTotalSeconds() + savingsBeforeSecs));
679             LocalDateTime dt = deduplicate(timeDefinition.createDateTime(ldt, standardOffset, wallOffset));
680             ZoneOffset offsetAfter = deduplicate(ZoneOffset.ofTotalSeconds(standardOffset.getTotalSeconds() + savingAmountSecs));
681             return new ZoneOffsetTransition(dt, wallOffset, offsetAfter);
682         }
683 
684         /**
685          * Converts this to a transition rule.
686          *
687          * @param standardOffset  the active standard offset, not null
688          * @param savingsBeforeSecs  the active savings before the transition in seconds
689          * @return the transition, not null
690          */
691         ZoneOffsetTransitionRule toTransitionRule(ZoneOffset standardOffset, int savingsBeforeSecs) {
692             // optimize stored format
693             if (dayOfMonthIndicator < 0) {
694                 if (month != Month.FEBRUARY) {
695                     dayOfMonthIndicator = month.maxLength() - 6;
696                 }
697             }
698             if (timeEndOfDay && dayOfMonthIndicator > 0 && (dayOfMonthIndicator == 28 && month == Month.FEBRUARY) == false) {
699                 LocalDate date = LocalDate.of(2004, month, dayOfMonthIndicator).plusDays(1);  // leap-year
700                 month = date.getMonth();
701                 dayOfMonthIndicator = date.getDayOfMonth();
702                 if (dayOfWeek != null) {
703                     dayOfWeek = dayOfWeek.plus(1);
704                 }
705                 timeEndOfDay = false;
706             }
707 
708             // build rule
709             ZoneOffsetTransition trans = toTransition(standardOffset, savingsBeforeSecs);
710             return new ZoneOffsetTransitionRule(
711                     month, dayOfMonthIndicator, dayOfWeek, time, timeEndOfDay, timeDefinition,
712                     standardOffset, trans.getOffsetBefore(), trans.getOffsetAfter());
713         }
714 
715         public int compareTo(TZRule other) {
716             int cmp = year - other.year;
717             cmp = (cmp == 0 ? month.compareTo(other.month) : cmp);
718             if (cmp == 0) {
719                 // convert to date to handle dow/domIndicator/timeEndOfDay
720                 LocalDate thisDate = toLocalDate();
721                 LocalDate otherDate = other.toLocalDate();
722                 cmp = thisDate.compareTo(otherDate);
723             }
724             cmp = (cmp == 0 ? time.compareTo(other.time) : cmp);
725             return cmp;
726         }
727 
728         private LocalDate toLocalDate() {
729             LocalDate date;
730             if (dayOfMonthIndicator < 0) {
731                 int monthLen = month.length(IsoChronology.INSTANCE.isLeapYear(year));
732                 date = LocalDate.of(year, month, monthLen + 1 + dayOfMonthIndicator);
733                 if (dayOfWeek != null) {
734                     date = date.with(previousOrSame(dayOfWeek));
735                 }
736             } else {
737                 date = LocalDate.of(year, month, dayOfMonthIndicator);
738                 if (dayOfWeek != null) {
739                     date = date.with(nextOrSame(dayOfWeek));
740                 }
741             }
742             if (timeEndOfDay) {
743                 date = date.plusDays(1);
744             }
745             return date;
746         }
747 
748         // no equals() or hashCode()
749     }
