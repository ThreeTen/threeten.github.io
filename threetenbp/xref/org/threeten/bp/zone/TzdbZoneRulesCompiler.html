94      /**
95       * Reads a set of TZDB files and builds a single combined data file.
96       *
97       * @param args  the arguments
98       */
99      public static void main(String[] args) {
100         if (args.length < 2) {
101             outputHelp();
102             return;
103         }
104 
105         // parse args
106         String version = null;
107         File baseSrcDir = null;
108         File dstDir = null;
109         boolean unpacked = false;
110         boolean verbose = false;
111 
112         // parse options
113         int i;
114         for (i = 0; i < args.length; i++) {
115             String arg = args[i];
116             if (arg.startsWith("-") == false) {
117                 break;
118             }
119             if ("-srcdir".equals(arg)) {
120                 if (baseSrcDir == null && ++i < args.length) {
121                     baseSrcDir = new File(args[i]);
122                     continue;
123                 }
124             } else if ("-dstdir".equals(arg)) {
125                 if (dstDir == null && ++i < args.length) {
126                     dstDir = new File(args[i]);
127                     continue;
128                 }
129             } else if ("-version".equals(arg)) {
130                 if (version == null && ++i < args.length) {
131                     version = args[i];
132                     continue;
133                 }
134             } else if ("-unpacked".equals(arg)) {
135                 if (unpacked == false) {
136                     unpacked = true;
137                     continue;
138                 }
139             } else if ("-verbose".equals(arg)) {
140                 if (verbose == false) {
141                     verbose = true;
142                     continue;
143                 }
144             } else if ("-help".equals(arg) == false) {
145                 System.out.println("Unrecognised option: " + arg);
146             }
147             outputHelp();
148             return;
149         }
150 
151         // check source directory
152         if (baseSrcDir == null) {
153             System.out.println("Source directory must be specified using -srcdir: " + baseSrcDir);
154             return;
155         }
156         if (baseSrcDir.isDirectory() == false) {
157             System.out.println("Source does not exist or is not a directory: " + baseSrcDir);
158             return;
159         }
160         dstDir = (dstDir != null ? dstDir : baseSrcDir);
161 
162         // parse source file names
163         List<String> srcFileNames = Arrays.asList(Arrays.copyOfRange(args, i, args.length));
164         if (srcFileNames.isEmpty()) {
165             System.out.println("Source filenames not specified, using default set");
166             System.out.println("(africa antarctica asia australasia backward etcetera europe northamerica southamerica)");
167             srcFileNames = Arrays.asList("africa", "antarctica", "asia", "australasia", "backward",
168                     "etcetera", "europe", "northamerica", "southamerica");
169         }
170 
171         // find source directories to process
172         List<File> srcDirs = new ArrayList<File>();
173         if (version != null) {
174             File srcDir = new File(baseSrcDir, version);
175             if (srcDir.isDirectory() == false) {
176                 System.out.println("Version does not represent a valid source directory : " + srcDir);
177                 return;
178             }
179             srcDirs.add(srcDir);
180         } else {
181             File[] dirs = baseSrcDir.listFiles();
182             for (File dir : dirs) {
183                 if (dir.isDirectory() && dir.getName().matches("[12][0-9][0-9][0-9][A-Za-z0-9._-]+")) {
184                     srcDirs.add(dir);
185                 }
186             }
187         }
188         if (srcDirs.isEmpty()) {
189             System.out.println("Source directory contains no valid source folders: " + baseSrcDir);
190             return;
191         }
192 
193         // check destination directory
194         if (dstDir.exists() == false && dstDir.mkdirs() == false) {
195             System.out.println("Destination directory could not be created: " + dstDir);
196             return;
197         }
198         if (dstDir.isDirectory() == false) {
199             System.out.println("Destination is not a directory: " + dstDir);
200             return;
201         }
202         process(srcDirs, srcFileNames, dstDir, unpacked, verbose);
203     }

212         System.out.println("   -dstdir <directory>   Where to output generated files (default srcdir)");
213         System.out.println("   -version <version>    Specify the version, such as 2009a (optional)");

218         System.out.println(" Each directory must have the name of the version, such as 2009a");

225     /**
226      * Process to create the jar files.
227      */
228     private static void process(List<File> srcDirs, List<String> srcFileNames, File dstDir, boolean unpacked, boolean verbose) {
229         // build actual jar files
230         Map<Object, Object> deduplicateMap = new HashMap<Object, Object>();
231         Map<String, SortedMap<String, ZoneRules>> allBuiltZones = new TreeMap<String, SortedMap<String, ZoneRules>>();
232         Set<String> allRegionIds = new TreeSet<String>();
233         Set<ZoneRules> allRules = new HashSet<ZoneRules>();
234         SortedMap<LocalDate, Byte> bestLeapSeconds = null;
235 
236         for (File srcDir : srcDirs) {
237             // source files in this directory
238             List<File> srcFiles = new ArrayList<File>();
239             for (String srcFileName : srcFileNames) {
240                 File file = new File(srcDir, srcFileName);
241                 if (file.exists()) {
242                     srcFiles.add(file);
243                 }
244             }
245             if (srcFiles.isEmpty()) {
246                 continue;  // nothing to process
247             }
248             File leapSecondsFile = new File(srcDir, "leapseconds");
249             if (!leapSecondsFile.exists()) {
250                 System.out.println("Version " + srcDir.getName() + " does not include leap seconds information.");
251                 leapSecondsFile = null;
252             }
253 
254             // compile
255             String loopVersion = srcDir.getName();
256             TzdbZoneRulesCompiler compiler = new TzdbZoneRulesCompiler(loopVersion, srcFiles, leapSecondsFile, verbose);
257             compiler.setDeduplicateMap(deduplicateMap);
258             try {
259                 // compile
260                 compiler.compile();
261                 SortedMap<String, ZoneRules> builtZones = compiler.getZones();
262                 SortedMap<LocalDate, Byte> parsedLeapSeconds = compiler.getLeapSeconds();
263 
264                 // output version-specific file
265                 if (unpacked == false) {
266                     File dstFile = new File(dstDir, "threeten-TZDB-" + loopVersion + ".jar");
267                     if (verbose) {
268                         System.out.println("Outputting file: " + dstFile);
269                     }
270                     outputFile(dstFile, loopVersion, builtZones, parsedLeapSeconds);
271                 }
272 
273                 // create totals
274                 allBuiltZones.put(loopVersion, builtZones);
275                 allRegionIds.addAll(builtZones.keySet());
276                 allRules.addAll(builtZones.values());
277 
278                 // track best possible leap seconds collection
279                 if (compiler.getMostRecentLeapSecond() != null) {
280                     // we've got a live one!
281                     if (bestLeapSeconds == null || compiler.getMostRecentLeapSecond().compareTo(bestLeapSeconds.lastKey()) > 0) {
282                         // found the first one, or found a better one
283                         bestLeapSeconds = parsedLeapSeconds;
284                     }
285                 }
286             } catch (Exception ex) {
287                 System.out.println("Failed: " + ex.toString());
288                 ex.printStackTrace();
289                 System.exit(1);
290             }
291         }
292 
293         // output merged file
294         if (unpacked) {
295             if (verbose) {
296                 System.out.println("Outputting combined files: " + dstDir);
297             }
298             outputFilesDat(dstDir, allBuiltZones, allRegionIds, allRules, bestLeapSeconds);
299         } else {
300             File dstFile = new File(dstDir, "threeten-TZDB-all.jar");
301             if (verbose) {
302                 System.out.println("Outputting combined file: " + dstFile);
303             }
304             outputFile(dstFile, allBuiltZones, allRegionIds, allRules, bestLeapSeconds);
305         }
306     }

309      * Outputs the DAT files.

311     private static void outputFilesDat(File dstDir, Map<String, SortedMap<String, ZoneRules>> allBuiltZones,
312             Set<String> allRegionIds, Set<ZoneRules> allRules, SortedMap<LocalDate, Byte> leapSeconds) {
313         File tzdbFile = new File(dstDir, "TZDB.dat");
314         tzdbFile.delete();
315         try {
316             FileOutputStream fos = null;
317             try {
318                 fos = new FileOutputStream(tzdbFile);
319                 outputTzdbDat(fos, allBuiltZones, allRegionIds, allRules);
320             } finally {
321                 if (fos != null) {
322                     fos.close();
323                 }
324             }
325         } catch (Exception ex) {
326             System.out.println("Failed: " + ex.toString());
327             ex.printStackTrace();
328             System.exit(1);
329         }
330     }

335     private static void outputFile(File dstFile, String version, SortedMap<String, ZoneRules> builtZones, SortedMap<LocalDate, Byte> leapSeconds) {

338         Set<String> loopAllRegionIds = new TreeSet<String>(builtZones.keySet());

340         outputFile(dstFile, loopAllBuiltZones, loopAllRegionIds, loopAllRules, leapSeconds);

385     /**
386      * Outputs the timezone DAT file.
387      */
388     private static void outputTzdbDat(OutputStream jos,
389             Map<String, SortedMap<String, ZoneRules>> allBuiltZones,
390             Set<String> allRegionIds, Set<ZoneRules> allRules) throws IOException {
391         DataOutputStream out = new DataOutputStream(jos);
392 
393         // file version
394         out.writeByte(1);
395         // group
396         out.writeUTF("TZDB");
397         // versions
398         String[] versionArray = allBuiltZones.keySet().toArray(new String[allBuiltZones.size()]);
399         out.writeShort(versionArray.length);
400         for (String version : versionArray) {
401             out.writeUTF(version);
402         }
403         // regions
404         String[] regionArray = allRegionIds.toArray(new String[allRegionIds.size()]);
405         out.writeShort(regionArray.length);
406         for (String regionId : regionArray) {
407             out.writeUTF(regionId);
408         }
409         // rules
410         List<ZoneRules> rulesList = new ArrayList<ZoneRules>(allRules);
411         out.writeShort(rulesList.size());
412         ByteArrayOutputStream baos = new ByteArrayOutputStream(1024);
413         for (ZoneRules rules : rulesList) {
414             baos.reset();
415             DataOutputStream dataos = new DataOutputStream(baos);
416             Ser.write(rules, dataos);
417             dataos.close();
418             byte[] bytes = baos.toByteArray();
419             out.writeShort(bytes.length);
420             out.write(bytes);
421         }
422         // link version-region-rules
423         for (String version : allBuiltZones.keySet()) {
424             out.writeShort(allBuiltZones.get(version).size());
425             for (Map.Entry<String, ZoneRules> entry : allBuiltZones.get(version).entrySet()) {
426                  int regionIndex = Arrays.binarySearch(regionArray, entry.getKey());
427                  int rulesIndex = rulesList.indexOf(entry.getValue());
428                  out.writeShort(regionIndex);
429                  out.writeShort(rulesIndex);
430             }
431         }
432         out.flush();
433     }

536     /**
537      * Parses the leap seconds file.
538      *
539      * @throws Exception if an error occurs
540      */
541     private void parseLeapSecondsFile() throws Exception {
542         printVerbose("Parsing leap second file: " + leapSecondsFile);
543         int lineNumber = 1;
544         String line = null;
545         BufferedReader in = null;
546 
547         try {
548             in = new BufferedReader(new FileReader(leapSecondsFile));
549             for ( ; (line = in.readLine()) != null; lineNumber++) {
550                 int index = line.indexOf('#');  // remove comments (doesn't handle # in quotes)
551                 if (index >= 0) {
552                     line = line.substring(0, index);
553                 }
554                 if (line.trim().length() == 0) {  // ignore blank lines
555                     continue;
556                 }
557                 LeapSecondRule secondRule = parseLeapSecondRule(line);
558                 leapSeconds.put(secondRule.leapDate, secondRule.secondAdjustment);
559             }
560         } catch (Exception ex) {
561             throw new Exception("Failed while processing file '" + leapSecondsFile + "' on line " + lineNumber + " '" + line + "'", ex);
562         } finally {
563             try {
564                 if (in != null) {
565                     in.close();
566                 }
567             } catch (Exception ex) {
568                 // ignore NPE and IOE
569             }
570         }
571     }

573     private LeapSecondRule parseLeapSecondRule(String line) {
574         //    # Leap    YEAR    MONTH    DAY    HH:MM:SS    CORR    R/S
575         //    Leap    1972    Jun    30    23:59:60    +    S
576         //    Leap    1972    Dec    31    23:59:60    +    S
577         //    Leap    1973    Dec    31    23:59:60    +    S
578         //    Leap    1974    Dec    31    23:59:60    +    S
579         //    Leap    1975    Dec    31    23:59:60    +    S
580         //    Leap    1976    Dec    31    23:59:60    +    S
581         //    Leap    1977    Dec    31    23:59:60    +    S
582         //    Leap    1978    Dec    31    23:59:60    +    S
583         //    Leap    1979    Dec    31    23:59:60    +    S
584         //    Leap    1981    Jun    30    23:59:60    +    S
585         //    Leap    1982    Jun    30    23:59:60    +    S
586         //    Leap    1983    Jun    30    23:59:60    +    S
587 
588         StringTokenizer st = new StringTokenizer(line, " \t");
589         String first = st.nextToken();
590         if (first.equals("Leap")) {
591             if (st.countTokens() < 6) {
592                 printVerbose("Invalid leap second line in file: " + leapSecondsFile + ", line: " + line);
593                 throw new IllegalArgumentException("Invalid leap second line");
594             }
595         } else {
596             throw new IllegalArgumentException("Unknown line");
597         }
598 
599         int year = Integer.parseInt(st.nextToken());
600         Month month = parseMonth(st.nextToken());
601         int dayOfMonth = Integer.parseInt(st.nextToken());
602         LocalDate leapDate = LocalDate.of(year, month, dayOfMonth);
603         String timeOfLeapSecond = st.nextToken();
604 
605         byte adjustmentByte = 0;
606         String adjustment = st.nextToken();
607         if (adjustment.equals("+")) {
608             if (!("23:59:60".equals(timeOfLeapSecond))) {
609                 throw new IllegalArgumentException("Leap seconds can only be inserted at 23:59:60 - Date:" + leapDate);
610             }
611             adjustmentByte = +1;
612         } else if (adjustment.equals("-")) {
613             if (!("23:59:59".equals(timeOfLeapSecond))) {
614                 throw new IllegalArgumentException("Leap seconds can only be removed at 23:59:59 - Date:" + leapDate);
615             }
616             adjustmentByte = -1;
617         } else {
618             throw new IllegalArgumentException("Invalid adjustment '" + adjustment + "' in leap second rule for " + leapDate);
619         }
620 
621         String rollingOrStationary = st.nextToken();
622         if (!"S".equalsIgnoreCase(rollingOrStationary)) {
623             throw new IllegalArgumentException("Only stationary ('S') leap seconds are supported, not '" + rollingOrStationary + "'");
624         }
625         return new LeapSecondRule(leapDate, adjustmentByte);
626     }

628     /**
629      * Parses a source file.
630      *
631      * @param file  the file being read, not null
632      * @throws Exception if an error occurs
633      */
634     private void parseFile(File file) throws Exception {
635         int lineNumber = 1;
636         String line = null;
637         BufferedReader in = null;
638         try {
639             in = new BufferedReader(new FileReader(file));
640             List<TZDBZone> openZone = null;
641             for ( ; (line = in.readLine()) != null; lineNumber++) {
642                 int index = line.indexOf('#');  // remove comments (doesn't handle # in quotes)
643                 if (index >= 0) {
644                     line = line.substring(0, index);
645                 }
646                 if (line.trim().length() == 0) {  // ignore blank lines
647                     continue;
648                 }
649                 StringTokenizer st = new StringTokenizer(line, " \t");
650                 if (openZone != null && Character.isWhitespace(line.charAt(0)) && st.hasMoreTokens()) {
651                     if (parseZoneLine(st, openZone)) {
652                         openZone = null;
653                     }
654                 } else {
655                     if (st.hasMoreTokens()) {
656                         String first = st.nextToken();
657                         if (first.equals("Zone")) {
658                             if (st.countTokens() < 3) {
659                                 printVerbose("Invalid Zone line in file: " + file + ", line: " + line);
660                                 throw new IllegalArgumentException("Invalid Zone line");
661                             }
662                             openZone = new ArrayList<TZDBZone>();
663                             zones.put(st.nextToken(), openZone);
664                             if (parseZoneLine(st, openZone)) {
665                                 openZone = null;
666                             }
667                         } else {
668                             openZone = null;
669                             if (first.equals("Rule")) {
670                                 if (st.countTokens() < 9) {
671                                     printVerbose("Invalid Rule line in file: " + file + ", line: " + line);
672                                     throw new IllegalArgumentException("Invalid Rule line");
673                                 }
674                                 parseRuleLine(st);
675 
676                             } else if (first.equals("Link")) {
677                                 if (st.countTokens() < 2) {
678                                     printVerbose("Invalid Link line in file: " + file + ", line: " + line);
679                                     throw new IllegalArgumentException("Invalid Link line");
680                                 }
681                                 String realId = st.nextToken();
682                                 String aliasId = st.nextToken();
683                                 links.put(aliasId, realId);
684 
685                             } else {
686                                 throw new IllegalArgumentException("Unknown line");
687                             }
688                         }
689                     }
690                 }
691             }
692         } catch (Exception ex) {
693             throw new Exception("Failed while processing file '" + file + "' on line " + lineNumber + " '" + line + "'", ex);
694         } finally {
695             if (in != null) {
696                 in.close();
697             }
698         }
699     }

701     /**
702      * Parses a Rule line.
703      *
704      * @param st  the tokenizer, not null
705      */
706     private void parseRuleLine(StringTokenizer st) {
707         TZDBRule rule = new TZDBRule();
708         String name = st.nextToken();
709         if (rules.containsKey(name) == false) {
710             rules.put(name, new ArrayList<TZDBRule>());
711         }
712         rules.get(name).add(rule);
713         rule.startYear = parseYear(st.nextToken(), 0);
714         rule.endYear = parseYear(st.nextToken(), rule.startYear);
715         if (rule.startYear > rule.endYear) {
716             throw new IllegalArgumentException("Year order invalid: " + rule.startYear + " > " + rule.endYear);
717         }
718         parseOptional(st.nextToken());  // type is unused
719         parseMonthDayTime(st, rule);
720         rule.savingsAmount = parsePeriod(st.nextToken());
721         rule.text = parseOptional(st.nextToken());
722     }

724     /**
725      * Parses a Zone line.
726      *
727      * @param st  the tokenizer, not null
728      * @return true if the zone is complete
729      */
730     private boolean parseZoneLine(StringTokenizer st, List<TZDBZone> zoneList) {
731         TZDBZone zone = new TZDBZone();
732         zoneList.add(zone);
733         zone.standardOffset = parseOffset(st.nextToken());
734         String savingsRule = parseOptional(st.nextToken());
735         if (savingsRule == null) {
736             zone.fixedSavingsSecs = 0;
737             zone.savingsRule = null;
738         } else {
739             try {
740                 zone.fixedSavingsSecs = parsePeriod(savingsRule);
741                 zone.savingsRule = null;
742             } catch (Exception ex) {
743                 zone.fixedSavingsSecs = null;
744                 zone.savingsRule = savingsRule;
745             }
746         }
747         zone.text = st.nextToken();
748         if (st.hasMoreTokens()) {
749             zone.year = Year.of(Integer.parseInt(st.nextToken()));
750             if (st.hasMoreTokens()) {
751                 parseMonthDayTime(st, zone);
752             }
753             return false;
754         } else {
755             return true;
756         }
757     }

759     /**
760      * Parses a Rule line.
761      *
762      * @param st  the tokenizer, not null
763      * @param mdt  the object to parse into, not null
764      */
765     private void parseMonthDayTime(StringTokenizer st, TZDBMonthDayTime mdt) {
766         mdt.month = parseMonth(st.nextToken());
767         if (st.hasMoreTokens()) {
768             String dayRule = st.nextToken();
769             if (dayRule.startsWith("last")) {
770                 mdt.dayOfMonth = -1;
771                 mdt.dayOfWeek = parseDayOfWeek(dayRule.substring(4));
772                 mdt.adjustForwards = false;
773             } else {
774                 int index = dayRule.indexOf(">=");
775                 if (index > 0) {
776                     mdt.dayOfWeek = parseDayOfWeek(dayRule.substring(0, index));
777                     dayRule = dayRule.substring(index + 2);
778                 } else {
779                     index = dayRule.indexOf("<=");
780                     if (index > 0) {
781                         mdt.dayOfWeek = parseDayOfWeek(dayRule.substring(0, index));
782                         mdt.adjustForwards = false;
783                         dayRule = dayRule.substring(index + 2);
784                     }
785                 }
786                 mdt.dayOfMonth = Integer.parseInt(dayRule);
787             }
788             if (st.hasMoreTokens()) {
789                 String timeStr = st.nextToken();
790                 int secsOfDay = parseSecs(timeStr);
791                 if (secsOfDay == 86400) {
792                     mdt.endOfDay = true;
793                     secsOfDay = 0;
794                 }
795                 LocalTime time = deduplicate(LocalTime.ofSecondOfDay(secsOfDay));
796                 mdt.time = time;
797                 mdt.timeDefinition = parseTimeDefinition(timeStr.charAt(timeStr.length() - 1));
798             }
799         }
800     }

802     private int parseYear(String str, int defaultYear) {
803         str = str.toLowerCase();
804         if (matches(str, "minimum")) {
805             return Year.MIN_VALUE;
806         } else if (matches(str, "maximum")) {
807             return Year.MAX_VALUE;
808         } else if (str.equals("only")) {
809             return defaultYear;
810         }
811         return Integer.parseInt(str);
812     }

814     private Month parseMonth(String str) {
815         str = str.toLowerCase();
816         for (Month moy : Month.values()) {
817             if (matches(str, moy.name().toLowerCase())) {
818                 return moy;
819             }
820         }
821         throw new IllegalArgumentException("Unknown month: " + str);
822     }

824     private DayOfWeek parseDayOfWeek(String str) {
825         str = str.toLowerCase();
826         for (DayOfWeek dow : DayOfWeek.values()) {
827             if (matches(str, dow.name().toLowerCase())) {
828                 return dow;
829             }
830         }
831         throw new IllegalArgumentException("Unknown day-of-week: " + str);
832     }

834     private boolean matches(String str, String search) {
835         return str.startsWith(search.substring(0, 3)) && search.startsWith(str) && str.length() <= search.length();
836     }

838     private String parseOptional(String str) {
839         return str.equals("-") ? null : str;
840     }
841 
842     private int parseSecs(String str) {
843         if (str.equals("-")) {
844             return 0;
845         }
846         int pos = 0;
847         if (str.startsWith("-")) {
848             pos = 1;
849         }
850         ParsePosition pp = new ParsePosition(pos);
851         TemporalAccessor parsed = TIME_PARSER.parseUnresolved(str, pp);
852         if (parsed == null || pp.getErrorIndex() >= 0) {
853             throw new IllegalArgumentException(str);
854         }
855         long hour = parsed.getLong(HOUR_OF_DAY);
856         Long min = (parsed.isSupported(MINUTE_OF_HOUR) ? parsed.getLong(MINUTE_OF_HOUR) : null);
857         Long sec = (parsed.isSupported(SECOND_OF_MINUTE) ? parsed.getLong(SECOND_OF_MINUTE) : null);
858         int secs = (int) (hour * 60 * 60 + (min != null ? min : 0) * 60 + (sec != null ? sec : 0));
859         if (pos == 1) {
860             secs = -secs;
861         }
862         return secs;
863     }
864 
865     private ZoneOffset parseOffset(String str) {
866         int secs = parseSecs(str);
867         return ZoneOffset.ofTotalSeconds(secs);
868     }
869 
870     private int parsePeriod(String str) {
871         return parseSecs(str);
872     }

874     private TimeDefinition parseTimeDefinition(char c) {
875         switch (c) {
876             case 's':
877             case 'S':
878                 // standard time
879                 return TimeDefinition.STANDARD;
880             case 'u':
881             case 'U':
882             case 'g':
883             case 'G':
884             case 'z':
885             case 'Z':
886                 // UTC
887                 return TimeDefinition.UTC;
888             case 'w':
889             case 'W':
890             default:
891                 // wall time
892                 return TimeDefinition.WALL;
893         }
894     }

896     //-----------------------------------------------------------------------
897     /**
898      * Build the rules, zones and links into real zones.
899      *
900      * @throws Exception if an error occurs
901      */
902     private void buildZoneRules() throws Exception {
903         // build zones
904         for (String zoneId : zones.keySet()) {
905             printVerbose("Building zone " + zoneId);
906             zoneId = deduplicate(zoneId);
907             List<TZDBZone> tzdbZones = zones.get(zoneId);
908             ZoneRulesBuilder bld = new ZoneRulesBuilder();
909             for (TZDBZone tzdbZone : tzdbZones) {
910                 bld = tzdbZone.addToBuilder(bld, rules);
911             }
912             ZoneRules buildRules = bld.toRules(zoneId, deduplicateMap);
913             builtZones.put(zoneId, deduplicate(buildRules));
914         }
915 
916         // build aliases
917         for (String aliasId : links.keySet()) {
918             aliasId = deduplicate(aliasId);
919             String realId = links.get(aliasId);
920             printVerbose("Linking alias " + aliasId + " to " + realId);
921             ZoneRules realRules = builtZones.get(realId);
922             if (realRules == null) {
923                 realId = links.get(realId);  // try again (handle alias liked to alias)
924                 printVerbose("Relinking alias " + aliasId + " to " + realId);
925                 realRules = builtZones.get(realId);
926                 if (realRules == null) {
927                     throw new IllegalArgumentException("Alias '" + aliasId + "' links to invalid zone '" + realId + "' for '" + version + "'");
928                 }
929             }
930             builtZones.put(aliasId, realRules);
931         }
932 
933         // remove UTC and GMT
934         builtZones.remove("UTC");
935         builtZones.remove("GMT");
936         builtZones.remove("GMT0");
937         builtZones.remove("GMT+0");
938         builtZones.remove("GMT-0");
939     }

944      *
945      * @param <T> the generic type
946      * @param object  the object to deduplicate
947      * @return the deduplicated object

970     /**
971      * Class representing a month-day-time in the TZDB file.
972      */
973     abstract class TZDBMonthDayTime {
974         /** The month of the cutover. */
975         Month month = Month.JANUARY;
976         /** The day-of-month of the cutover. */
977         int dayOfMonth = 1;
978         /** Whether to adjust forwards. */
979         boolean adjustForwards = true;
980         /** The day-of-week of the cutover. */
981         DayOfWeek dayOfWeek;
982         /** The time of the cutover. */
983         LocalTime time = LocalTime.MIDNIGHT;
984         /** Whether this is midnight end of day. */
985         boolean endOfDay;
986         /** The time of the cutover. */
987         TimeDefinition timeDefinition = TimeDefinition.WALL;
988 
989         void adjustToFowards(int year) {
990             if (adjustForwards == false && dayOfMonth > 0) {
991                 LocalDate adjustedDate = LocalDate.of(year, month, dayOfMonth).minusDays(6);
992                 dayOfMonth = adjustedDate.getDayOfMonth();
993                 month = adjustedDate.getMonth();
994                 adjustForwards = true;
995             }
996         }
997     }

1020     /**
1021      * Class representing a linked set of zone lines in the TZDB file.
1022      */
1023     final class TZDBZone extends TZDBMonthDayTime {
1024         /** The standard offset. */
1025         ZoneOffset standardOffset;
1026         /** The fixed savings amount. */
1027         Integer fixedSavingsSecs;
1028         /** The savings rule. */
1029         String savingsRule;
1030         /** The text name of the zone. */
1031         String text;
1032         /** The year of the cutover. */
1033         Year year;
1034 
1035         ZoneRulesBuilder addToBuilder(ZoneRulesBuilder bld, Map<String, List<TZDBRule>> rules) {
1036             if (year != null) {
1037                 bld.addWindow(standardOffset, toDateTime(year.getValue()), timeDefinition);
1038             } else {
1039                 bld.addWindowForever(standardOffset);
1040             }
1041 
1042             if (fixedSavingsSecs != null) {
1043                 bld.setFixedSavingsToWindow(fixedSavingsSecs);
1044             } else {
1045                 List<TZDBRule> tzdbRules = rules.get(savingsRule);
1046                 if (tzdbRules == null) {
1047                     throw new IllegalArgumentException("Rule not found: " + savingsRule);
1048                 }
1049                 for (TZDBRule tzdbRule : tzdbRules) {
1050                     tzdbRule.addToBuilder(bld);
1051                 }
1052             }
1053 
1054             return bld;
1055         }
1056 
1057         private LocalDateTime toDateTime(int year) {
1058             adjustToFowards(year);
1059             LocalDate date;
1060             if (dayOfMonth == -1) {
1061                 dayOfMonth = month.length(Year.isLeap(year));
1062                 date = LocalDate.of(year, month, dayOfMonth);
1063                 if (dayOfWeek != null) {
1064                     date = date.with(TemporalAdjusters.previousOrSame(dayOfWeek));
1065                 }
1066             } else {
1067                 date = LocalDate.of(year, month, dayOfMonth);
1068                 if (dayOfWeek != null) {
1069                     date = date.with(TemporalAdjusters.nextOrSame(dayOfWeek));
1070                 }
1071             }
1072             date = deduplicate(date);
1073             LocalDateTime ldt = LocalDateTime.of(date, time);
1074             if (endOfDay) {
1075                 ldt = ldt.plusDays(1);
1076             }
1077             return ldt;
1078         }
