51  /**
52   * A date in the Minguo calendar system.
53   * <p>
54   * This implements {@code ChronoLocalDate} for the {@link MinguoChronology Minguo calendar}.
55   *
56   * <h3>Specification for implementors</h3>
57   * This class is immutable and thread-safe.
58   */
59  public final class MinguoDate
60          extends ChronoDateImpl<MinguoDate>
61          implements Serializable {
62      // this class is package-scoped so that future conversion to public
63      // would not change serialization
64  
65      /**
66       * Serialization version.
67       */
68      private static final long serialVersionUID = 1300372329181994526L;
69  
70      /**
71       * The underlying date.
72       */
73      private final LocalDate isoDate;
74  
75      /**
76       * Creates an instance from an ISO date.
77       *
78       * @param isoDate  the standard local date, validated not null
79       */
80      MinguoDate(LocalDate date) {
81          Objects.requireNonNull(date, "date");
82          this.isoDate = date;
83      }
84  
85      //-----------------------------------------------------------------------
86      @Override
87      public MinguoChronology getChronology() {
88          return MinguoChronology.INSTANCE;
89      }
90  
91      @Override
92      public int lengthOfMonth() {
93          return isoDate.lengthOfMonth();
94      }
95  
96      @Override
97      public ValueRange range(TemporalField field) {
98          if (field instanceof ChronoField) {
99              if (isSupported(field)) {
100                 ChronoField f = (ChronoField) field;
101                 switch (f) {
102                     case DAY_OF_MONTH:
103                     case DAY_OF_YEAR:
104                     case ALIGNED_WEEK_OF_MONTH:
105                         return isoDate.range(field);
106                     case YEAR_OF_ERA: {
107                         ValueRange range = YEAR.range();
108                         long max = (getProlepticYear() <= 0 ? -range.getMinimum() + 1 + YEARS_DIFFERENCE : range.getMaximum() - YEARS_DIFFERENCE);
109                         return ValueRange.of(1, max);
110                     }
111                 }
112                 return getChronology().range(f);
113             }
114             throw new DateTimeException("Unsupported field: " + field.getName());
115         }
116         return field.rangeRefinedBy(this);
117     }
118 
119     @Override
120     public long getLong(TemporalField field) {
121         if (field instanceof ChronoField) {
122             switch ((ChronoField) field) {
123                 case YEAR_OF_ERA: {
124                     int prolepticYear = getProlepticYear();
125                     return (prolepticYear >= 1 ? prolepticYear : 1 - prolepticYear);
126                 }
127                 case YEAR:
128                     return getProlepticYear();
129                 case ERA:
130                     return (getProlepticYear() >= 1 ? 1 : 0);
131             }
132             return isoDate.getLong(field);
133         }
134         return field.getFrom(this);
135     }
136 
137     private int getProlepticYear() {
138         return isoDate.getYear() - YEARS_DIFFERENCE;
139     }
140 
141     //-----------------------------------------------------------------------
142     @Override
143     public MinguoDate with(TemporalField field, long newValue) {
144         if (field instanceof ChronoField) {
145             ChronoField f = (ChronoField) field;
146             if (getLong(f) == newValue) {
147                 return this;
148             }
149             switch (f) {
150                 case YEAR_OF_ERA:
151                 case YEAR:
152                 case ERA: {
153                     f.checkValidValue(newValue);
154                     int nvalue = (int) newValue;
155                     switch (f) {
156                         case YEAR_OF_ERA:
157                             return with(isoDate.withYear(getProlepticYear() >= 1 ? nvalue + YEARS_DIFFERENCE : (1 - nvalue)  + YEARS_DIFFERENCE));
158                         case YEAR:
159                             return with(isoDate.withYear(nvalue + YEARS_DIFFERENCE));
160                         case ERA:
161                             return with(isoDate.withYear((1 - getProlepticYear()) + YEARS_DIFFERENCE));
162                     }
163                 }
164             }
165             return with(isoDate.with(field, newValue));
166         }
167         return field.adjustInto(this, newValue);
168     }
169 
170     //-----------------------------------------------------------------------
171     @Override
172     MinguoDate plusYears(long years) {
173         return with(isoDate.plusYears(years));
174     }
175 
176     @Override
177     MinguoDate plusMonths(long months) {
178         return with(isoDate.plusMonths(months));
179     }
180 
181     @Override
182     MinguoDate plusDays(long days) {
183         return with(isoDate.plusDays(days));
184     }
185 
186     private MinguoDate with(LocalDate newDate) {
187         return (newDate.equals(isoDate) ? this : new MinguoDate(newDate));
188     }
189 
190     @Override  // override for performance
191     public long toEpochDay() {
192         return isoDate.toEpochDay();
193     }
194 
195     //-------------------------------------------------------------------------
196     @Override  // override for performance
197     public boolean equals(Object obj) {
198         if (this == obj) {
199             return true;
200         }
201         if (obj instanceof MinguoDate) {
202             MinguoDate otherDate = (MinguoDate) obj;
203             return this.isoDate.equals(otherDate.isoDate);
204         }
205         return false;
206     }
207 
208     @Override  // override for performance
209     public int hashCode() {
210         return getChronology().getId().hashCode() ^ isoDate.hashCode();
211     }

213     //-----------------------------------------------------------------------
214     private Object writeReplace() {
215         return new Ser(Ser.MINGUO_DATE_TYPE, this);
216     }

218     void writeExternal(DataOutput out) throws IOException {
219         // MinguoChrono is implicit in the MINGUO_DATE_TYPE
220         out.writeInt(get(YEAR));
221         out.writeByte(get(MONTH_OF_YEAR));
222         out.writeByte(get(DAY_OF_MONTH));
223 
224     }

226     static ChronoLocalDate<?> readExternal(DataInput in) throws IOException {
227         int year = in.readInt();
228         int month = in.readByte();
229         int dayOfMonth = in.readByte();
230         return MinguoChronology.INSTANCE.date(year, month, dayOfMonth);
231     }
