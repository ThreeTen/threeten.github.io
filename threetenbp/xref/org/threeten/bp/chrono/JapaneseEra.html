51  /**
52   * An era in the Japanese Imperial calendar system.
53   * <p>
54   * This class defines the valid eras for the Japanese chronology.
55   * Japan introduced the Gregorian calendar starting with Meiji 6.
56   * Only Meiji and later eras are supported;
57   * dates before Meiji 6, January 1 are not supported.
58   * <p>
59   * The four supported eras are hard-coded.
60   * A single additional era may be registered using {@link #registerEra(LocalDate, String)}.
61   *
62   * <h3>Specification for implementors</h3>
63   * This class is immutable and thread-safe.
64   */
65  public final class JapaneseEra
66          extends DefaultInterfaceEra
67          implements Serializable {
68  
69      // The offset value to 0-based index from the era value.
70      // i.e., getValue() + ERA_OFFSET == 0-based index; except that -999 is mapped to zero
71      static final int ERA_OFFSET = 2;
72  
73      /**
74       * The singleton instance for the 'Meiji' era (1868-09-08 - 1912-07-29)
75       * which has the value -1.
76       */
77      public static final JapaneseEra MEIJI = new JapaneseEra(-1, LocalDate.of(1868, 9, 8), "Meiji");
78      /**
79       * The singleton instance for the 'Taisho' era (1912-07-30 - 1926-12-24)
80       * which has the value 0.
81       */
82      public static final JapaneseEra TAISHO = new JapaneseEra(0, LocalDate.of(1912, 7, 30), "Taisho");
83      /**
84       * The singleton instance for the 'Showa' era (1926-12-25 - 1989-01-07)
85       * which has the value 1.
86       */
87      public static final JapaneseEra SHOWA = new JapaneseEra(1, LocalDate.of(1926, 12, 25), "Showa");
88      /**
89       * The singleton instance for the 'Heisei' era (1989-01-08 - current)
90       * which has the value 2.
91       */
92      public static final JapaneseEra HEISEI = new JapaneseEra(2, LocalDate.of(1989, 1, 8), "Heisei");
93      /**
94       * The value of the additional era.
95       */
96      private static final int ADDITIONAL_VALUE = 3;
97  
98      /**
99       * Serialization version.
100      */
101     private static final long serialVersionUID = 1466499369062886794L;
102 
103     // array for the singleton JapaneseEra instances
104     private static final AtomicReference<JapaneseEra[]> KNOWN_ERAS;
105 
106     static {
107         JapaneseEra[] array = new JapaneseEra[4];
108         array[0] = MEIJI;
109         array[1] = TAISHO;
110         array[2] = SHOWA;
111         array[3] = HEISEI;
112         KNOWN_ERAS = new AtomicReference<JapaneseEra[]>(array);
113     }
114 
115     /**
116      * The era value.
117      * @serial
118      */
119     private final int eraValue;
120 
121     // the first day of the era
122     private final transient LocalDate since;
123     // the name of the era
124     private final transient String name;
125 
126     /**
127      * Creates an instance.
128      *
129      * @param eraValue  the era value, validated
130      * @param since  the date representing the first date of the era, validated not null
131      * @param name  the name
132      */
133     private JapaneseEra(int eraValue, LocalDate since, String name) {
134         this.eraValue = eraValue;
135         this.since = since;
136         this.name = name;
137     }
138 
139     /**
140      * Returns the singleton {@code JapaneseEra} corresponding to this object.
141      * It's possible that this version of {@code JapaneseEra} doesn't support the latest era value.
142      * In that case, this method throws an {@code ObjectStreamException}.
143      *
144      * @return the singleton {@code JapaneseEra} for this object
145      * @throws ObjectStreamException if the deserialized object has any unknown numeric era value.
146      */
147     private Object readResolve() throws ObjectStreamException {
148         try {
149             return of(eraValue);
150         } catch (DateTimeException e) {
151             InvalidObjectException ex = new InvalidObjectException("Invalid era");
152             ex.initCause(e);
153             throw ex;
154         }
155     }
156 
157     //-----------------------------------------------------------------------
158     /**
159      * Registers an additional instance of {@code JapaneseEra}.
160      * <p>
161      * A new Japanese era can begin at any time.
162      * This method allows one new era to be registered without the need for a new library version.
163      * If needed, callers should assign the result to a static variable accessible
164      * across the application. This must be done once, in early startup code.
165      * <p>
166      * NOTE: This method does not exist in Java SE 8.
167      *
168      * @param since  the date representing the first date of the era, validated not null
169      * @param name  the name
170      * @return the {@code JapaneseEra} singleton, not null
171      * @throws DateTimeException if an additional era has already been registered
172      */
173     public static JapaneseEra registerEra(LocalDate since, String name) {
174         JapaneseEra[] known = KNOWN_ERAS.get();
175         if (known.length > 4) {
176             throw new DateTimeException("Only one additional Japanese era can be added");
177         }
178         Jdk8Methods.requireNonNull(since, "since");
179         Jdk8Methods.requireNonNull(name, "name");
180         if (!since.isAfter(HEISEI.since)) {
181             throw new DateTimeException("Invalid since date for additional Japanese era, must be after Heisei");
182         }
183         JapaneseEra era = new JapaneseEra(ADDITIONAL_VALUE, since, name);
184         JapaneseEra[] newArray = Arrays.copyOf(known, 5);
185         newArray[4] = era;
186         if (!KNOWN_ERAS.compareAndSet(known, newArray)) {
187             throw new DateTimeException("Only one additional Japanese era can be added");
188         }
189         return era;
190     }
191 
192     /**
193      * Obtains an instance of {@code JapaneseEra} from an {@code int} value.
194      * <p>
195      * The {@link #SHOWA} era that contains 1970-01-01 (ISO calendar system) has the value 1
196      * Later era is numbered 2 ({@link #HEISEI}). Earlier eras are numbered 0 ({@link #TAISHO}),
197      * -1 ({@link #MEIJI}), only Meiji and later eras are supported.
198      *
199      * @param japaneseEra  the era to represent
200      * @return the {@code JapaneseEra} singleton, not null
201      * @throws DateTimeException if the value is invalid
202      */
203     public static JapaneseEra of(int japaneseEra) {
204         JapaneseEra[] known = KNOWN_ERAS.get();
205         if (japaneseEra < MEIJI.eraValue || japaneseEra > known[known.length - 1].eraValue) {
206             throw new DateTimeException("japaneseEra is invalid");
207         }
208         return known[ordinal(japaneseEra)];
209     }
210 
211     /**
212      * Returns the {@code JapaneseEra} with the name.
213      * <p>
214      * The string must match exactly the name of the era.
215      * (Extraneous whitespace characters are not permitted.)
216      *
217      * @param japaneseEra  the japaneseEra name; non-null
218      * @return the {@code JapaneseEra} singleton, never null
219      * @throws IllegalArgumentException if there is not JapaneseEra with the specified name
220      */
221     public static JapaneseEra valueOf(String japaneseEra) {
222         Jdk8Methods.requireNonNull(japaneseEra, "japaneseEra");
223         JapaneseEra[] known = KNOWN_ERAS.get();
224         for (JapaneseEra era : known) {
225             if (japaneseEra.equals(era.name)) {
226                 return era;
227             }
228         }
229         throw new IllegalArgumentException("Era not found: " + japaneseEra);
230     }
231 
232     /**
233      * Returns an array of JapaneseEras.
234      * <p>
235      * This method may be used to iterate over the JapaneseEras as follows:
236      * <pre>
237      * for (JapaneseEra c : JapaneseEra.values())
238      *     System.out.println(c);
239      * </pre>
240      *
241      * @return an array of JapaneseEras
242      */
243     public static JapaneseEra[] values() {
244         JapaneseEra[] known = KNOWN_ERAS.get();
245         return Arrays.copyOf(known, known.length);
246     }
247 
248     //-----------------------------------------------------------------------
249     /**
250      * Obtains an instance of {@code JapaneseEra} from a date.
251      *
252      * @param date  the date, not null
253      * @return the Era singleton, never null
254      */
255     static JapaneseEra from(LocalDate date) {
256         if (date.isBefore(MEIJI.since)) {
257             throw new DateTimeException("Date too early: " + date);
258         }
259         JapaneseEra[] known = KNOWN_ERAS.get();
260         for (int i = known.length - 1; i >= 0; i--) {
261             JapaneseEra era = known[i];
262             if (date.compareTo(era.since) >= 0) {
263                 return era;
264             }
265         }
266         return null;
267     }
268 
269     /**
270      * Returns the index into the arrays from the Era value.
271      * the eraValue is a valid Era number, -999, -1..2.
272      * @param eraValue the era value to convert to the index
273      * @return the index of the current Era
274      */
275     private static int ordinal(int eraValue) {
276         return eraValue + 1;
277     }
278 
279     /**
280      * Returns the start date of the era.
281      * @return the start date
282      */
283     LocalDate startDate() {
284         return since;
285     }
286 
287     /**
288      * Returns the end date of the era.
289      * @return the end date
290      */
291     LocalDate endDate() {
292         int ordinal = ordinal(eraValue);
293         JapaneseEra[] eras = values();
294         if (ordinal >= eras.length - 1) {
295             return LocalDate.MAX;
296         }
297         return eras[ordinal + 1].startDate().minusDays(1);
298     }
299 
300     //-----------------------------------------------------------------------
301     /**
302      * Returns the numeric value of this {@code JapaneseEra}.
303      * <p>
304      * The {@link #SHOWA} era that contains 1970-01-01 (ISO calendar system) has the value 1.
305      * Later eras are numbered from 2 ({@link #HEISEI}).
306      * Earlier eras are numbered 0 ({@link #TAISHO}) and -1 ({@link #MEIJI}).
307      *
308      * @return the era value
309      */
310     @Override
311     public int getValue() {
312         return eraValue;
313     }
314 
315     @Override
316     public ValueRange range(TemporalField field) {
317         if (field == ChronoField.ERA) {
318             return JapaneseChronology.INSTANCE.range(ChronoField.ERA);
319         }
320         return super.range(field);
321     }
322 
323     //-----------------------------------------------------------------------
324     @Override
325     public String toString() {
326         return name;
327     }
328 
329     //-----------------------------------------------------------------------
330     private Object writeReplace() {
331         return new Ser(Ser.JAPANESE_ERA_TYPE, this);
332     }
333 
334     void writeExternal(DataOutput out) throws IOException {
335         out.writeByte(this.getValue());
336     }
337 
338     static JapaneseEra readExternal(DataInput in) throws IOException {
339         byte eraValue = in.readByte();
340         return JapaneseEra.of(eraValue);
341     }
